CREATE VIEW [dbo].[bv_StudentTranscripts] AS 
    
/*****************************************************************************************
  Name:		    bv_StudentTranscripts

  Description:  Student Transcripts

  Created:	    09/25/2009 by John Holmes

  Modified:     01/11/2010 by John Holmes
                Remove course Prefix and Suffix columns.
                Modify Course column to contain full course number.
                
                09/06/2011 by John Holmes
                v1.1.0 - Source object renaming in support of ERP (Oracle) Warehouse.
                
*****************************************************************************************/ 
    
SELECT  
    [crec].[UF_UFID] AS [UFID], 
    [crec].[WCBA_TERM_CD6] AS [Term6], 
    [crec].[UF_COMM_TYPE] AS [CourseType], 
    [crec].[UF_COURSE_CD] AS [Course],  
    [crec].[UF_SECTION] AS [Section], 
    [crec].[UF_GORD_RULE] AS [GordonRule], 
    [crec].[UF_GRADE] AS [Grade], 
    [crec].[UF_CREDIT] AS [Credit],
    [crec].[UF_GRADE_VALUE] AS [Multiplier],
    (CASE 
	    WHEN [crec].[UF_RPT_NOCR] = 'X' THEN '-128' 
	    WHEN [crec].[UF_NOT_APPD_DEG] = 'X' THEN '16' 
	    WHEN [crec].[UF_CR_BEL_ACLVL] = 'X' THEN '32' 
	    WHEN [crec].[UF_GRD_FORGIVEN] = 'X' THEN '64' 
	    WHEN [crec].[UF_XFER_RPT_NOCR] = 'X' THEN 
			(CASE 
				WHEN [crec].[UF_4_YR_COL_CSE] = 'X' THEN '3' 
				WHEN [crec].[UF_JC_COL_CRS] = 'X' THEN '5' 
				ELSE '1' 
			END) 
	    WHEN [crec].[UF_4_YR_COL_CSE] = 'X' THEN 
			(CASE 
				WHEN [crec].[UF_RPT_NOCR] = 'X' THEN '-126' 
				WHEN [crec].[UF_VOC_CRS_NOCR] = 'X' THEN '10' 
				WHEN [crec].[UF_NOT_APPD_DEG] = 'X' THEN '18' 
				WHEN [crec].[UF_CR_BEL_ACLVL] = 'X' THEN '34'
				ELSE '2'
			END) 
	    WHEN [crec].[UF_JC_COL_CRS] = 'X' THEN 
			(CASE 
				WHEN [crec].[UF_RPT_NOCR] = 'X' THEN '-124' 
				WHEN [crec].[UF_VOC_CRS_NOCR] = 'X' THEN '12'
				WHEN [crec].[UF_NOT_APPD_DEG] = 'X' THEN '20' 
				WHEN [crec].[UF_CR_BEL_ACLVL] = 'X' THEN '36' 
				WHEN [crec].[UF_GRD_FORGIVEN] = 'X' THEN '68' 
				ELSE '4' 
			END) 
	    ELSE '0' 
    END) AS [Repeat],  
    (CASE 
	    WHEN [crec].[UF_GRADE_CHG] = 'X' THEN 'C'
	    WHEN [crec].[UF_ALW_GRAD_CHG] = 'X' THEN 'H'
	    WHEN [crec].[UF_NO_CREDIT] = 'X' THEN 'N'
	    WHEN [crec].[UF_NO_GPA] = 'X' THEN 'O'
	    WHEN [crec].[UF_ALW_RPT_CR] = 'X' THEN 'X' 
	    ELSE '' 
    END) AS [AuthorizationCode], 
    (CASE 
	    WHEN [term].[UF_CLASS] = 'X' OR [term].[UF_CLASS] IS NULL THEN '' 
	    ELSE [term].[UF_CLASS] END) + 
    (CASE 
	    WHEN [term].[UF_COLLEGE_CD] IN ('X', 'XX', 'X,', 'X.') OR [term].[UF_COLLEGE_CD] IS NULL THEN '' 
	    ELSE [term].[UF_COLLEGE_CD] 
    END) AS [CourseClassCollege], 
    [crec].[UF_COURSE_NAME] AS [CourseTitle], 
    [crec].[UF_HRS_EARNED] AS [HoursEarned], 
    [crec].[UF_HRS_CARRIED] AS [HoursCarried], 
    [term].[UF_GPA] AS [UFTermGPA], 
    [term].[UF_UF_CUMM_GPA] AS [UFCummulativeGPA], 
    [crec].[UF_AUTO_INDEX] AS [AutoIndex], 
    [crec].[LOAD_DTTM] AS [LoadDate], 
    [term].[UF_ABC] AS [TermABCFlag], 
    [term].[UF_GRADE_PTS] AS [TermGradePoints] 
FROM
    [BaseData].[bd].[bt_CREC_COURSE] AS [crec] 
LEFT OUTER JOIN	
    [BaseData].[bd].[bt_CREC_TERM] AS [term]
    ON [crec].[UF_UFID] = [term].[UF_UFID] 
    AND [crec].[WCBA_TERM_CD6] = [term].[WCBA_TERM_CD6] 
    AND [crec].[UF_COMM_TYPE] = [term].[UF_COMM_TYPE]  
    
    
    
UNION ALL 


-- Add the Beta Gamma Sigma records
SELECT
    [rem].[UF_UFID] AS [UFID], 
    [rem].[WCBA_TERM_CD6] AS [Term6],
    [rem].[UF_COMM_TYPE] AS [CourseType], 
    [rem].[UF_COMM_IND] + [rem].[UF_COMM_CD] AS [Course], 
    '0000' AS [Section], 
    '' AS [GordonRule], 
    '' AS [Grade], 
    '0.00' AS [Credit], 
    '0.0' AS [Multiplier], 
    '' AS [Repeat], 
    '' AS [AuthorizationCode], 
    (CASE 
	    WHEN [term].[UF_CLASS] = 'X' OR [term].[UF_CLASS] IS NULL THEN '' 
	    ELSE [term].[UF_CLASS] END) + 
    (CASE 
	    WHEN [term].[UF_COLLEGE_CD] IN ('X', 'XX', 'X,', 'X.') OR [term].[UF_COLLEGE_CD] IS NULL THEN '' 
	    ELSE [term].[UF_COLLEGE_CD] 
    END) AS [CourseClassCollege], 
    [rem].[UF_REMARK_TEXT] AS [CourseTitle], 
    '0.00' AS [HoursEarned], 
    '0.00' AS [HoursCarried], 
    [term].[UF_GPA] AS [UFTermGPA], 
    [term].[UF_UF_CUMM_GPA] AS [UFCummulativeGPA], 
    [rem].[UF_AUTO_INDEX] AS [AutoIndex], 
    [rem].[LOAD_DTTM] AS [LoadDate], 
    '' AS [TermABCFlag], 
    NULL AS [TermGradePoints] 
FROM 
    [BaseData].[bd].[bt_CREC_REMARK] AS [rem] 
LEFT OUTER JOIN 
    [BaseData].[bd].[bt_CREC_TERM] AS [term] 
    ON [rem].[UF_UFID] = [term].[UF_UFID] 
    AND [rem].[WCBA_TERM_CD6] = [term].[WCBA_TERM_CD6] 
    AND [rem].[UF_COMM_TYPE] = [term].[UF_COMM_TYPE]  
WHERE
    [rem].[UF_COMM_IND] = '111' 
    AND [rem].[UF_COMM_CD] = '0333'
