CREATE VIEW [dbo].[bv_StudentRegistration_cs] AS 
    
/*****************************************************************************************
  Name:		    bv_StudentRegistration_cs

  Description:  Student Registration (previous, current and future terms)
				
				This is a compatibility view using CS data

  Created:	    04/10/2018 by John Holmes

  Modified:     
                
*****************************************************************************************/ 
    
SELECT
	reg.EMPLID AS UFID,
	reg.CLASS_SECTION_CD AS Section,
	CAST(reg.ERN_UNIT AS decimal(2,0)) AS CreditHours,
	g.GRADE_CD AS Grade,
	'' AS StatusFlag,
	st.UF_ABC AS SummerFlag,
	reg.TERM6 AS Term6,
	CASE 
		WHEN er.ENRLMT_RSN_SID IN (7,8) THEN 'A'
		WHEN er.ENRLMT_RSN_SID IN (3,5,16) THEN 'D'
		ELSE NULL
	END AS DropAddFlag,
	COALESCE(reg.LAST_DROP_DTTM_STMP, reg.LAST_ENRL_DTTM_STMP) AS DropAddCode,
	NULL AS LoadDate,
	reg.CLASS_NBR AS classNumber,
	er.ENRLMT_RSN_LD AS dropAddActionDesc
FROM
	BaseData.bd.bt_cs_F_ENROLLMENT_COURSE AS reg
JOIN 
	BaseData.bd.bt_cs_D_GRADE AS g
	ON g.GRADE_SID = reg.GRADE_SID
JOIN 
	BaseData.bd.bt_cs_D_ENRLMT_RSN AS er
	ON er.ENRLMT_RSN_SID = reg.ENRLMT_REAS_SID
JOIN 
	BaseData.bd.bt_cs_D_SES_TERM AS st
	ON st.SES_TERM_SID = reg.SES_TERM_SID
;
