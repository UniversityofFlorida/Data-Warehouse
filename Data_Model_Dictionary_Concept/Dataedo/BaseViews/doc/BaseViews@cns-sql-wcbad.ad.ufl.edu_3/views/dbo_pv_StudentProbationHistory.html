CREATE VIEW [dbo].[pv_StudentProbationHistory] AS 
    
/*****************************************************************************************
  Name:		    pv_StudentProbationHistory

  Description:  List of students who are being tracked in the SB Probation application.

  Created:	    04/29/2014 by William McCombie

  Modified:		06/01/2017 by John Holmes
				Rewrite against hsbProbation   
                
*****************************************************************************************/ 

SELECT
	sh.UFID,
	sh.dateChanged,
	sh.probationTerm6,
	tm.termName AS 'ProbationTerm',
	sh.class,
	sh.college,
	sh.major,
	s.street,
	s.city,
	s.state,
	s.zip,
	sh.ufGPA,
	sh.ufDeficits,
	sh.coreGPA,
	sh.coreDeficits,
	sh.majorGPA,
	sh.majorDeficits,
	sh.totalHoursEarned,
	sh.currentProbationStatusID,
	sh.currentProbationStatusCode AS 'currentStatus',
	sh.previousProbationStatusID,
	sh.previousProbationStatusCode AS 'previousStatus',
	sh.hadPetition,
	sh.endOfTermAction,
	sh.changedByUFID,
	sh.comment
FROM
	hsbProbation.dbo.v_studentHistory AS sh
LEFT JOIN
	BaseViews.dbo.dv_termsModules AS tm
	ON sh.probationTerm6 = tm.term6
	AND tm.moduleTypeID = 1
LEFT JOIN
	hsbProbation.dbo.v_students AS s
	ON sh.UFID = s.ufid
;
