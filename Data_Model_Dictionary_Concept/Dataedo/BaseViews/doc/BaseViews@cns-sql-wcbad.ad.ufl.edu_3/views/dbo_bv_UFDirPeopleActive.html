/*****************************************************************************************

WITH 
[ufBusinessName] ([UFID], [Name]) AS (
	SELECT 
		 [UF_UFID]
		,[UF_NAME_TXT]
	FROM 
		[BaseData].[bd].[bt_UFDIR_NAME]
	WHERE 
		[UF_TYPE_CD] = 33
)
--,[ufDirectoryName] ([UFID], [Name]) AS (

SELECT 
     [p].[UF_UFID] AS [UFID]                    -- UFID
    ,[n].[Name] AS [UFBusinessName]				-- UF Business Name
    ,[p].[UF_DISPLAY_NM] AS [UFDisplayName]     -- UF Display Name
    ,[a].[USERIDALIAS] AS [GatorLinkID]         -- Gatorlink ID
    ,[p].[UF_EMAIL] AS [Email]                  -- UF Business Email
    ,[p].[UF_PRIMARY_AFFL] AS [Affiliation]     -- Primary Affiliation

    ,[p].[UF_PUBLISH_FLG] AS [PublishFlag]      -- Publish Flag (Y|N)
    ,[p].[UF_SECURITY_FLG] AS [SecurityFlag]    -- Security Flag (Y|N)
    --,[p].[UF_WORK_TITLE] AS [PositionTitle]     -- UF Position Title
    ,[p].[UF_ETHNICITY_CD] AS [Ethnicity]       -- Ethnicity Code

    ,[p].[UF_CITIZEN_CTRY] AS [Citizenship]     -- Country of Citizenship (2 char. code)
    ,[p].[UF_BIRTH_DT] AS [BirthDate]           -- Birth Date
	,[p].[UF_WORK_TITLE] AS [UFWorkingTitle]	-- UF Working Title
	--,[dn].[Name] AS [UFDirectoryName]			-- UF Directory Name

FROM 
    [BaseData].[bd].[bt_UFDIR_PERSON] [p]
LEFT JOIN 
    [BaseData].[bd].[bt_GL_ACCT] [a]
    ON [p].[UF_UFID] = [a].[OPRID] 
LEFT JOIN 
    [ufBusinessName] AS [n] 
    ON [p].[UF_UFID] = [n].[UFID]
--LEFT JOIN 
--    [ufDirectoryName] AS [dn]
