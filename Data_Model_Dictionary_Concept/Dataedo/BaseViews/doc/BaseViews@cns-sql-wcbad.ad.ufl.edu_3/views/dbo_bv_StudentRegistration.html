CREATE VIEW [dbo].[bv_StudentRegistration] AS 
    
/*****************************************************************************************
  Name:		    bv_StudentRegistration

  Description:  Student Registration (current and future terms)

  Created:	    09/25/2009 by John Holmes

  Modified:     11/20/2009 by John Holmes
                Modified to join on WCBA_TERM6
                
                08/10/2010 by John Holmes
                Modified join on bt_CREC_COURSE to use WCBA_SummerTerm6
                
                09/06/2011 by John Holmes
                v1.1.0 - Source object renaming in support of ERP (Oracle) Warehouse.
                
                03/29/2012 by John Holmes
                Return to previous specification (OUR Warehouse) of containing only current 
                and future terms (ESODBC now contains historical data in STU_CLASS and STUDENT_REG).
                
*****************************************************************************************/ 
    
SELECT 
    [R].[UF_UFID] AS [UFID],                  -- UFID
    [C].[UF_SECTION] AS [Section],            -- Section Numer
    [C].[UF_CREDIT_HRS] AS [CreditHours],     -- Credit Hours
    (CASE 
        WHEN COALESCE([C].[UF_GRADE], '') != '' THEN [C].[UF_GRADE] 
        ELSE COALESCE([CREC].[UF_GRADE], '') 
    END) AS [Grade],                         -- Grade 
    [C].[UF_FLAG] AS [StatusFlag],            -- Student Course Status Flag
    [R].[UF_SUMMER_A_FLG] AS [SummerAFlag],   -- Summer A Registration Status Flag (C|R|W)
    [R].[UF_SUMMER_B_FLG] AS [SummerBFlag],   -- Summer B Registration Status Flag (C|R|W)
    [R].[UF_SUMMER_C_FLG] AS [SummerCFlag],   -- Summer C Registration Status Flag (C|R|W)
    [R].[WCBA_TERM6] AS [Term6],              -- Term Code (6 char) 
    [C].[UF_DROP_ADD_FLG] AS [DropAddFlag],   -- Drop Add Flag (A|D|Z)
    [C].[UF_ADD_ID] AS [DropAddCode],         -- Drop Add Tranaction Code 
    [C].[LOAD_DTTM] AS [LoadDate]             -- Load Timestamp 
FROM 
    [BaseData].[bd].[bt_STUDENT_REG] [R] 
INNER JOIN 
    [BaseData].[bd].[bt_STU_CLASS] [C] 
    ON [C].[UF_UFID] = [R].[UF_UFID]
    AND [C].[WCBA_TERM6] = [R].[WCBA_TERM6] 
LEFT JOIN 
    [BaseData].[bd].[bt_CREC_COURSE] [CREC] 
    ON [CREC].[UF_UFID] = [R].[UF_UFID]
    AND [CREC].[UF_SECTION] = [C].[UF_SECTION] 
    AND [CREC].[WCBA_SummerTerm6] = [R].[WCBA_TERM6]
WHERE
    [C].[UF_UFID] != '02062630'            -- Remove 'Alberta' active GL account data
    AND (
        [C].[UF_DROP_ADD_FLG] != 'Z'       -- See definitions of Grade & DropAddFlag for interdependencies 
        OR 
        ([C].[UF_DROP_ADD_FLG] = 'Z' AND ([C].[UF_GRADE] = 'OK' OR [CREC].[UF_GRADE] = 'OK'))
    )
    AND [R].[WCBA_TERM6] > = [BaseViews].[dbo].[fn_GetTerm]('C',6)
