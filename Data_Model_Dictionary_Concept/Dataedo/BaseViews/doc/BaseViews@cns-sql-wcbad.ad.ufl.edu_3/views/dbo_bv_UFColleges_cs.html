CREATE VIEW [dbo].[bv_UFColleges_cs] AS 
    
/*****************************************************************************************
  Name:		    bv_UFColleges_cs

  Description:  UF Academic Groups (Colleges)-  This is a compatibility view using CS data

  Created:	    05/17/2018 by John Holmes

  Modified:     

                
*****************************************************************************************/ 
    
SELECT
	col.UF_COLLEGE_CD AS CollegeCode,
	col.UF_COLLEGE_NO AS CollegeNumber,
	ag.ACAD_GRP_SD AS CollegeName,
	ag.ACAD_GRP_LD AS CollegeTitle,
	col.UF_COLLEGE_TYPE AS CollegeType,
	ag.LASTUPD_EW_DTTM AS LastModifiedDate,
	NULL AS LoadDate,
	ag.ACAD_GRP_SID AS groupSID,
	ag.ACAD_GRP_CD AS groupCode
FROM
    BaseData.bd.bt_cs_D_ACAD_GRP AS ag
LEFT JOIN
	BaseData.bd.dt_cs_xwalk_groupsToColleges AS xw
	ON xw.ACAD_GRP_SID = ag.ACAD_GRP_SID
LEFT JOIN 
	BaseData.bd.bt_COLLEGES AS col
	ON col.UF_COLLEGE_CD = xw.UF_COLLEGE_CD
WHERE 
	ag.ACAD_GRP_SID > 0		-- Filter out "default" acad group
;
