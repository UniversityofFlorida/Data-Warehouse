SELECT
    [FD].[UF_UFID] AS [UFID],                     -- UFID 
    [FD].[UF_FLAG_CD] AS [Flag],                  -- Flag Code (hold type) 
    [FD].[UF_APPLY_DATE] AS [AppliedDate],        -- Date flag applied 
    [FD].[UF_APPLIED_BY] AS [AppliedBy],          -- User ID of Person Who Applied Flag
    [FS].[UF_LST_RFRSH_DT] AS [LastRefreshDate],  -- Last Refresh Date 
    [FS].[UF_CURR_COLLEGE] AS [CurrentCollege],   -- Current College (2 char) 
	[FS].[UF_CURR_MAJOR] AS [CurrentMajor],       -- Current Major (3 char)  
	[FS].[UF_NEXT_COLLEGE] AS [NextCollege],      -- Next College (2 char) 
	[FS].[UF_NEXT_MAJOR] AS [NextMajor],          -- Next Major (3 char) 
	[D].[UF_FLAG_NM] AS [FlagName],               -- Flag Name 
	[FD].[LOAD_DTTM] AS [LoadDate]                -- Load Timestamp
FROM 
    [BaseData].[bd].[bt_FLAG_DETAIL] [FD] 
INNER JOIN 
    [BaseData].[bd].[bt_FLAG_SUMMARY] [FS]
    ON [FS].[UF_UFID] = [FD].[UF_UFID] 
LEFT JOIN
    [BaseData].[bd].[bt_FLAG_DESC] [D] 
    ON [D].[UF_FLAG_CD] = [FD].[UF_FLAG_CD] 
WHERE 
    COALESCE([FD].[UF_APPLY_DATE], '') != ''
