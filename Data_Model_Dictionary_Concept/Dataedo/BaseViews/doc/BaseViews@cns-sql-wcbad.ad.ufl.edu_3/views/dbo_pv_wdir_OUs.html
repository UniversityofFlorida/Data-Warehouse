CREATE VIEW [dbo].[pv_wdir_OUs] AS 
    
/*****************************************************************************************
  Name:		    pv_wdir_OUs

  Description:  Published View of Warrington Directory - OUs 

  Created:	    01/17/2017 by John Holmes

  Modified:     
*****************************************************************************************/ 

WITH 
	[email](
		 [ouID]
		,[typeID]
		,[typeDescription]
		,[email]
		 )
		AS (
			SELECT
				 [ouID]
				,[typeID]
				,[typeDescription]
				,[email]
			FROM
				[warringtonDirectory].[dbo].[v_email]
			WHERE 
				[typeID] = -2000	-- Warrington Business Email
		)
	,[office](
		 [ouID]
		,[room]
		,[buildingCode]
		,[buildingName]
		)
		AS (
			SELECT
				 [ouID]
				,[room]
				,[buildingCode]
				,[buildingName]
			FROM
				[warringtonDirectory].[dbo].[v_locations]
			WHERE 
				[ouID] IS NOT NULL
				AND [typeID] = -5000	-- Office location
		)
	,[telephone](
		 [ouID]
		,[typeID]
		,[areaCode]
		,[phone]
		,[ext]
		)
		AS (
			SELECT
				 [ouID]
				,[typeID]
				,[areaCode]
				,[phone]
				,[ext]
			FROM
				[warringtonDirectory].[dbo].[v_telephone]
			WHERE 
				[ouID] IS NOT NULL 
		)
SELECT 
	 [ou].[ouID]
	,[ou].[code]
    ,[ou].[name]
	,[ou].[ouTypeID]
	,[ou].[ouTypeCode]
	,[ou].[departmentCode]
	,[ou].[departmentCodeShort]
	,[ou].[publicDisplay]
	,[email].[email] AS [businessEmail]
	,[office].[room]
	,[office].[buildingCode]
	,[office].[buildingName]
	,COALESCE('('+[officeTelephone].[areaCode]+') ', '') +
	 LEFT([officeTelephone].[phone], 3) + '-' + RIGHT([officeTelephone].[phone], 4) +
	 COALESCE(' x'+[officeTelephone].[ext], '') 
	 AS [telephone]
	,COALESCE('('+[fax].[areaCode]+') ', '') +
	 LEFT([fax].[phone], 3) + '-' + RIGHT([fax].[phone], 4) +
	 COALESCE(' x'+[fax].[ext], '') 
	 AS [fax]
FROM 
    [warringtonDirectory].[dbo].[dv_organizationalUnits] AS [ou]
LEFT JOIN  
	[email]
	ON [ou].[ouID] = [email].[ouID]
LEFT JOIN 
	[office]
	ON [ou].[ouID] = [office].[ouID]
LEFT JOIN 
	[telephone] AS [officeTelephone]
	ON [ou].[ouID] = [officeTelephone].[ouID]
	AND [officeTelephone].[typeID] = -4000
LEFT JOIN 
	[telephone] AS [fax]
	ON [ou].[ouID] = [fax].[ouID]
	AND [fax].[typeID] = -4001
WHERE 
	[ou].[ouID] > 0		-- Filter out Default OU
;
