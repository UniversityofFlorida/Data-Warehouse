CREATE VIEW [dbo].[pv_WcbaDepartments] AS 
    
/*****************************************************************************************
  Name:		    pv_WcbaDepartments

  Description:  Published View of Warrington Directory - Departments

  Created:	    08/04/2015 by John Holmes

  Modified:     
*****************************************************************************************/ 

WITH 
	[email](
		 [ouID]
		,[email]
		 )
		AS (
			SELECT
				 [ouID]
				,[email]
			FROM
				[warringtonDirectory].[dbo].[v_email]
			WHERE 
				[typeID] = -2000	-- Business Email
				AND [ouID] IS NOT NULL 
				AND [publicDisplay] = 1
		)
	,[location](
		 [ouID]
		,[room]
		,[buildingCode]
		 )
		AS (
			SELECT
				 [ouID]
				,[room]
				,[buildingCode]
			FROM
				[warringtonDirectory].[dbo].[v_locations]
			WHERE 
				[typeID] = -5000	-- Office Locatoin
				AND [ouID] IS NOT NULL 
				AND [publicDisplay] = 1
		)
	,[tel](
		 [ouID]
		,[areaCode]
		,[phone]
		,[ext]
		 )
		AS (
			SELECT
				 [ouID]
				,[areaCode]
				,[phone]
				,[ext]
			FROM
				[warringtonDirectory].[dbo].[v_telephone]
			WHERE 
				[typeID] = -4000	-- Business Telephone
				AND [ouID] IS NOT NULL 
				AND [publicDisplay] = 1
		)
	,[fax](
		 [ouID]
		,[areaCode]
		,[phone]
		,[ext]
		 )
		AS (
			SELECT
				 [ouID]
				,[areaCode]
				,[phone]
				,[ext]
			FROM
				[warringtonDirectory].[dbo].[v_telephone]
			WHERE 
				[typeID] = -4001	-- FAX Telephone
				AND [ouID] IS NOT NULL 
				AND [publicDisplay] = 1
		)
	,[urls](
		 [ouID]
		,[url]
		 )
		AS (
			SELECT
				 [ouID]
				,[url]
			FROM
				[warringtonDirectory].[dbo].[v_urls]
			WHERE 
				[urlTypeID] = 1	-- Homepage
				AND [ouID] IS NOT NULL 
				AND [publicDisplay] = 1
		)
	
SELECT
	 [ou].[code] AS [ShortName]
	,[ou].[name] AS [FullName]
	,[email].[email] AS [Email]
	,[location].[room] AS [Room]
	,[location].[buildingCode] AS [Building]
	,COALESCE([tel].[ext],'') AS [Extension]
	,LEFT([tel].[phone],3) + '-' + RIGHT([tel].[Phone],4) AS [Phone]
	,LEFT([fax].[phone],3) + '-' + RIGHT([fax].[Phone],4) AS [Fax]
	,[urls].[url] AS [URL]
FROM
	[warringtonDirectory].[dbo].[dv_organizationalUnits] AS [ou]
LEFT JOIN 
	[email]
	ON [ou].[ouID] = [email].[ouID]
LEFT JOIN 
	[location]
	ON [ou].[ouID] = [location].[ouID]
LEFT JOIN
	[tel]
	ON [ou].[ouID] = [tel].[ouID]
LEFT JOIN
	[fax]
	ON [ou].[ouID] = [fax].[ouID]
LEFT JOIN
	[urls]
	ON [ou].[ouID] = [urls].[ouID]
WHERE 
	[ou].[ouTypeID] = 1
;
