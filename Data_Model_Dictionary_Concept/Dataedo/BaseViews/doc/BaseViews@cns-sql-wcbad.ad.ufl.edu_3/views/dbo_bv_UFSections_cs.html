CREATE VIEW [dbo].[bv_UFSections_cs] AS 
    
/*****************************************************************************************
  Name:		    bv_UFSections_cs

  Description:  Classes (sections) - Schedule of Classes by Term
				
				This is a compatibility view using CS data

  Created:	    05/14/2018 by John Holmes

  Modified:     

*****************************************************************************************/ 
    
SELECT
	cls.TERM6 AS Term6,
	cls.CLASS_SECTION AS Section,
	cls.CLASS_NBR AS classNumber,
	cls.UF_COURSE_NUMBER AS Course,
	CASE
		WHEN cls.SSR_COMP_SID = 8 THEN 'Z' -- Lab
		ELSE 'X'
	END AS Suffix,
	sess.UF_ABC AS SemesterABC,
	crs.CRSE_LD AS CourseTitle,
	crs.ACAD_PRGRS_UNITS AS Credit,
	CASE
		WHEN crs.MIN_UNITS <> crs.MAX_UNITS THEN 'Y'
		ELSE ''
	END AS VariableCredit,
	'NEEDED' AS SUOption,		-- UFIT - "in the queue to be calculated"
	CASE
		WHEN cls.OMIR_ONL_MIRROR_FLG = 'Y' THEN 'Y'
		ELSE ''
	END AS TVReplay,
	'NEEDED' AS WebCourse,		-- cls.UF_DELIVERY_METHOD
	'NEEDED' AS SpecialProgram, -- UFIT - "in the queue to be calculated"
	CASE
		WHEN cls.SELF_FUND_FLG = 'Y' THEN 'Y'
		ELSE ''
	END AS OffBook,
	'' AS Cancelled,			-- Cancelled classes (sections) are no longer listed
	cls.ENRL_TOT AS Enrollment,
	cls.ENRL_CAP AS MaximumEnrollment,
	cls.MIN_ROOM_CAPACITY AS RoomCapacity,
	CASE
		WHEN MR_MR_FLG = 'Y' THEN 'M'
		ELSE ''
	END + CASE
			  WHEN WR_2000_FLG = 'Y' THEN '2'
			  ELSE ''
		  END + CASE
					WHEN WR_4000_FLG = 'Y' THEN '4'
					ELSE ''
				END + CASE
						  WHEN WR_6000_FLG = 'Y' THEN '6'
						  ELSE ''
					  END + CASE
								WHEN WR_WR_FLG = 'Y' THEN '2'
								ELSE ''
							END AS GordonRuleFLag,
	CASE
		WHEN cls.GEN_BIOSCI_FLG = 'Y' THEN 'B'
		ELSE ''
	END + CASE
			  WHEN cls.GEN_COMP_FLG = 'Y' THEN 'C'
			  ELSE ''
		  END + CASE
					WHEN cls.GEN_DIVERS_FLG = 'Y' THEN 'D'
					ELSE ''
				END + CASE
						  WHEN cls.GEN_HUM_FLG = 'Y' THEN 'H'
						  ELSE ''
					  END + CASE
								WHEN cls.GEN_INTL_FLG = 'Y' THEN 'N'
								ELSE ''
							END + CASE
									  WHEN cls.GEN_INTL_I_FLG = 'Y' THEN 'I'
									  ELSE ''
								  END + CASE
											WHEN cls.GEN_LIT_FLG = 'Y' THEN 'L'
											ELSE ''
										END + CASE
												  WHEN cls.GEN_MATH_FLG = 'Y' THEN 'M'
												  ELSE ''
											  END + CASE
														WHEN cls.GEN_PHYSSCI_FLG = 'Y' THEN 'P'
														ELSE ''
													END + CASE
															  WHEN cls.GEN_SOCSCI_FLG = 'Y' THEN 'S'
															  ELSE ''
														  END AS GeneralEducationCode,
	'NEEDED' AS LabFee,
	dept.ACAD_ORG_CD AS DepartmentCode,
	dept.ACAD_ORG_FD AS DepartmentName,
	col.ACAD_GRP_CD AS College,
	'NEEDED' AS ExamGroup,	--cls.EXAM_TIME_CODE
	cls.LASTUPD_EW_DTTM AS LastModifiedDate,
	NULL AS LoadDate,
	cls.CLASS_START_DT AS ClassBeginDate,
	cls.CLASS_END_DT AS ClassEndDate
FROM
	BaseData.bd.bt_cs_D_CLASS AS cls
JOIN
	BaseData.bd.bt_cs_D_SES_TERM AS sess
	ON sess.SES_TERM_SID = cls.SESSION_SID
JOIN
	BaseData.bd.bt_cs_D_CRSE AS crs
	ON crs.CRSE_SID		 = cls.CRSE_SID
JOIN
	BaseData.bd.bt_cs_D_ACAD_ORG AS dept
	ON dept.ACAD_ORG_SID = cls.ACAD_ORG_SID
JOIN
	BaseData.bd.bt_cs_D_ACAD_GRP AS col
	ON col.ACAD_GRP_SID	 = cls.ACAD_GRP_SID ;
