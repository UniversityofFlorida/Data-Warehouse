CREATE VIEW [dbo].[dv_termsSummer] AS 
    
/*****************************************************************************************
  name:		    dv_termsSummer

  Description:  Summer A & B Term (Module) Definitions and Dates Compatibility View

  Created:	    04/06/2016 by John Holmes

  Modified:     
                
*****************************************************************************************/ 
    
SELECT
	 CASE 
		WHEN [tm].[moduleID] = 6 THEN LEFT([tm].[term6],4) + '06'
		ELSE [tm].[term6]
	 END AS [term6]
	,[tm].[term6] AS [baseTerm6]
	,[m].[name] + ' ' + LEFT([tm].[term6],4) AS [termName]
	,CAST([tm].[termStart] AS [datetime2](0)) AS [termStartDate]
	,CAST([tm].[classesStart] AS [datetime2](0)) AS [classesStartDate]
	,DATEADD(SECOND,86399,CAST([tm].[dropAddEnd] AS [datetime2](0))) AS [dropAddEndDate]
	,DATEADD(SECOND,86399,CAST([tm].[classesEnd] AS [datetime2](0))) AS [classesEndDate]
	,DATEADD(SECOND,86399,CAST([tm].[termEnd] AS [datetime2](0))) AS [termEndDate]
	,CAST([tm].[advRegistrationStart] AS [datetime2](0)) AS [advRegistrationStartDate]
FROM
    [BaseData].[bd].[dt_termsModules] AS [tm]
JOIN 
	[BaseData].[bd].[dt_modules] AS [m]
	ON [tm].[moduleID] = [m].[moduleID]
WHERE 
	[tm].[moduleID] IN (5,6)
;
