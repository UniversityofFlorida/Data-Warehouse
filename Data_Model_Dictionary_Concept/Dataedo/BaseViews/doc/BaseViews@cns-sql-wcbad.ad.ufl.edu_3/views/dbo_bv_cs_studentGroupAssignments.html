CREATE VIEW [dbo].[bv_cs_studentGroupAssignments] AS 
    
/*****************************************************************************************

  Name:		    bv_cs_studentGroupAssignments

  Description:  Student Group Assignments with Effective Dates - Campus Solutions data

				Maintains the effective starting date and effective end date of a person's student 
				group assignments and includes the history of the student groups. 
				This is a bridge table to be joined between the person object by personSID and the 
				studentGroups object by studentGroupSID. 
				Student groups are assigned to a person for tagging.

  Created:	    04/18/2018 by John Holmes

  Modified:     
                
*****************************************************************************************/ 
    
SELECT
	p.UFID,
	sg.STDNT_GROUP_CD AS groupCode,
	sg.STDNT_GROUP_SD AS groupShortDesc,
	sg.STDNT_GROUP_LD AS groupLongDesc,
	sd.DAY_DT AS effectiveStartDate,
	ed.DAY_DT AS effectiveEndDate,
	psg.EFF_STATUS AS activeStatus,
	psg.PERSON_SID AS personSID,
	psg.STDNT_GROUP_SID AS studentGroupSID,
	--psg.INSTITUTION_SID AS institutionSID,
	psg.EFFDT_SID AS effStartDateSID,
	psg.END_EFFDT_SID AS effEndDateSID
FROM
	BaseData.bd.bt_cs_B_PERSON_STDNT_GRP AS psg
JOIN 
	BaseData.bd.bt_cs_D_UF_DAY AS sd
	ON sd.DAY_SID = psg.EFFDT_SID
JOIN 
	BaseData.bd.bt_cs_D_UF_DAY AS ed
	ON ed.DAY_SID = psg.END_EFFDT_SID
JOIN
	BaseData.bd.bt_cs_D_STDNT_GROUP AS sg
	ON sg.STDNT_GROUP_SID = psg.STDNT_GROUP_SID
JOIN
	BaseData.bd.bt_cs_D_MDM_PERSON AS p
	ON p.PERSON_SID = psg.PERSON_SID

;
