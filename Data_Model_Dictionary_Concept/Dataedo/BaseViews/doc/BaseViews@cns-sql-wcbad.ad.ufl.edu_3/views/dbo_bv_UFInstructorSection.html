CREATE VIEW [dbo].[bv_UFInstructorSection] AS 
    
/*****************************************************************************************
  Name:		    bv_UFInstructorSection

  Description:  Instructor to Section Mapping.
                
  Notes:        1. Only non-expired records are included in this view. 
                   Expired records (BaseData.bd.bt_SECT_INSTRUCTOR.UF_EXPIRE_TS != NULL) 
                   indicate an instructor that has been 'deleted' from a section.
                   
                2. Until Summer 2010 the WCBA relied on Instructor Workload data to determine
                   instructor-to-section mapping.  In the Select from bt_SECT_INSTRUCTOR, 
                   the Where clause restriction of [UF_MTRM_TERM_CD] > '201001' is a result of 
                   this, and the fact that the WCBA was not using (and therefore not updating) 
                   SECT_INSTRUCTOR until Summer 2010 (201005).  Thus, SECT_INSTRUCTOR data up 
                   through Spring 2010 (201001) does not accurately reflect the WCBA's operations.
                   The data in ht_UFInstructorSection has been synthesized from historical
                   Instructor Workload data so that we use a consistent historical view of 
                   instructor-to-section mapping, combined with the newer data from SECT_INSTRUCTOR.

  Created:	    11/10/2009 by John Holmes

  Modified:     02/05/2010 by John Holmes
                Added Course Code (Number) and union with history table
                
                09/06/2011 by John Holmes
                v1.1.0 - Source object renaming in support of ERP (Oracle) Warehouse.

				01/21/2016 by John Holmes
				Join to bv_UFSections
                
*****************************************************************************************/ 
    
SELECT 
    [ish].[Term6] AS [Term6]                        -- Term Code (6 char)
    ,[ish].[Section] AS [Section]                   -- Section
    ,[ish].[Course] AS [Course]                     -- Course Code (Number)
    ,[ish].[InstructorUFID] AS [InstructorUFID]     -- Instructor UFID
    ,[ish].[InstructorName] AS [InstructorName]     -- Instructor Name
    ,[ish].[DepartmentCode] AS [DepartmentCode]     -- Department Code
    ,[ou].[OUShortName] AS [DepartmentName]         -- Department (short) Name 
    ,'' AS [ClassRollViewerFlag]                    -- Class Roll Viewer Flag
    ,'' AS [GraderFlag]                             -- Grader Flag   
FROM 
    [BaseDataHistory].[dbo].[ht_UFInstructorSection] AS [ish]
LEFT JOIN 
    [BaseViews].[dbo].[dv_OrganizationalUnits] AS [ou]
    ON [ou].[DeptCode] = [ish].[DepartmentCode]
    
UNION ALL 

SELECT 
    [si].[UF_MTRM_TERM_CD] AS [Term6]                   -- Term Code (6 char)
    ,[si].[UF_SECTION] AS [Section]                     -- Section
    ,[sec].[Course]										-- Course Code (Number)
    ,[si].[UF_INSTR_UFID] AS [InstructorUFID]			-- Instructor UFID
    ,[si].[UF_INSTR_NAME] AS [InstructorName]           -- Instructor Name
    ,[sec].[DepartmentCode]								-- Department Code
    ,[ou].[OUShortName] AS [DepartmentName]             -- Department (short) Name 
    ,[si].[UF_CLASSROLL_VIEWER] AS [ClassRollViewerFlag]-- Class Roll Viewer Flag
    ,[si].[UF_GRADER] AS [GraderFlag]                   -- Grader Flag
FROM 
    [BaseData].[bd].[bt_SECT_INSTRUCTOR] AS [si]
--JOIN 
--    [BaseData].[bd].[bt_SECTION] AS [sec]
--    ON [si].[UF_MTRM_TERM_CD] = [sec].[WCBA_TERM6]
--    AND [si].[UF_SECTION] = [sec].[UF_SECTION]
JOIN 
    [BaseViews].[dbo].[bv_UFSections] AS [sec]
    ON [si].[UF_MTRM_TERM_CD] = [sec].[TERM6]
    AND [si].[UF_SECTION] = [sec].[SECTION]
LEFT JOIN 
    [BaseViews].[dbo].[dv_OrganizationalUnits] AS [ou]
    ON [ou].[DeptCode] = [sec].[DepartmentCode]
WHERE 
    [si].[UF_EXPIRE_TS] IS NULL         -- Active records only
    AND [sec].[DepartmentCode] LIKE '0117%'  -- Business sections only 
    AND [si].[UF_MTRM_TERM_CD] > '201001'
