/*****************************************************************************************
  Name:		    bv_StudentTranscriptCourses

  Description:  Student Courses from Transcript

  Created:	    12/11/2015 by William McCombie
				forked from bv_StudentTranscripts originally created
					09/25/2009 by John Holmes

SELECT  
    [crec].[UF_UFID] AS [UFID], 
    [crec].[WCBA_TERM_CD6] AS [Term6], 
    [crec].[UF_COMM_TYPE] AS [CourseType], 
    [crec].[UF_COURSE_CD] AS [Course],  
    [crec].[UF_SECTION] AS [Section], 
    [crec].[UF_GORD_RULE] AS [GordonRule], 
    [crec].[UF_GRADE] AS [Grade], 
    [crec].[UF_CREDIT] AS [Credit],
    [crec].[UF_GRADE_VALUE] AS [Multiplier],
	[crec].[UF_RPT_NOCR] AS [RepeatNoCredit], --128
	[crec].[UF_GRD_FORGIVEN] AS [GradeForgiven], --64
	[crec].[UF_CR_BEL_ACLVL] AS [CreditBelowAcceptableLevel], --32
	[crec].[UF_NOT_APPD_DEG] AS [NotAppliedToDegree], -- 16
	[crec].[UF_VOC_CRS_NOCR] AS [VocationalNoCredit], -- 8
	[crec].[UF_JC_COL_CRS] AS [JuniorCollege],       -- 4
	[crec].[UF_4_YR_COL_CSE] AS [FourYearCollege], -- 2
	[crec].[UF_XFER_RPT_NOCR] AS [TransferRepeatNoCredit], -- 1
	[crec].[UF_GRADE_CHG] AS [GradeChange],
	[crec].[UF_ALW_GRAD_CHG] AS [AllowGradeChange],
	[crec].[UF_NO_CREDIT] AS [NoCredit],
	[crec].[UF_NO_GPA] AS [NoGPA],
	[crec].[UF_ALW_RPT_CR] AS [AllowRepeatCredit],
    (CASE 
	    WHEN [term].[UF_CLASS] = 'X' OR [term].[UF_CLASS] IS NULL THEN '' 
	    ELSE [term].[UF_CLASS] END) AS [CourseClass],
    (CASE 
	    WHEN [term].[UF_COLLEGE_CD] IN ('X', 'XX', 'X,', 'X.') OR [term].[UF_COLLEGE_CD] IS NULL THEN '' 
	    ELSE [term].[UF_COLLEGE_CD] 
    END) AS [CourseClassCollege], 
    [crec].[UF_COURSE_NAME] AS [CourseTitle], 
    [crec].[UF_HRS_EARNED] AS [HoursEarned], 
    [crec].[UF_HRS_CARRIED] AS [HoursCarried], 
    [term].[UF_GPA] AS [UFTermGPA], 
    [term].[UF_UF_CUMM_GPA] AS [UFCummulativeGPA], 
    [crec].[UF_AUTO_INDEX] AS [AutoIndex], 
    [crec].[LOAD_DTTM] AS [LoadDate], 
    [term].[UF_ABC] AS [TermABCFlag], 
    [term].[UF_GRADE_PTS] AS [TermGradePoints] 
FROM
    [BaseData].[bd].[bt_CREC_COURSE] AS [crec] 
LEFT OUTER JOIN	
    [BaseData].[bd].[bt_CREC_TERM] AS [term]
    ON [crec].[UF_UFID] = [term].[UF_UFID] 
    AND [crec].[WCBA_TERM_CD6] = [term].[WCBA_TERM_CD6] 
    AND [crec].[UF_COMM_TYPE] = [term].[UF_COMM_TYPE]
