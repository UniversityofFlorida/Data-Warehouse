CREATE VIEW [dbo].[pv_CopyCenterOrders]
AS
    /*****************************************************************************************
  Name:		    pv_CopyCenterOrders

  Description:  List of orders processed by the Copy Center

  Created:	    08/07/2013 by William McCombie

  Modified:     
                
*****************************************************************************************/ 
    
SELECT
    
    [Orders].[OrderID]
   ,[Orders].[CustomerID]
   ,[Orders].[Description]
   ,[Orders].[Comments]
   ,[Orders].[OrderDate]
   ,[Orders].[RequiredDate]
   ,[Orders].[CompletedDate]
   ,[Orders].[ShippedDate]
   ,[Orders].[IsExamFlag]
   ,[Orders].[PickedUpBy]
   ,[Orders].[CompletedBy]
   ,[Orders].[EmployeeID]
   ,[Orders].[FeeTypeId]
   ,[Orders].[Term6]
   --,[Orders].[FeeData]
   ,
   CASE WHEN [Orders].[FeeTypeId] = 2 THEN COALESCE([People].[UFBusinessName], [FeeData]) else
   [Orders].[FeeData] END AS [FeeDetails]

   ,[FeeTypes].[FeeName]
   ,[CopyCodes].[ID]
   ,[CopyCodes].[Code]
   ,[CopyCodes].[DeptID]
   ,[CopyCodes].[AccountName]
   ,[CopyCodes].[ActiveFlag]
   ,[CopyCodes].[OldDeptID]
FROM
    [CopyWind].[dbo].[Orders]
LEFT JOIN [CopyWind].[dbo].[dt_FeeTypes] AS [FeeTypes]
    ON [FeeTypes].[FeeTypeId] = [Orders].[FeeTypeId]
LEFT JOIN [CopyWind].[dbo].[t_CopyCodes] AS [CopyCodes]
    ON [CopyCodes].[ID] = [Orders].[CustomerID]
	LEFT JOIN [BaseViews].[dbo].[bv_UFDirPeople] AS [People]
    ON [People].[UFID] = [Orders].[FeeData]
       AND [FeeTypes].[FeeName] = 'Research'
