CREATE VIEW [dbo].[pv_CapManagementMentorPoints]
AS
    /*****************************************************************************************
  Name:		    pv_CapManagementMentorPoints

  Description:  List of entries made by CAP mentors for "CAP Points" program

  Created:	    08/13/2013 by William McCombie

  Modified:     08/26/2013 by William McCombie
				Altered due to schema change
                
				01/02/2014 by William McCombie
				Add event date field
*****************************************************************************************/ 
    
SELECT
    [Term6]
   ,[Points].TimeIn
   ,[Points].[TimeIn] AS [EventDate]
   ,DATEDIFF(MINUTE, TimeIn, TimeOut) AS [Duration]
   ,DATEPART(MONTH, [Points].TimeIn) AS [Month]
   ,[People].[UFBusinessName]
   ,[Points].[Description]
   ,[Points].[CategoryDescription] AS [Category]
   ,[Points].[DefaultScore] AS [Score]
   ,[Points].[Comments]
FROM
    [SBCAPManagement].[dbo].[v_MentorPoints] AS [Points]
LEFT JOIN [BaseViews].[dbo].[bv_UFDirPeople] AS [People]
    ON [People].[UFID] = [Points].[MentorUFID]
