CREATE VIEW [dbo].[dv_calendar] AS 
    
/*****************************************************************************************
  name:		    dv_calendar

  Description:  Calendar

  Created:	    09/08/2016 by John Holmes

  Modified:     
                
*****************************************************************************************/ 

SELECT
	 [c].[date]
	,[c].[day]
	,[c].[calendarDay]
	,[c].[weekNumber]
	,[c].[monthNumber]
	,[c].[quarterNumber]
	,[c].[year]
	,[c].[dayOfWeek]
	,[c].[isLastDayOfMonth]
	,[c].[isWeekendDay]
	,CONVERT([bit], CASE WHEN [c].[isWeekendDay] = 0 AND [ufh].[date] IS NULL AND [ush].[date] IS NULL THEN 1 ELSE 0 END) AS [isWorkDay]
	,CASE WHEN [ush].[date] IS NOT NULL THEN 1 ELSE 0 END AS [isUSHoliday]
	,CASE WHEN [ufh].[date] IS NOT NULL THEN 1 ELSE 0 END AS [isUFHoliday]
	,COALESCE([ush].[name], 'UF - '+[ufh].[name], NULL) AS [holidayName]
	,[c].[isLastDowInMonth]
	,[c].[dowCount]
FROM
    [BaseData].[bd].[dt_calendar] AS [c]
LEFT JOIN 
	[BaseData].[bd].[dt_ufHolidays] AS [ufh]
	ON [c].[date] = [ufh].[date]
LEFT JOIN 
	[BaseData].[bd].[dt_usHolidays] AS [ush]
	ON [c].[date] = [ush].[date]
;
