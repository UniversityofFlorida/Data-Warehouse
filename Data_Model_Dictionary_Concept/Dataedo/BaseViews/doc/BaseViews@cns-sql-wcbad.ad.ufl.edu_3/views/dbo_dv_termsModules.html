CREATE VIEW [dbo].[dv_termsModules] AS 
    
/*****************************************************************************************
  name:		    dv_termsModules

  Description:  Terms and Modules Definitions and Dates Master View

  Created:	    04/06/2016 by John Holmes

  Modified:     05/22/2017 by John Holmes
				Add gsMidpointOfTerm
                
*****************************************************************************************/ 
    
SELECT
	 [recID]
	,CASE 
		WHEN [tm].[moduleID] = 6 THEN LEFT([tm].[term6],4) + '06'
		ELSE [tm].[term6]
	 END AS [term6]
	,[tm].[term6] AS [baseTerm6]
	,CASE 
		WHEN [tm].[moduleID] = 0 THEN [t].[name] + ' ' + LEFT([tm].[term6],4)
		ELSE [m].[name] + ' ' + LEFT([tm].[term6],4)
	 END AS [termName]
	,[mt].[name] AS [moduleType]
	,[tm].[status]
	,CAST([tm].[advRegistrationStart] AS [datetime2](0)) AS [advRegistrationStart]
	,CAST([tm].[priorityRegistrationStart] AS [datetime2](0)) AS [priorityRegistrationStart]
	,CAST([tm].[regularRegistrationStart] AS [datetime2](0)) AS [regularRegistrationStart]
	,CAST([tm].[termStart] AS [datetime2](0)) AS [termStart]
	,DATEADD(SECOND,86399,CAST([tm].[termEnd] AS [datetime2](0))) AS [termEnd]
	,CAST([tm].[classesStart] AS [datetime2](0)) AS [classesStart]
	,DATEADD(SECOND,86399,CAST([tm].[classesEnd] AS [datetime2](0))) AS [classesEnd]
	,CAST([tm].[dropAddStart] AS [datetime2](0)) AS [dropAddStart]
	,DATEADD(SECOND,86399,CAST([tm].[dropAddEnd] AS [datetime2](0))) AS [dropAddEnd]
	,DATEADD(SECOND,61200,CAST([tm].[degreeApplicationDeadline] AS [datetime2](0))) AS [degreeApplicationDeadline]
	,DATEADD(SECOND,55800,CAST([tm].[feesDue] AS [datetime2](0))) AS [feesDue]
	,DATEADD(SECOND,86399,CAST([tm].[dropDeadline] AS [datetime2](0))) AS [dropDeadline]
	,CAST([tm].[breakStart] AS [datetime2](0)) AS [breakStart]
	,DATEADD(SECOND,86399,CAST([tm].[breakEnd] AS [datetime2](0))) AS [breakEnd]
	,CAST([tm].[finalsStart] AS [datetime2](0)) AS [finalsStart]
	,DATEADD(SECOND,86399,CAST([tm].[finalsEnd] AS [datetime2](0))) AS [finalsEnd]
	,DATEADD(SECOND,86399,CAST([tm].gsMidpointOfTerm AS [datetime2](0))) AS gsMidpointOfTerm
	,[tm].[termID]
	,[tm].[moduleID]
	,[tm].[moduleTypeID]
FROM
    [BaseData].[bd].[dt_termsModules] AS [tm]
JOIN 
	[BaseData].[bd].[dt_terms] AS [t]
	ON [tm].[termID] = [t].[termID]
JOIN
	[BaseData].[bd].[dt_modules] AS [m]
	ON [tm].[moduleID] = [m].[moduleID]
JOIN
	[BaseData].[bd].[dt_moduleTypes] AS [mt]
	ON [tm].[moduleTypeID] = [mt].[moduleTypeID]
;
