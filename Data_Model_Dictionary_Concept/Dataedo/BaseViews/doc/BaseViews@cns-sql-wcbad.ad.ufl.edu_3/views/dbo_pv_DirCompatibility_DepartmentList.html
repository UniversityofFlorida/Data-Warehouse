CREATE VIEW [dbo].[pv_DirCompatibility_DepartmentList] AS 
    
/*****************************************************************************************
  Name:		    pv_DirCompatibility_DepartmentList

  Description:  Legacy directory compatibility view. Replaces v_DEPTLIST.
				List of organizational units (departments) for consumption by public website contact info pages

  Note:			This is a functional replacement for the legacy directory view above.
				It is used for BACKWARD COMPATIBILITY ONLY so that old 'n busted ASP.net web pages
				don't know the directory has been upgraded.

  Created:	    04/25/2016 by John Holmes

  Modified:     
*****************************************************************************************/ 

WITH 
	[email](
		 [ouID]
		,[email]
		 )
		AS (
			SELECT
				 [ouID]
				,[email]
			FROM
				[warringtonDirectory].[dbo].[v_email]
			WHERE 
				[typeID] = -2000	-- Business Email
		)
	,[office](
		 [ouID]
		,[room]
		,[buildingCode]
		,[buildingName]
		)
		AS (
			SELECT
				 [ouID]
				,[room]
				,[buildingCode]
				,[buildingName]
			FROM
				[warringtonDirectory].[dbo].[v_locations]
			WHERE 
				[ouID] IS NOT NULL
                AND [typeID] = -5000	-- Office Location
		)
	,[address](
		 [ouID]
		,[address]
		,[city]
		,[state]
		,[zip]
		)
		AS (
			SELECT
				 [ouID]
				,[address]
				,[city]
				,[state]
				,[zip]
			FROM
				[warringtonDirectory].[dbo].[v_addresses]
			WHERE 
				[ouID] IS NOT NULL
                AND [typeID] = -3000	-- Mailing Address
		)
	,[telephone](
		 [ouID]
		,[typeID]
		,[areaCode]
		,[phone]
		,[ext]
		)
		AS (
			SELECT
				 [ouID]
				,[typeID]
				,[areaCode]
				,[phone]
				,[ext]
			FROM
				[warringtonDirectory].[dbo].[v_telephone]
			WHERE 
				[ouID] IS NOT NULL 
		)
	,[url](
		 [ouID]
		,[url]
		)
		AS (
			SELECT
				 [ouID]
				,[url]
			FROM
				[warringtonDirectory].[dbo].[v_urls]
			WHERE 
				[ouID] IS NOT NULL
                AND [typeID] = -10000	-- General
				AND [urlTypeID] = 1		-- Homepage
		)
SELECT 
	 [ou].[ouID] AS [OFFICEID]
	,[ou].[name] AS [NAME]
	,[ou].[ouTypeID] AS [ENTITYTYPEID]
	,[email].[email] AS [EMAIL]
	,[office].[room] AS [RM]
	,[office].[buildingCode] AS [BLDG]
	,CASE 
		WHEN [address].[address] LIKE 'PO Box %' THEN REPLACE([address].[address], 'PO Box ', '')
		ELSE ''
	 END AS [PO]
	,'' AS [OTHERADDRESS]
	,LEFT([phone].[phone],3) + '-' + RIGHT([phone].[phone],4) AS [PHONE]
	,[phone].[ext] AS [EXT]
	,LEFT([fax].[phone],3) + '-' + RIGHT([fax].[phone],4) AS [FAX]
	,[url].[url] AS [URL]
	,[ou].[code] AS [DEPT]
	,NULL AS [EDITDEPT]
	,[ou].[code] AS [DEPTID]
	,[ou].[ouTypeCode] AS [ENTITYTYPE]
	--,CASE 
	--	WHEN [address].[address] LIKE 'PO Box %' THEN REPLACE([address].[address], 'PO Box ', '')
	--	ELSE ''
	-- END AS [CAMPBOX]
	--,[office].[room] AS [CAMPRM]
	--,[office].[buildingCode] AS [CBLDG]
	--,LEFT([officeTelephone].[phone],3) + '-' + RIGHT([officeTelephone].[phone],4) AS [CPHONE]
	--,[officeTelephone].[ext] AS [EXT]
	--,[email].[email] AS [EMAIL]
	--,CASE [position].[positionTypeID] WHEN 1 THEN 'Y' ELSE 'N' END AS [FACDIR]
	--,CASE 
	--	WHEN [people].[webID] IS NULL THEN [people].[personID]
	--	ELSE [people].[webID]
	-- END AS [WEBID]
	--,[namedPosition].[namedPosition] AS [NAMEDPOSTITLE]
	--,UPPER([position].[positionTypeCode]) AS [POSTYP]
	--,[people].[UFID] AS [SSN]
	--,[primaryAff].[url] AS [URL]
FROM 
    [warringtonDirectory].[dbo].[dv_organizationalUnits] AS [ou]
LEFT JOIN  
	[email]
	ON [ou].[ouID] = [email].[ouID]
LEFT JOIN 
	[office]
	ON [ou].[ouID] = [office].[ouID]
LEFT JOIN 
	[address]
	ON [ou].[ouID] = [address].[ouID]
LEFT JOIN 
	[telephone] AS [phone]
	ON [ou].[ouID] = [phone].[ouID]
	AND [phone].[typeID] = -4000	-- Business Telephone
LEFT JOIN 
	[telephone] AS [fax]
	ON [ou].[ouID] = [fax].[ouID]
	AND [fax].[typeID] = -4001	-- FAX Telephone
LEFT JOIN
	[url]
	ON [ou].[ouID] = [url].[ouID]
WHERE 
	[ou].[ouID] > 0
;
