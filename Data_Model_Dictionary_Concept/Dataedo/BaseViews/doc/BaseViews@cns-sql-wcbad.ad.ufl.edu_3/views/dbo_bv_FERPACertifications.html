CREATE VIEW [dbo].[bv_FERPACertifications] AS 
    
/*****************************************************************************************
  Name:		    bv_FERPACertifications

  Description:  FERPA Certification Data for WCBA Users.
                Accepted courses:
                    PRV802 - FERPA Basics
                    PRV803 - FERPA for Faculty
  
  Notes:        1. This view contains ONLY people present in the WCBA Directory.
                2. If an individual has BOTH PRV802 and PRV803 certifications, only the most 
                   recent is included in this view.

  Created:	    07/06/2015 by John Holmes

  Modified:     
                
*****************************************************************************************/ 
    
SELECT 
     [cert].[EMPLID] AS [UFID]
    ,COALESCE([p].[UFBusinessName],'') AS [UFBusinessName]
    ,COALESCE([p].[Email],'') AS [Email]
    ,CASE
        WHEN [base].[ActiveFlag] = 'Y' THEN 1
        ELSE 0
     END AS [IsActiveWcbaDir]
    ,CAST([cert].[LAST_COMPLETED_DT] AS DATE) AS [CertificationDate]
    ,CAST([cert].[EXPIRATION_DATE] AS DATE) AS [ExpirationDate]
    ,DATEDIFF(DAY,GETDATE(),DATEADD(YEAR, 1, [cert].[LAST_COMPLETED_DT])) AS [DaysUntilExpiration]
    ,CASE
        WHEN DATEDIFF(DAY,GETDATE(),[EXPIRATION_DATE]) < 0 THEN CAST(0 AS BIT)
        ELSE CAST(1 AS BIT)
     END AS [IsValid]
    ,[cert].[COURSE] AS [Course]
    ,CASE
        WHEN [cert].[COURSE_NAME] LIKE 'FERPA Basics%' THEN 'FERPA Basics'
        WHEN [cert].[COURSE_NAME] LIKE 'FERPA for Faculty%' THEN 'FERPA for Faculty'
        ELSE [cert].[COURSE_NAME]
     END AS [CourseTitle]
FROM 
    [BaseData].[bd].[bt_COURSE_COMPLETIONS] AS [cert]
JOIN 
    (
        SELECT 
			 [cert].[EMPLID] AS [UFID]
			,[wp].[ActiveFlag]
            ,MAX([cert].[LAST_COMPLETED_DT]) AS [LastDate]
        FROM 
            [BaseData].[bd].[bt_COURSE_COMPLETIONS] AS [cert]
        JOIN 
            [BaseViews].[dbo].[pv_WcbaPeople] AS [wp]
            ON [cert].[EMPLID] = [wp].[UFID]
        GROUP BY 
             [cert].[EMPLID]
			,[wp].[ActiveFlag]
    ) AS [base]
    ON [cert].[EMPLID] = [base].[UFID]
    AND [cert].[LAST_COMPLETED_DT] = [base].[LastDate]
LEFT JOIN 
    [BaseViews].[dbo].[bv_UFDirPeople] AS [p]
    ON [cert].[EMPLID] = [p].[UFID]
WHERE 
    [cert].[COURSE] IN ('PRV802','PRV803')
;
