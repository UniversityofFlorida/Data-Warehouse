CREATE VIEW [dbo].[bv_UFDepartments_cs] AS 
    
/*****************************************************************************************
  Name:		    bv_UFDepartments_cs

  Description:  Academic Organizations (Departments)-  This is a compatibility view using CS data

  Created:	    05/17/2018 by John Holmes

  Modified:     
                
*****************************************************************************************/ 

SELECT
	d.UF_DEP_CD AS DepartmentCode,
	d.UF_COLLEGE_CD AS CollegeCode,
	org.ACAD_ORG_LD AS [Name],
	org.ACAD_ORG_FD AS Title,
	org.ACAD_ORG_SID AS organizationSID,
	org.ACAD_ORG_CD AS organizationCode,
	org.ACAD_ORG_SD AS organizationShortDesc
FROM
	BaseData.bd.bt_cs_D_ACAD_ORG AS org
LEFT JOIN
	BaseData.bd.bt_DEPARTMENTS  AS d
	ON RIGHT(d.UF_DEP_CD, 7) + '0' = org.ACAD_ORG_CD
WHERE 
	org.ACAD_ORG_SID > 0 		-- Filter out "default" acadademic organization
;
