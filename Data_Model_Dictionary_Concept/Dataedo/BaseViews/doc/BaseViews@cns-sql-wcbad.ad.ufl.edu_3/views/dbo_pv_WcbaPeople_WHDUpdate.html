CREATE VIEW [dbo].[pv_WcbaPeople_WHDUpdate] AS 
    
/*****************************************************************************************
  Name:		    pv_WcbaPeople_WHDUpdate

  Description:  Published View of warringtonDirectory - People
                Consumed by WHD (Web Help Desk) during Dept/Bldg/Room/Client update.

  Created:	    10/14/2013 by John Holmes

  Modified:     02/18/2014 by John Holmes
                [Email] = UF Directory email
                [Email2] = Warrington Directory email

				04/27/2016 by John Holmes
				Modified for new directory

*****************************************************************************************/ 

SELECT 
     [GatorLinkID] AS [GLID]
    ,CASE 
        WHEN COALESCE([Nickname], '') != '' 
            THEN RTRIM([Nickname])
        ELSE RTRIM([FirstName])
     END AS [FirstName]
    ,[LastName] AS [LastName]
    ,[Email] AS [Email]
    ,[Email] AS [Email2]
    ,CASE
        WHEN COALESCE([OfficePhone], '') = '' 
            THEN ''
        ELSE '(352) ' + [OfficePhone]
	 END AS [Phone]
    ,CASE 
        --If Office Phone is blank and Home Phone is not, include it here
        WHEN (COALESCE([OfficePhone], '') = '' AND COALESCE([HomePhone], '') != '') 
            THEN '(' + [HomeAreaCode] + ') ' + [HomePhone]
        ELSE ''
	 END AS [Phone2]
    ,CASE 
        WHEN [Department] = 'SB' THEN 'HSB'
        WHEN [Department] = 'MKG' THEN 'MAR'
        WHEN [Department] = 'MGT' THEN 'MAN'
        ELSE [Department]
	 END AS [Department]
    ,CASE 
        --If Office Building and Office Room are both blank enter 'Off-Campus'
        WHEN ((COALESCE([OfficeBuilding], '') = '') AND (COALESCE([OfficeRoom], '') = '')) 
            THEN 'Off-Campus'
        ELSE [OfficeBuildingName]
	 END AS [Building]
    ,CASE 
        --If Office Building and Office Room  are both blank enter 'n/a'
        WHEN ((COALESCE([OfficeBuilding], '') = '') AND (COALESCE([OfficeRoom], '') = '')) 
            THEN 'n/a'
        ELSE [OfficeRoom]
	 END AS [Room]
    ,CASE 
        WHEN [ActiveFlag] = 'Y' THEN 1
        ELSE 0
     END AS [IsActive]
FROM 
    [BaseViews].[dbo].[pv_WcbaPeople] AS [wp]
WHERE 
    [GatorLinkID] != 'UNKNOWN'
;
