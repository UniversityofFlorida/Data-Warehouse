CREATE VIEW [dbo].[pv_DirCompatibility_FacultyPublicDisplay] AS 
    
/*****************************************************************************************
  Name:		    pv_DirCompatibility_FacultyPublicDisplay

  Description:  Legacy directory compatibility view. Replaces v_FacDirDisplayPublic.
				List of faculty for consumption by public website contact info pages

  Note:			This is a functional replacement for the legacy directory view above.
				It is used for BACKWARD COMPATIBILITY ONLY so that old 'n busted ASP.net web pages
				don't know the directory has been upgraded.

  Created:	    04/22/2016 by John Holmes

  Modified:     01/05/2017 by John Holmes
				Return empty string when room or building is "none"

				01/06/2017 by John Holmes
				Return legacy Dept Name if it exists

				01/11/2017 by John Holmes
				Do not return records where ufProtect is set

				01/12/2017 by John Holmes
				Return department code for DEPT

				01/31/2017 by John Holmes
				Include area code in cphone

				02/14/2017 by John Holmes
				Return Warrington email if it exists, UF Email otherwise

				02/20/2017 by John Holmes
				Add ufDirectoryName

*****************************************************************************************/ 

WITH
	people(
		 SSN
		,personID
		,DEPT
		,listingOU
		,primaryOU
		,LASTN
		,FIRSTN
		,MIDINIT
		,PREFIX
		,SUFFIX
		,NICKNAME
		,INTRNLTITLE
		,WEBID
		,ufDirectoryName
		)
		AS (
			SELECT 
				 people.UFID
				,people.personID
				,listAff.affiliatedOUCode
				,listAff.affiliatedOUCode
				,ou.name
				,BaseViews.dbo.fn_GetUFBusinessNamePart(people.ufBusinessName, 'L')
				,BaseViews.dbo.fn_GetUFBusinessNamePart(people.ufBusinessName, 'F')
				,LEFT(BaseViews.dbo.fn_GetUFBusinessNamePart(people.ufBusinessName, 'M'),1)
				,COALESCE(people.salutation, '')
				,BaseViews.dbo.fn_GetUFBusinessNamePart(people.ufBusinessName, 'S')
				,BaseViews.dbo.fn_ProperCase(
					CASE
						WHEN BaseViews.dbo.fn_GetUFBusinessNamePart(people.ufDisplayName,'F') != BaseViews.dbo.fn_GetUFBusinessNamePart(people.ufBusinessName,'F')
							THEN BaseViews.dbo.fn_GetUFBusinessNamePart(people.ufDisplayName,'F')
						ELSE ''
					END 
				 )
				,COALESCE(people.ufWorkingTitle,'')
				,CASE 
					WHEN people.webID IS NULL THEN people.personID
					ELSE people.webID
				 END
				,people.ufDirectoryName
			FROM 
				warringtonDirectory.dbo.v_people AS people
			JOIN 
				warringtonDirectory.dbo.v_affiliations AS listAff
				ON listAff.UFID = people.UFID
				AND listAff.typeID = -7001
			JOIN 
				warringtonDirectory.dbo.dv_organizationalUnits AS ou
				ON ou.ouID = people.primaryAffiliationOUID
			WHERE 
				people.active = 1
				AND people.publicDisplay = 1
				AND people.primaryAffiliationOUID > 0
				AND people.ufProtect = 0
		)
	,warrEmail(
		 UFID
		,EMAIL
		 )
		AS (
			SELECT
				 UFID
				,email
			FROM
				warringtonDirectory.dbo.v_email
			WHERE 
				typeID = -2001	-- Warrington Email
		)
	,ufEmail(
		 UFID
		,EMAIL
		 )
		AS (
			SELECT
				 UFID
				,email
			FROM
				warringtonDirectory.dbo.v_email
			WHERE 
				typeID = 1	-- UF Business Email
		)
	,position(
		 UFID
		,positionName
		)
		AS (
			SELECT
				 UFID
				,positionName
			FROM
				warringtonDirectory.dbo.v_positionAssignments
			WHERE 
				positionTypeID = 1	-- Faculty
		)
	,namedPosition(
		 UFID
		,namedPosition
		,namedPositionType
		,namedPositionTypeCode
		)
		AS (
			SELECT
				 UFID
				,namedPosition
				,namedPositionType
				,namedPositionTypeCode
			FROM
				warringtonDirectory.dbo.v_namedPositionAssignments
		)
	,office(
		 UFID
		,typeID
		,room
		,buildingCode
		,buildingName
		)
		AS (
			SELECT
				 UFID
				,typeID
				,room
				,buildingCode
				,buildingName
			FROM
				warringtonDirectory.dbo.v_locations
		)
	,telephone(
		 UFID
		,typeID
		,areaCode
		,phone
		,ext
		)
		AS (
			SELECT
				 UFID
				,typeID
				,areaCode
				,phone
				,ext
			FROM
				warringtonDirectory.dbo.v_telephone
			WHERE 
				typeID = 10	--UF Business telephone
		)
	,primaryAff(
		 UFID
		,ouID
		,ouName
		,ouCode
		,url
		)
		AS (
			SELECT
				 p.UFID
				,o.ouID
				,o.name
				,o.code
				,u.url
			FROM
				warringtonDirectory.dbo.v_people AS p
			JOIN
				warringtonDirectory.dbo.dv_organizationalUnits AS o
				ON p.primaryAffiliationOUID = o.ouID
			LEFT JOIN 
				warringtonDirectory.dbo.v_urls AS u
				ON u.ouID = o.ouID
		)
	,address(
		 UFID
		,typeID
		,address
		,city
		,state
		,zip
		)
		AS (
			SELECT
				 UFID
				,typeID
				,address
				,city
				,state
				,zip
			FROM
				warringtonDirectory.dbo.v_addresses
			WHERE 
				typeID = 2	-- UF Mailing Address
				AND UFID IS NOT NULL
		)
	,interests(
		 UFID
		,text
		)
		AS (
			SELECT
				 UFID
				,text
			FROM
				warringtonDirectory.dbo.v_text
			WHERE 
				typeID = -13000	-- Research Areas
		)
	,urls(
		 UFID
		,urlTypeID
		,url
		)
		AS (
			SELECT
				 UFID
				,urlTypeID
				,url
			FROM
				warringtonDirectory.dbo.v_urls
		)
	,legacy(
		 UFID
		,DEPT
		)
		AS (
			SELECT
				 UFID
				,DEPT
			FROM
				warringtonDirectory.dbo.v_legacy
		)
SELECT 
	 people.SSN
	,people.PREFIX
	,people.LASTN
	,people.FIRSTN
	,people.MIDINIT
	,people.NICKNAME
	,people.WEBID
	,interests.text AS INTERESTS
	,homepage.url AS HOMEPAGE
	,bio.url AS PROFILE
	,vita.url AS VITA
	,publish.url AS PUBLISH
	,working.url AS WORKING
	,photo.url AS PHOTO
	,namedPosition.namedPosition AS NAMEDPOSTITLE
	,position.positionName AS UFTITLE
	,people.SUFFIX
	,people.INTRNLTITLE
	,people.DEPT
	,CASE 
		WHEN address.address LIKE 'PO Box %' THEN REPLACE(address.address, 'PO Box ', '')
		ELSE ''
	 END AS CAMPBOX
	,CASE WHEN office.room = 'none' THEN '' ELSE office.room END AS CAMPRM
	,CASE WHEN office.buildingCode = 'none' THEN '' ELSE office.buildingCode END AS CBLDG
	,telephone.areaCode + '.' + LEFT(telephone.phone,3) + '.' + RIGHT(telephone.phone,4) AS CPHONE
	,telephone.ext AS EXT
	,CASE 
		WHEN warrEmail.EMAIL IS NULL THEN ufEmail.EMAIL
		ELSE warrEmail.EMAIL
	 END AS EMAIL
	,namedPosition.namedPositionTypeCode AS TYPEID
	,'Y' AS FACDIR
	,primaryAff.url AS URL
	,people.primaryOU
	,people.ufDirectoryName
FROM 
    people
LEFT JOIN  
	warrEmail
	ON people.SSN = warrEmail.UFID
LEFT JOIN  
	ufEmail
	ON people.SSN = ufEmail.UFID
JOIN
	position
	ON people.SSN = position.UFID
LEFT JOIN 
	namedPosition
	ON people.SSN = namedPosition.UFID
LEFT JOIN 
	primaryAff
	ON people.SSN = primaryAff.UFID
LEFT JOIN 
	office
	ON people.SSN = office.UFID
	AND office.typeID = -5000
LEFT JOIN 
	telephone
	ON people.SSN = telephone.UFID
LEFT JOIN
	address
	ON people.SSN = address.UFID
LEFT JOIN
	interests
	ON people.SSN = interests.UFID
LEFT JOIN 
	urls AS homepage
	ON  people.SSN = homepage.UFID
	AND homepage.urlTypeID = 1
LEFT JOIN 
	urls AS bio
	ON  people.SSN = bio.UFID
	AND bio.urlTypeID = 8
LEFT JOIN 
	urls AS vita
	ON  people.SSN = vita.UFID
	AND vita.urlTypeID = 9
LEFT JOIN 
	urls AS publish
	ON  people.SSN = publish.UFID
	AND publish.urlTypeID = 11
LEFT JOIN 
	urls AS working
	ON  people.SSN = working.UFID
	AND working.urlTypeID = 10
LEFT JOIN 
	urls AS photo
	ON  people.SSN = photo.UFID
	AND photo.urlTypeID = 2
LEFT JOIN
	legacy
	ON people.SSN = legacy.UFID
;
