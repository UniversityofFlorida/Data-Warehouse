CREATE VIEW [dbo].[bv_UFCohorts]

AS

/*****************************************************************************************
  Name:		    bv_UFCohorts

  Description:  UF Cohort Details

  Created:	    01/21/2014 by John Holmes

  Modified:     
                
*****************************************************************************************/ 
    

SELECT
     [c].[MANAGED_COHORT] AS [Cohort]
    ,[cd].[EFFECTIVE_DATE] AS [EffectiveDate]
    ,[cd].[EXPIRE_DATE] AS [ExpirationDate]
    ,[cd].[SHORT_DESC] AS [ShortDescription]
    ,[cd].[LONG_DESC] AS [LongDescription]
    ,[c].[RESTRICT_FLG] AS [RestrictionFlag]
    ,[cd].[NOTE] AS [Notes]
    ,CASE 
        WHEN [cd].[VALID_ROW] = 'Y' THEN 1
        ELSE 0
     END AS [IsValid]
    ,[cd].[CREATE_TS] AS [CreateTimestamp]
    ,[cd].[UPDATE_TS] AS [UpdateTimestamp]
FROM
    [BaseData].[bd].[bt_SW_COHORT_ID] AS [c]
LEFT JOIN 
    [BaseData].[bd].[bt_SW_COHORT_DETAIL] AS [cd]
    ON [c].[MANAGED_COHORT] = [cd].[MANAGED_COHORT]
