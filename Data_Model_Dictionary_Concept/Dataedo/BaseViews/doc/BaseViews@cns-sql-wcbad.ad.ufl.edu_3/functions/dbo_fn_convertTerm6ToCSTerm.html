/**************************************************************************************************

  Name:		        fn_convertTerm6ToCSTerm

  Description:	    BaseViews - Convert Term6 Term string into Campus Solutions 4-char format
  
  Input:            @termIN:    Character string representing the term. LEN = 6
  
  Usage:            BaseViews.dbo.fn_convertTerm6ToCSTerm('Term_To_Convert')
  
                    Example: SELECT BaseViews.dbo.fn_convertTerm6ToCSTerm('201808')
                    Returns: '2188'

  Created:	        04/03/2018 by John Holmes

  Modified:
  	
**************************************************************************************************/

RETURNS varchar(4)

AS

BEGIN
    
	-- Declare and initialize the return and working variables
	DECLARE @termOUT AS varchar(4) = '' ;
	
	
	IF LEN(@termIN) <> 6			-- Error Checking: input length
		SET @termOUT = 'INPERR' ;
    ELSE							-- If no input errors, perform conversion
		SET @termOUT = LEFT(@termIN, 1) + RIGHT(LEFT(@termIN, 4), 2) + RIGHT(@termIN, 1) ;
	

	-- Return the result
	RETURN @termOUT 
	
END
