(

/**************************************************************************************************

  Name:		        fn_GetSystemEmailAddress

  Description:	    Return a system email address as defined in [AdminTools].[dbo].[t_SystemEmailAddresses].
  
  Input:            @name VARCHAR(10) - The "Name" label of an address 
                        Values: SELECT [Name] FROM [AdminTools].[dbo].[t_SystemEmailAddresses].
                        Default value = 'Reports'
  
  Usage:            [BaseViews].[dbo].[fn_GetSystemEmailAddress](@name)
  
                    Example: SELECT [BaseViews].[dbo].[fn_GetSystemEmailAddress]('Reports')

  Created:	        02/06/2012 by John Holmes

  Modified:
  	
**************************************************************************************************/

RETURNS VARCHAR(256)

AS

BEGIN
    
	-- Declare the return variable
	DECLARE @address AS VARCHAR(256) ;
	
	-- For testing
	-- DECLARE @name VARCHAR(10) = 'Reports' ;
	
	-- Do the work
	SET @address = (
	    SELECT 
	        [EmailAddress] 
	    FROM 
	        [AdminTools].[dbo].[t_SystemEmailAddresses]
	    WHERE 
	        [Name] = @name
	) ;
	        
	
	-- Return the result
	RETURN @address ;
	
END ;
