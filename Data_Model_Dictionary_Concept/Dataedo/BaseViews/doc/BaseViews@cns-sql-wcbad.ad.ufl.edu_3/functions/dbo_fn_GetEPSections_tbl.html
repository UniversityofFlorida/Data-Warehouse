CREATE FUNCTION [dbo].[fn_GetEPSections_tbl]
(
    @termCodeIn AS CHAR(1)
)

/**************************************************************************************************

  Name:		        fn_GetEPSections_tbl

  Description:	    Returns a table containing data on the requested term's (P, C or N) EP Courses.
  
  Input:            @termCodeIn char(1):   Char representing term requested [ P | C | N ]
  
  Output:           Table of the format:
                    Term6           varchar(6), 
                    Section         varchar(4), 
                    Course          varchar(7), 
                    SemesterABC     varchar(1) 
                    Type            varchar(7)
                    SpecialProgram  varchar(5)
  
  Usage:            SELECT * FROM BaseViews.dbo.fn_GetEPSections_tbl(@termCodeIn)
                    Example: 
                        SELECT * FROM BaseViews.dbo.fn_GetEPSections_tbl('N')

  Created:	        08/31/2009 by John Holmes

  Modified:         01/11/2010 by John Holmes
                    Added Type and SpecialProgram columns
                    
                    04/28/2010 by William McCombie
                    Adjusted to include Cancelled and Enrollment fields
                    
**************************************************************************************************/

RETURNS 
@Result TABLE 
(
	[Term6] VARCHAR(6)
	,[Section] VARCHAR(4) 
	,[Course] VARCHAR(7) 
	,[SemesterABC] VARCHAR(1) 
	,[Type] VARCHAR(7)
	,[SpecialProgram] VARCHAR(5)
	,[Cancelled] VARCHAR(1)
	,[Enrollment] int
)
AS
BEGIN
    -- Setup working vars
    DECLARE @Term VARCHAR(6)
    DECLARE @TermCode VARCHAR(1)
        
    -- Check for valid input
    IF (@termCodeIn = 'P' OR @termCodeIn = 'C' OR @termCodeIn = 'N')
        BEGIN
            -- Get Previous term data
	        IF @termCodeIn = 'P'
	            BEGIN 
	                SET @Term = [BaseViews].[dbo].[fn_GetTerm]('P',6)
	                SET @TermCode = RIGHT(@Term, 1)
		        END
            -- Get Current term data
	        ELSE IF @termCodeIn = 'C' 
	            BEGIN 
	                SET @Term = [BaseViews].[dbo].[fn_GetTerm]('C',6)
	                SET @TermCode = RIGHT(@Term, 1)
		        END
            -- Get Next term data
	        ELSE 
                BEGIN
	                SET @Term = [BaseViews].[dbo].[fn_GetTerm]('N',6)
	                SET @TermCode = RIGHT(@Term, 1)
		        END
		        
		    
		    -- Get the EP data
		    INSERT INTO 
		        @Result (
		            [Term6]
		            ,[Section]
		            ,[Course]
		            ,[SemesterABC]
		            ,[Type]
		            ,[SpecialProgram]
		            ,[Cancelled]
		            ,[Enrollment]
		        ) 
            SELECT 
                [sec].[Term6] 
                ,[sec].[Section]
                ,[sec].[Course] 
                ,[sec].[SemesterABC] 
                ,'EP'
                ,[sec].[SpecialProgram]
                ,[sec].[Cancelled]
                ,[sec].[Enrollment]
            FROM 
                [BaseViews].[dbo].[bv_UFSections] [sec]
            JOIN 
                (
                    SELECT 
                        [Course]
                    FROM 
                        [BaseViews].[dbo].[dv_CourseAssociations]
                    WHERE 
                        [AssociationTerm] = @TermCode
                        AND [AssociationTypeID] = 4
                ) [ep]
                ON [sec].[Course] = [ep].[Course]
            WHERE 
                [sec].[Term6] = @Term
                AND [sec].[DepartmentCode] LIKE '0117%'
                AND COALESCE([sec].[SpecialProgram], '') = ''
            ORDER BY 
                [sec].[Course], [sec].[Section] 
		        
		    
		    -- Get the OBSBA data
		    INSERT INTO 
		        @Result (
		            [Term6]
		            ,[Section]
		            ,[Course]
		            ,[SemesterABC]
		            ,[Type]
		            ,[SpecialProgram]
					,[Cancelled]
		            ,[Enrollment]
		        ) 
            SELECT 
                [sec].[Term6] 
                ,[sec].[Section]
                ,[sec].[Course] 
                ,[sec].[SemesterABC] 
                ,'OBSBA'
                ,[sec].[SpecialProgram]
                ,[sec].[Cancelled]
		        ,[sec].[Enrollment]
            FROM 
                [BaseViews].[dbo].[bv_UFSections] [sec]
            WHERE 
                [sec].[Term6] = @Term
                AND [sec].[DepartmentCode] LIKE '0117%'
                AND [sec].[SpecialProgram] IN ('IBA','IBO','IBT')
			ORDER BY 
                [sec].[Course], [sec].[Section] 
		        
		    
		    -- Get the IntBSBA data
		    INSERT INTO 
		        @Result (
		            [Term6]
		            ,[Section]
		            ,[Course]
		            ,[SemesterABC]
		            ,[Type]
		            ,[SpecialProgram]
					,[Cancelled]
		            ,[Enrollment]
		        ) 
            SELECT 
                [sec].[Term6] 
                ,[sec].[Section]
                ,[sec].[Course] 
                ,[sec].[SemesterABC] 
                ,'IntBSBA'
                ,[sec].[SpecialProgram]
				,[sec].[Cancelled]
		        ,[sec].[Enrollment]
            FROM 
                [BaseViews].[dbo].[bv_UFSections] [sec]
            WHERE 
                [sec].[Term6] = @Term
                AND [sec].[DepartmentCode] LIKE '0117%'
                AND [sec].[SpecialProgram] IN ('BAF','BAP','BAS','ITB','UKB')
                AND [sec].[Course] NOT IN ('GEB4117','GEB4956')
            ORDER BY 
                [sec].[Course], [sec].[Section]
                
        END

    RETURN 
    
END
