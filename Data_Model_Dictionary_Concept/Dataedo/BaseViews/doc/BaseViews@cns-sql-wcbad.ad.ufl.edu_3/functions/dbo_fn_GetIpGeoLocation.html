/*******************************************************************************************
  Name:         fn_GetIpGeoLocation
  
  Description:  Returns location information for an IP address.
                
                Based on an article by Oscar D. Garcia
                http://www.sqlservercentral.com/articles/SQL+Server/67215/
  
  Input:        @ip NVARCHAR    -   The IP address to locate
  
  Output:       Location information in table form.
  
  Notes:        This function relies on GeoLocation data stored in 2 BaseData tables:
                    bd.dt_GeoBlocks
                    bd.dt_GeoLocation
                The data in these tables must be manually loaded from CSV files publicly available 
                from: http://www.maxmind.com/app/geolite
                
                See https://itsp.wiki.warrington.ufl.edu/wiki/BaseData_-_Update_IP_GeoLocation_Data 
                for full data update instructions.
  
  Usage:        SELECT * FROM [BaseViews].[dbo].[fn_GetIpGeoLocation] ('128.227.001.001')
                Returns:    Country Region  City        PostalCode  Latitude    Longitude   MetroCode   AreaCode
                            US	    FL	    Gainesville 32611	    29.65160	-82.32480   592	        352
  
  Created:      07/20/2012 by John Holmes
  
  Modified:     
  
*******************************************************************************************/

BEGIN

    DECLARE @ipnum BIGINT = [BaseViews].[dbo].[fn_ConvertIp2Num](@ip) ;
    
    INSERT INTO @Result(
         [Country]
        ,[Region]
        ,[City]
        ,[PostalCode]
        ,[Latitude]
        ,[Longitude]
        ,[MetroCode]
        ,[AreaCode]
    )
    SELECT
         [l].[Country]
        ,[l].[Region]
        ,[l].[City]
        ,[l].[PostalCode]
        ,[l].[Latitude]
        ,[l].[Longitude]
        ,[l].[MetroCode]
        ,[l].[AreaCode]
    FROM 
        [BaseData].[bd].[dt_GeoBlocks] AS [b]
    INNER JOIN
        [BaseData].[bd].[dt_GeoLocation] AS [l]
        ON [b].[LocId] = [l].[LocId]
    WHERE 
        @ipnum BETWEEN [b].[StartIpNum] AND [b].[EndIpNum]
    
    

    RETURN ;
END ;
