CREATE FUNCTION [dbo].[fn_GetActiveTerms_tbl]
(
    @type AS int
)

/**************************************************************************************************

  Name:		        fn_GetActiveTerms_tbl

  Description:	    Returns a table containing all 'active' terms (previous, current, next and nextnext)
                    in either 6 (YYYYMM), 5 (YYYYM), 4 (YYMM) or 3 (YYM) character format.
  
  Input:            @type:   Integer representing output format (number of characters) [ 3 | 4 | 5 | 6 ]
  
  Output:           Previous, Current, Next and NextNext terms. 
  
  Usage:            SELECT * FROM BaseViews.dbo.fn_GetActiveTerms_tbl(@type)
                    Example: 
                        SELECT * FROM BaseViews.dbo.fn_GetActiveTerms_tbl(6)

  Created:	        04/08/2016 by John Holmes

  Modified:
  	
**************************************************************************************************/

RETURNS 
@Result table 
(
	Term varchar(6)
)
AS
BEGIN
    INSERT INTO 
		@Result(Term)
    SELECT 
		[BaseViews].[dbo].[fn_ConvertTerm](Term6, @type) AS [Term] 
    FROM 
		[BaseViews].[dbo].[dv_termsModules]
    WHERE 
		[Status] IN ('P','C','N','NN')
	;

	RETURN ;
END;
