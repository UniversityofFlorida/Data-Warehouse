CREATE FUNCTION [dbo].[fn_GetUFBusinessNamePart](

    @name AS VARCHAR(44)
    ,@part AS VARCHAR(1)
    
)

/**************************************************************************************************

  Name:		        fn_GetUFBusinessNamePart

  Description:	    Returns the specified part of UFBusinessName.
  
  Input:            @name:  UFBusinessName (Lastname,Firstname MI,Suffix)
                    @part:  ['L' (last) | 'F' (first) | 'M' (middle) | 'S' (suffix) | 'P' (proper name)]
                    
  Algorithm:        1. Split @name at first comma into [1a]|[1b] - [Lastname]|[Firstname MI,Suffix]
                    2. Split [1b] at comma into [2a]|[2b] - [Firstname MI]|[Suffix]
                    3. Split [2a] at space into [3a]|[3b] - [Firstname]|[MI]
  
  Output:           Character string.

  Error Condition:  If the input cannot be parsed for the requested part, the empty string ('') is returned.
  
  Usage:            BaseViews.dbo.fn_GetUFBusinessNamePart(@name, @part)
                    Example: 
                        SELECT BaseViews.dbo.fn_GetUFBusinessNamePart('Von Last,First MI,Suffix', 'L')
                        Returns: 'Von Last'

  Created:	        10/21/2009 by John Holmes

  Modified:
  	
**************************************************************************************************/

RETURNS VARCHAR(44)

AS

BEGIN   
    
    -- Cleanse the input
    SET @name = [BaseViews].[dbo].[fn_LRTrim](@name)
    SET @part = UPPER(@part)
    
    -- Delcare output var
    DECLARE @out VARCHAR(44)    -- The final output
    SET @out = ''
    
    -- If @part is a valid input
    IF @part IN ('L','F','M','S','P')
    BEGIN
        -- Init working vars
        DECLARE @index INT = 0          -- The index of a character
        DECLARE @1a VARCHAR(44) = ''    -- Lastname
        DECLARE @1b VARCHAR(44) = ''    -- Firstname MI,Suffix
        DECLARE @2a VARCHAR(44) = ''    -- Firstname MI
        DECLARE @2b VARCHAR(44) = ''    -- Suffix
        DECLARE @3a VARCHAR(44) = ''    -- Firstname
        DECLARE @3b VARCHAR(44) = ''    -- MI
        
        
        -- Perform first split on @name
        SET @index = CHARINDEX(',',@name)   -- position of the first comma
        IF @index > 0   -- If there is a comma
            BEGIN
                SET @1a = LEFT(@name,(@index - 1))
                SET @1b = SUBSTRING(@name, (@index + 1), LEN(@name))
            END 
        ELSE 
            SET @1a = ''    -- Lastname
        
        
        -- Perform second split on @1b
        SET @index = CHARINDEX(',',@1b)     -- position of second comma
        IF @index > 0   -- If there is a comma
            BEGIN
                SET @2a = LEFT(@1b,(@index - 1))
                SET @2b = SUBSTRING(@1b, (@index + 1), LEN(@1b))
            END
        ELSE
            BEGIN 
                SET @2a = @1b
                SET @2b = ''    -- Suffix
            END 
            
            
        -- Perform third split on @2a
        SET @index = CHARINDEX(' ',@2a)     -- position of space
        IF @index >0    -- If there is a space
            BEGIN
                SET @3a = LEFT(@2a,(@index - 1))                    -- Firstname
                SET @3b = SUBSTRING(@2a, (@index + 1), LEN(@2a))    -- MI
            END
        ELSE
            BEGIN
                SET @3a = @2a   -- Firstname
                SET @3b = ''    -- MI
            END
            
            
        -- Calculate result
        IF @part = 'L'          -- Lastname
            SET @out = @1a
            
        ELSE IF @part = 'F'     -- Firstname
            SET @out = @3a
            
        ELSE IF @part = 'M'     -- MI
            SET @out = @3b
            
        ELSE IF @part = 'S'     -- Suffix
            SET @out = @2b
        
        ELSE                    -- Proper Name
            SET @out = 
                REPLACE(
                    @3a + ' ' + @3b + ' ' + @1a + ' ' + @2b,
                    '  ',
                    ' '
                ) 
            
    END
    
    -- Return result
    RETURN @out
END
