CREATE FUNCTION [dbo].[fn_GetFiscalYear] 
(
    @termIN AS VARCHAR(7) 
)

/**************************************************************************************************

  Name:		        fn_GetFiscalYear

  Description:	    Returns the Fiscal year of a term.
  
  Input:            @termIN: Character string representing the term.  
                    Can be any valid term format, i.e. 3, 4, 5 or 6 char.
  
  Output:           On Success: Fiscal Year in YYYY-YY format
                    On Failure: The string 'INP_ERR
  
  Usage:            BaseViews.dbo.fn_GetFiscalYear('Term_To_Convert')
                    
                    Example: SELECT BaseViews.dbo.fn_GetFiscalYear('200608')
                    Returns: '2006-07' 
                    
  Process Overview: 1. @termIN is converted to YYYYMM format
                    2. If MM < 5 Then return (YYYY-1)-YY
                       Else return YYYY-(YY+1)

  Created:	        09/29/2008 by John Holmes

  Modified:
  	
**************************************************************************************************/

RETURNS VARCHAR(7)

AS

BEGIN
    
	-- Declare and initialize the return and working variables
	DECLARE @OUT AS VARCHAR(7)
	DECLARE @YEAR AS INT 
	DECLARE @MONTH AS INT 
	DECLARE @TERM AS VARCHAR(6)
	SET @OUT = ''
	SET @YEAR = 0 
	SET @MONTH = 0
	
	
	
	-- Convert term to 6 char and perform error checking
	SET @TERM = BaseViews.dbo.fn_ConvertTerm(@termIN, 6)
    IF (@TERM = 'INPERR' OR @TERM = '')
    BEGIN
        SET @OUT = 'INP_ERR'
    END
    
    
    -- If no input errors, calculate output
    ELSE
    BEGIN
        SET @YEAR = CAST(LEFT(@TERM, 4) AS INT)
        SET @MONTH = CAST(RIGHT(@TERM, 2) AS INT)
        SET @OUT = 
	        CASE 
	            WHEN @MONTH < 5 
	            THEN CAST((@YEAR - 1) AS VARCHAR(4)) + '-' + RIGHT(LEFT(@TERM, 4) , 2)
	            
	            ELSE LEFT(@TERM, 4) + '-' + RIGHT(CAST((@YEAR + 1)AS VARCHAR(4)), 2)
            END
	END
	
	
	-- Return the result
	RETURN @OUT 
	
END
