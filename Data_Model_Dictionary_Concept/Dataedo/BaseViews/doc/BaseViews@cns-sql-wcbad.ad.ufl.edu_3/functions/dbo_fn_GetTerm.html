CREATE FUNCTION [dbo].[fn_GetTerm]
(
     @term		AS [varchar](2)
	,@format	AS [integer]
)

/**************************************************************************************************

  Name:		        fn_GetTerm

  Description:	    Returns the (P)revious, (C)urrent, (N)ext or (NN)extNext term in the specified format.
  
  Input:            @term:    Character representing requested term [P | C | N | NN ]
                    @format:  Integer representing output format (number of characters) [ 3 | 4 | 5 | 6 ]
  
  Output:           Character string.
  
  Usage:            BaseViews.dbo.fn_GetTerm(@term, @format)
                    Example: 
                        SELECT BaseViews.dbo.fn_GetTerm('C', 6)

  Created:	        04/08/2016 by John Holmes

  Modified:
  	
**************************************************************************************************/

RETURNS [varchar](6)

AS

BEGIN
    DECLARE @out [varchar](6) ;
    
    SET @term = [BaseViews].[dbo].[fn_LRTrim](@term) ;
     
    IF  @term IN ('P','C','N','NN') AND (@format BETWEEN 3 AND 6)
        SET @out = (SELECT [BaseViews].[dbo].[fn_ConvertTerm](term6, @format) FROM [BaseViews].[dbo].[dv_termsModules] WHERE [status] = @term) ;
    ELSE 
        SET @out ='INPERR' ;
        
    RETURN @out ;
END
