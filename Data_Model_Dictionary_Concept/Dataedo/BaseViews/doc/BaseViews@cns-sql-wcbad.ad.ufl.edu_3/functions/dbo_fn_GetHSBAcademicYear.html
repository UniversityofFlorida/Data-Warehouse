CREATE FUNCTION [dbo].[fn_GetHSBAcademicYear] 
(
	@MatriculationTerm6 varchar(6)
)

/**************************************************************************************************

  Name:		        fn_GetHSBAcademicYear

  Description:	    Given a student matriculation term, determine the academic year for SB classification.
					This is NOT the same as a regular academic year. SB starts its academic year in Summer A
  
  Input:            MatriculationTerm6 varchar(6) - Matriculation term
  
  Usage:            [Stats].[dbo].[GetStudentYear](@<term6>)
  
                    Example: SELECT [Stats].[dbo].[fn_GetHSBAcademicYear]('200508')
                    Returns: 2005

					Example: SELECT [Stats].[dbo].[fn_GetHSBAcademicYear]('200501')
                    Returns: 2004

  Created:	        08/10/2016 by William McComibe

  Modified:
  	
**************************************************************************************************/

RETURNS INT

AS

BEGIN
    
	-- Declare the return variable
	DECLARE @year AS int
	
	SET @year = 
		CASE WHEN @MatriculationTerm6 LIKE '_____[56789]'
			   THEN CAST(LEFT(@MatriculationTerm6, 4) AS int)
			ELSE CAST(LEFT(@MatriculationTerm6, 4) AS int) - 1 END


	-- Return the result
	RETURN @year
	
END
