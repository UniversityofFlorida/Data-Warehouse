CREATE FUNCTION [dbo].[fn_ConvertNum2Ip] (

    @num BIGINT
    
)

RETURNS NVARCHAR(15)

AS 

/*******************************************************************************************
  Name:         fn_ConvertNum2Ip
  
  Description:  Converts IP number into IP address for (reverse) use by GeoLocation functions.
                
                Based on an article by Oscar D. Garcia
                http://www.sqlservercentral.com/articles/SQL+Server/67215/
  
  Input:        @num BIGINT    -   The IP number to convert
  
  Output:       The IP address (NVARCHAR)
  
  Usage         SELECT [BaseViews].[dbo].[fn_ConvertNum2Ip] (3236045770)
                Returns '192.226.35.202'
  
  Created:      07/20/2012 by John Holmes
  
  Modified:     
  
*******************************************************************************************/

BEGIN
    
    RETURN
        CAST(@num/16777216 AS NVARCHAR(15)) + '.' +
        CAST((@num/65536)%256 AS NVARCHAR(15)) + '.' +
        CAST((@num/256)%256 AS NVARCHAR(15)) + '.' +
        CAST(@num%256 AS NVARCHAR(15)) ;
        
END ;
