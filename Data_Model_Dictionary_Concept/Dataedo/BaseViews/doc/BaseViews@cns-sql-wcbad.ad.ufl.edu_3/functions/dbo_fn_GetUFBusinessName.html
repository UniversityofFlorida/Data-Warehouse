/**************************************************************************************************

  Name:		    fn_GetUFBusinessName

  Description:  Returns UFBusinessName.
  
  Input:        @input  varchar(70)     REQUIRED. UF ID (8 char) or GatorLink user name.
                
                @type   int             REQUIRED. @input type.
                                        1 = UFID
                                        2 = GLID

  Output:       1. On Success: Character string.
                2. On Failure: System error message.

  Usage:        BaseViews.dbo.fn_GetUFBusinessName(@input, @type)
                Examples: 
                    SELECT BaseViews.dbo.fn_GetUFBusinessName('00000542', 1) ;
                    Returns: 'Alligator,Albert'
                    
                    SELECT BaseViews.dbo.fn_GetUFBusinessName('alberta', 2) ;
                    Returns: 'Alligator,Alberta A'

  Created:	    05/13/2013 by John Holmes

  Modified:     05/16/2013 by John Holmes
                Add ability to translate GLID.
  	
**************************************************************************************************/

RETURNS VARCHAR(44)

AS

BEGIN  
    
    -- Delcare output var
    DECLARE @out VARCHAR(50) ;
    
    IF @type = 1
        SET @out = (
            SELECT 
                [UFBusinessName]
            FROM 
                [BaseViews].[dbo].[bv_UFDirPeople]
            WHERE 
                [UFID] = @input 
        );
    
    ELSE IF @type = 2
        SET @out = (
            SELECT 
                [UFBusinessName]
            FROM 
                [BaseViews].[dbo].[bv_UFDirPeople]
            WHERE 
                [GatorLinkID] = @input 
        );
        
    ELSE
        -- Invalid input
        SET @out ='Error: fn_GetUFBusinessName: @type invalid.' ;

    RETURN @out ;
END ;
