CREATE FUNCTION [dbo].[fn_GetPhoneList]
    (
     @UFIDs XML
    ,@PhoneType VARCHAR(1) = 'L'
    )

/**************************************************************************************************

  Name:		        fn_GetPhoneList

  Description:	    Given a list of UFIDs and a phone number type, return a list of phone number
  
  Input:            UFID - UFIDs in XML format
							Create with SELECT UFID FROM <table> FOR XML RAW('Person'), ROOT('People')
					PhoneType varchar(1) - Phone type (Local, Permanent, UF).
  
  Output:            
  
  Usage:            SELECT * FROM [BaseViews].[dbo].[fn_GetPhoneList](UFIDs, PhoneType)
                    
  Created:	        04/22/2010 by William McCombie

  Modified:         
  	
**************************************************************************************************/
RETURNS @Phones TABLE
    (
     [UFID] VARCHAR(8)
    ,[AreaCode] VARCHAR(5)
    ,[Phone] VARCHAR(10)
    ,[Ext] VARCHAR(6)
    )
AS 
    BEGIN
    
        INSERT  INTO @Phones
                (
                 UFID
                )
                SELECT
                    UFIDs.Person.value('@UFID[1]', 'varchar(8)') AS UFID
                FROM
                    @UFIDs.nodes('//People/Person') AS UFIDs (Person)    
       
        
    /*
		10 = UF Business, 
		12 = Local Home, 
		13 = Permanent Home, 
		29 = Parents Home, 
		110 = UF Housing
	
	*/
        IF @PhoneType = 'L' 
            BEGIN
	
    ----------------------------------------------------------------
    	    -- Add Phones
        -- Preferred Phone is Type 12 = Local Home
                UPDATE
                    @Phones
                SET 
                    [AreaCode] = [phn].[AreaCode]
                   ,[Phone] = [phn].[Phone]
                   ,[Ext] = [phn].[ext]
                FROM
                    @Phones AS [r]
                LEFT JOIN [BaseViews].[dbo].[bv_UFDirPeopleActiveTelePhone] AS [phn]
                    ON [r].[UFID] = [phn].[UFID]
                WHERE
                    [phn].[Type] = 12

        -- Add type 10
        -- Type 10 - UF Business
                UPDATE
                    @Phones
                SET 
                    [AreaCode] = [phn].[AreaCode]
                   ,[Phone] = [phn].[Phone]
                   ,[Ext] = [phn].[ext]
                FROM
                    @Phones AS [r]
                LEFT JOIN [BaseViews].[dbo].[bv_UFDirPeopleActiveTelePhone] AS [phn]
                    ON [r].[UFID] = [phn].[UFID]
                WHERE
                    [phn].[Type] = 10
                    AND COALESCE([r].[Phone], '') = ''
        
        
        
        -- Add type 110
        -- Type 110 - UF Housing
                UPDATE
                    @Phones
                SET 
                    [AreaCode] = [phn].[AreaCode]
                   ,[Phone] = [phn].[Phone]
                   ,[Ext] = [phn].[ext]
                FROM
                    @Phones AS [r]
                LEFT JOIN [BaseViews].[dbo].[bv_UFDirPeopleActiveTelePhone] AS [phn]
                    ON [r].[UFID] = [phn].[UFID]
                WHERE
                    [phn].[Type] = 110
                    AND COALESCE([r].[Phone], '') = ''
        
  
        -- Add type 13 Phones
        -- Type 13 = Permanent Home
                UPDATE
                    @Phones
                SET 
                    [AreaCode] = [phn].[AreaCode]
                   ,[Phone] = [phn].[Phone]
                   ,[Ext] = [phn].[ext]
                FROM
                    @Phones AS [r]
                LEFT JOIN [BaseViews].[dbo].[bv_UFDirPeopleActiveTelePhone] AS [phn]
                    ON [r].[UFID] = [phn].[UFID]
                WHERE
                    [phn].[Type] = 13
                    AND COALESCE([r].[Phone], '') = ''
            
                -- Add type 29 Phones
        -- Type 29 = Parents Home
                UPDATE
                    @Phones
                SET 
                    [AreaCode] = [phn].[AreaCode]
                   ,[Phone] = [phn].[Phone]
                   ,[Ext] = [phn].[ext]
                FROM
                    @Phones AS [r]
                LEFT JOIN [BaseViews].[dbo].[bv_UFDirPeopleActiveTelePhone] AS [phn]
                    ON [r].[UFID] = [phn].[UFID]
                WHERE
                    [phn].[Type] = 29
                    AND COALESCE([r].[Phone], '') = ''
            
            END
        ELSE 
            IF @PhoneType = 'P' 
                BEGIN

        -- Type 13 = Permanent Home
                    UPDATE
                        @Phones
                    SET 
                        [AreaCode] = [phn].[AreaCode]
                       ,[Phone] = [phn].[Phone]
                       ,[Ext] = [phn].[ext]
                    FROM
                        @Phones AS [r]
                    LEFT JOIN [BaseViews].[dbo].[bv_UFDirPeopleActiveTelePhone] AS [phn]
                        ON [r].[UFID] = [phn].[UFID]
                    WHERE
                        [phn].[Type] = 13
                      
                -- Add type 29 Phones
        -- Type 29 = Parents Home
                    UPDATE
                        @Phones
                    SET 
                        [AreaCode] = [phn].[AreaCode]
                       ,[Phone] = [phn].[Phone]
                       ,[Ext] = [phn].[ext]
                    FROM
                        @Phones AS [r]
                    LEFT JOIN [BaseViews].[dbo].[bv_UFDirPeopleActiveTelePhone] AS [phn]
                        ON [r].[UFID] = [phn].[UFID]
                    WHERE
                        [phn].[Type] = 29
                        AND COALESCE([r].[Phone], '') = ''
                END
    
    
    
    
       
        RETURN
    END
