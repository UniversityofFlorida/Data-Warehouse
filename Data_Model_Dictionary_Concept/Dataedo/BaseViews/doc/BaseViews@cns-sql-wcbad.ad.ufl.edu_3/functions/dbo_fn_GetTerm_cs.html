CREATE FUNCTION dbo.fn_GetTerm_cs (
	@term AS   varchar(2)
)

/**************************************************************************************************

  Name:		        fn_GetTerm_cs

  Description:	    Returns the (P)revious, (C)urrent, (N)ext or (NN)extNext term in Campus Solutions 4-char format.
  
  Input:            @term:    Character representing requested term [P | C | N | NN ]
  
  Output:           Character string.
  
  Usage:            BaseViews.dbo.fn_GetTerm_cs(@term)
                    Example: 
                        SELECT BaseViews.dbo.fn_GetTerm_cs('C')

  Created:	        05/30/2018 by John Holmes

  Modified:
  	
**************************************************************************************************/

RETURNS varchar(4)

AS

	BEGIN
		DECLARE @out varchar(4) ;

		SET @term = BaseViews.dbo.fn_LRTrim(@term) ;

		IF @term IN ('P', 'C', 'N', 'NN')
			SET @out = (BaseViews.dbo.fn_convertTerm6ToCSTerm(BaseViews.dbo.fn_GetTerm(@term, 6)))	;
		ELSE SET @out = 'ERR' ;

		RETURN @out ;
	END ;
