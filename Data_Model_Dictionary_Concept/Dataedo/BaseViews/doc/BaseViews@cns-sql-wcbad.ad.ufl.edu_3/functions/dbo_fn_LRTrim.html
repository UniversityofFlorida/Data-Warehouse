CREATE FUNCTION [dbo].[fn_LRTrim] 
(
    @strIN AS VARCHAR(200)
)

/**************************************************************************************************

  Name:		        fn_LRTrim

  Description:	    BaseViews - Trim leading and trailing spaces from a string.
  
  Input:            @strIN: Character string. 0 <= LEN <= 200 
  
  Usage:            BaseViews.dbo.fn_LRTrim('Some_String')
                    
                    Example: SELECT BaseViews.dbo.fn_LRTrim('   START-----END  ')
                    Returns: 'START-----END'

  Created:	        09/24/2009 by John Holmes

  Modified:
  	
**************************************************************************************************/

RETURNS VARCHAR(200)

AS

BEGIN
    
	-- Declare the return variable
	DECLARE @strOUT AS VARCHAR(200)
	SET @strOUT = ''
	
	
	-- Perform the trim
    SET @strOUT = LTRIM(RTRIM(@strIN))
	
	
	-- Return the result
	RETURN @strOUT

END
