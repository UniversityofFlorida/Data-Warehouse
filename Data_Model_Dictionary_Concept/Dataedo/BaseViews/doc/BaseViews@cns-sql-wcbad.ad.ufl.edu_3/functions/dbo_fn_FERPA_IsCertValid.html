CREATE FUNCTION [dbo].[fn_FERPA_IsCertValid] 
(
	@UFID VARCHAR(8)
)

/**************************************************************************************************

  Name:		        fn_FERPA_IsCertValid

  Description:	    Verifies that a user's FERPA certification has not expired
  
  Input:            UFID VARCHAR(8) - 8-char UF identifier (UFID)
  
  Output:           1 (True) or 0 (False)
  
  Usage:            [BaseViews].[dbo].[fn_FERPA_IsCertValid]('UFID')
  
                    Example: SELECT [BaseViews].[dbo].[fn_FERPA_IsCertValid]('00000000')
                    Returns: 0 (False)

  Created:	        08/06/2010 by John Holmes

  Modified:         10/11/2012 by John Holmes
                    Use FERPA data from BaseData

					11/05/2013 by William McCombie
					Fixed bug relating to having overlapping FERPA statuses

					01/04/2017 by John Holmes
					Reference bv_FERPACertifications_simple
  	
**************************************************************************************************/

RETURNS BIT

AS

BEGIN
    
	-- Declare the return variable
	DECLARE @result AS BIT  ;
	
	-- Do the work
	IF EXISTS (
        SELECT 
            [IsValid] 
        FROM 
            [BaseViews].[dbo].[bv_FERPACertifications_simple] 
        WHERE 
            [UFID] = @UFID 
            AND [IsValid] = 1
    )
	    SET @result = CAST(1 AS BIT) ;
	ELSE
	    SET @result = CAST(0 AS BIT) ;
	
	-- Return the result
	RETURN @result ;
	
END ;
