CREATE FUNCTION [dbo].[fn_IsStartSummerBTerm] ()

/**************************************************************************************************

  Name:		        fn_IsStartSummerBTerm

  Description:	    Determines if today is the first day of Summer B term.
  
  Usage:            [BaseViews].[dbo].[IsStartSummerBTerm]()
  
                    Example: SELECT [BaseViews].[dbo].[fn_IsStartSummerBTerm]()
  
  Returns:          1 if today is the first day of Summer B term
                    0 otherwise

  Created:	        05/07/2010 by John Holmes

  Modified:
  	
**************************************************************************************************/

RETURNS INT

AS

BEGIN
    
	-- Declare working variables
	DECLARE 
	    @result AS INT = 0          -- Return value
	    ,@currentYear AS VARCHAR(4) -- Current Year (YYYY)
	
	-- Determine current year
	SET @currentYear = LEFT([BaseViews].[dbo].[fn_GetTerm]('C',6),4)
	
	-- If this is summer term
	IF (RIGHT([BaseViews].[dbo].[fn_GetTerm]('C',6),2) = '05')
	BEGIN
	    -- Calculate the difference in days between today and the date Summer B Term starts.
	    IF (
            DATEDIFF(
                [day] 
                ,(
                    SELECT 
                        [TermStartDate] 
                    FROM 
                        [BaseViews].[dbo].[dv_TermsSummer] 
                    WHERE 
                        [Term6] = (@currentYear + '06')
                )
                ,GETDATE()
            ) = 0
        )
	        SET @result = 1     -- If diff is 0, today is the last day of Summer A Term
    END
	
	-- Return the result
	RETURN @result 
	
END
