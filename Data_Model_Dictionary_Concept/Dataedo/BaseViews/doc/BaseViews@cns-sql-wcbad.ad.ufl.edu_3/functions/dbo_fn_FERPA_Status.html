CREATE FUNCTION [dbo].[fn_FERPA_Status] 
(

)

/**************************************************************************************************

  Name:		        fn_FERPA_Status

  Description:	    Returns information about a user's FERPA Certification status
  
					PARAM NAME      DATATYPE		REQUIRED    NOTES
  Input:			@ufid			varchar(8)		Yes			UF identifier (8-char)
  
  Output:           Positive Integer	= Valid, number of days until certificate expires
					0					= Invalid, no certification on record
					Negative Integer	= Invalid, number of days since certificate expired
  
  Usage:            [BaseViews].[dbo].[fn_FERPA_Status](@ufid)
  
                    Example: SELECT [BaseViews].[dbo].[fn_FERPA_Status]('00000000')
                    Returns: 0

  Created:	        05/21/2015 by John Holmes

  Modified:         01/04/2017 by John Holmes
					Reference bv_FERPACertifications_simple
  	
**************************************************************************************************/

RETURNS [integer]

AS

BEGIN
	
	-- Declare the return variable
	DECLARE @result AS [integer]  ;
	
	-- Do the work
	SELECT 
		@result = [DaysUntilExpiration]
	FROM
		[BaseViews].[dbo].[bv_FERPACertifications_simple]
	WHERE
		[UFID] = @UFID 
	;

	IF @result = 0
		SET @result = -1;
	ELSE IF @result IS NULL 
		SET @result = 0
	;
	
	-- Return the result
	RETURN @result ;
	
END ;
