/**************************************************************************************************

  Name:		        fn_GetGradeByType_tbl

  Description:	    Returns all grades of the specified type.
  
  Input:            @gradetype:  ['P' (pass) | 'F' (fail) | 'O' (other) | W' (withdraw)]
  
  Output:           A table of grades of the specified type
  
  Usage:            SELECT * FROM BaseViews.dbo.fn_GetGradeByType_tbl('P')
                    
  Created:	        10/22/2009 by John Holmes

  Modified:         06/28/2011 by John Holmes
                    Add alias for dv_Grades
  	
**************************************************************************************************/

RETURNS 
@results TABLE(
    [Grade] VARCHAR(2)
)

AS 

BEGIN

    -- Init working var
    DECLARE @type AS VARCHAR(8)
    SET @type = 
        CASE @gradetype
            WHEN 'P' THEN 'Pass'
            WHEN 'F' THEN 'Fail'
            WHEN 'O' THEN 'Other'
            ELSE 'Withdraw'
        END
        
    
    -- Insert Results into table
    INSERT INTO 
        @results(
            [Grade]
        )
    SELECT 
        [g].[Grade]
    FROM 
        [BaseViews].[dbo].[dv_Grades] AS [g]
    WHERE 
        [g].[GradeType] = @type
    
    
    RETURN
END
