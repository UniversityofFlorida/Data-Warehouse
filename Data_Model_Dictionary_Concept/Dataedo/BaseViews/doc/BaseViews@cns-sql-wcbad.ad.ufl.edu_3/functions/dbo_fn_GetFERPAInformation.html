CREATE FUNCTION [dbo].[fn_GetFERPAInformation] (@User VARCHAR(70) = NULL)

/**************************************************************************************************

  Name:		        fn_GetFERPAInformation

  Description:	    Retrieve FERPA status information
  
  Input:            @User - Can be a Gatorlink (e.g. 'Alberta'), a UFID ('12345678') or NULL, in which case
					the user will be set to the currently connected user account
  
  Output:            
  
  Usage:            SELECT * FROM [BaseViews].[dbo].[fn_GetFerpaInformation]('alberta')
					SELECT * FROM [BaseViews].[dbo].[fn_GetFerpaInformation]('44609180')
					SELECT * FROM [BaseViews].[dbo].[fn_GetFerpaInformation](NULL)
                    
  Created:	        08/05/2010 by William McCombie

  Modified:         
  	
**************************************************************************************************/
RETURNS @MyInfo TABLE
    (
     UFID VARCHAR(8)
    ,GatorLink VARCHAR(70)
    ,IsFerpaCertified BIT
    )
AS 
    BEGIN
    
        DECLARE @UFID VARCHAR(8)
        DECLARE @GatorLink VARCHAR(70)
		DECLARE @IsFerpaCertified bit
    
        IF @User IS NULL --No UFID/Gatorlink, so assume currently logged in user
            BEGIN
                SET @GatorLink = (
                                  SELECT
                                    SUBSTRING(SYSTEM_USER,
                                              CHARINDEX('\', SYSTEM_USER) + 1,
                                              70)
                                 )
    
                SELECT
                    @UFID = [People].[UFID]
                FROM
                    [BaseViews].[dbo].[bv_UFDirPeople] AS [People]
                WHERE
                    [People].[GatorLinkID] = @GatorLink
    
    
            END
        ELSE 
            IF @User LIKE '[0-9][0-9][0-9][0-9][0-9][0-9][0-9][0-9]' --UFID
                BEGIN
                    SELECT
						@UFID = @User,
                        @GatorLink = [People].[GatorLinkID]
                    FROM
                        [BaseViews].[dbo].[bv_UFDirPeople] AS [People]
                    WHERE
                        [People].[UFID] = @User
	    
                END
            ELSE 
                BEGIN
                        
                    SELECT
                        @UFID = [People].[UFID],
                        @GatorLink = @User
                    FROM
                        [BaseViews].[dbo].[bv_UFDirPeople] AS [People]
                    WHERE
                        [People].[GatorLinkID] = @User
    
                END
	  

			
		
			SET @IsFerpaCertified = [BaseViews].[dbo].[fn_FERPA_IsCertValid](@UFID)
			
			INSERT INTO @MyInfo
			(UFID,
			[GatorLink],
			IsFerpaCertified
			)
			SELECT @UFID, @GatorLink, @IsFerpaCertified
			
	--		SELECT @IsFerpaCertified
    
        RETURN
    END
