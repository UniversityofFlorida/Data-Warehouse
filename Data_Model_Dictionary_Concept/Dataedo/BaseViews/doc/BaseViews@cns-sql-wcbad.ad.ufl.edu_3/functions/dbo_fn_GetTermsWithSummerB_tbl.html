CREATE FUNCTION [dbo].[fn_GetTermsWithSummerB_tbl] ()

/**************************************************************************************************

  Name:		        fn_GetTermsWithSummerB_tbl

  Description:	    Returns a modified version of the Terms that includes Summer B terms
  
  Input:            None.
  
  Output:           Previous, Current, Next and NextNext terms. 
  
  Usage:            SELECT * FROM BaseViews.dbo.fn_GetTermsWithSummerB_tbl()
                    
  Created:	        08/28/2009 by William McCombie

  Modified:         09/25/2009 by John Holmes
                    Moved to BaseViews
  	
**************************************************************************************************/

RETURNS 
@Result TABLE(
    [Term6] VARCHAR(6)
    ,[TermName] varchar(15)
    ,[Status] varchar(2)
)

AS 

BEGIN
    DECLARE @currentTerm varchar(6)
    DECLARE @currentYear int

    SET @currentTerm = [BaseViews].[dbo].[fn_GetTerm]('C', 6)
    SET @currentyear = CAST(LEFT(@currentTerm, 4) AS INT)

    declare @lastTerm varchar(6)
    DECLARE @nextTerm VARCHAR(6)
    DECLARE @nextNextTerm VARCHAR(6)


    IF RIGHT(@currentterm, 1) = '5' AND DATEPART(MONTH, GETDATE()) IN ('7', '8') 
        BEGIN --summer B
            SET @currentTerm = LEFT(@currentTerm, 4) + '06'
            SET @lastTerm = left(@currentTerm, 4) + '05'
            SET @nextTerm = left(@currentTerm, 4) + '08'
            SET @nextNextTerm = cast(@currentYear + 1 AS varchar) + '01'
    	
        END
    ELSE 
        IF RIGHT(@currentTerm, 1) = '5' AND DATEPART(MONTH, GETDATE()) IN ('5', '6') 
            BEGIN --summer A
                SET @currentTerm = LEFT(@currentTerm, 4) + '05'
                SET @lastTerm = left(@currentTerm, 4) + '01'
                SET @nextTerm = left(@currentTerm, 4) + '06'
                SET @nextNextTerm = left(@currentTerm, 4) + '08'

            END
        ELSE 
            IF RIGHT(@currentTerm, 1) = '8' 
                BEGIN --fall
                    SET @lastTerm = left(@currentTerm, 4) + '06'
                    SET @nextTerm = cast(@currentYear + 1 AS varchar) + '01'
                    SET @nextNextTerm = cast(@currentYear + 1 AS varchar) + '05'
                END
            ELSE 
                IF RIGHT(@currentTerm, 1) = '1' 
                    BEGIN --spring
                        SET @lastTerm = cast(@currentYear - 1 AS varchar) + '08'
                        SET @nextTerm = left(@currentTerm, 4) + '05'
                        SET @nextNextTerm = left(@currentTerm, 4) + '06'
                    END 


    INSERT INTO
        @result
    SELECT
        [Term6]
       ,CASE WHEN RIGHT([Term6], 1) = '1'
             THEN 'Spring ' + LEFT([Term6], 4)
             WHEN RIGHT([Term6], 1) = '5'
             THEN 'Summer A/C ' + LEFT([Term6], 4)
             WHEN RIGHT([Term6], 1) = '8' THEN 'Fall ' + LEFT([Term6], 4)
        END AS [TermName]
       ,(CASE WHEN [Term6] = @CurrentTerm THEN 'C'
              WHEN [Term6] = @LastTerm THEN 'P'
              WHEN [Term6] = @NextTerm THEN 'N'
              WHEN [Term6] = @NextNextTerm THEN 'NN'
              ELSE ''
         END) AS [Status]
    FROM
        [BaseViews].[dbo].[dv_Terms]
    UNION
    SELECT
        LEFT([Term6], 4) + '06'
       ,'Summer B/C ' + LEFT([Term6], 4) AS [TermName]
       ,(CASE WHEN LEFT([Term6], 4) + '06' = @CurrentTerm THEN 'C'
              WHEN LEFT([Term6], 4) + '06' = @LastTerm THEN 'P'
              WHEN LEFT([Term6], 4) + '06' = @NextTerm THEN 'N'
              WHEN LEFT([Term6], 4) + '06' = @NextNextTerm THEN 'NN'
              ELSE ''
         END) AS [Status]
    FROM
        [BaseViews].[dbo].[dv_Terms]
    WHERE
        RIGHT([Term6], 1) = '5'


    RETURN
END
