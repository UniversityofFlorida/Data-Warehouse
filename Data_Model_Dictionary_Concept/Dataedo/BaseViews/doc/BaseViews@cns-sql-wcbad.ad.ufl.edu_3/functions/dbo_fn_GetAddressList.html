CREATE FUNCTION [dbo].[fn_GetAddressList]
    (
     @UFIDs XML
    ,@AddressType VARCHAR(1) = 'L'
    )

/**************************************************************************************************

  Name:		        fn_GetAddressList

  Description:	    Given a list of UFIDs and a an address type, return a list of addresses
  
  Input:            UFIDs in XML format
							Create with SELECT UFID FROM <table> FOR XML RAW('Person'), ROOT('People')
					AddressType - 'L' for Local, 'P' for Permanent
					
  Output:            
  
  Usage:            SELECT * FROM [BaseViews].[dbo].[fn_GetAddressList](UFIDs, AddressType)
                    
  Created:	        04/22/2010 by William McCombie

  Modified:         06/15/2010 by John Holmes
                    Return ALL addresses (not just active).
                    
                    06/21/2010 by John Holmes
                    Add Emergency Contact (Type 23) as a last resort to both Local and Permanent.
  	
**************************************************************************************************/
RETURNS @Addresses TABLE
    (
     [UFID] VARCHAR(8)
    ,[Address] VARCHAR(44)
    ,[City] VARCHAR(30)
    ,[State] VARCHAR(2)
    ,[Zip] VARCHAR(12)
    )
AS 
    BEGIN
    
    /* Example code to create XML parameter using UF Directory data
      DECLARE @UFIDs XML
	  SET @UFIDs = (SELECT UFID FROM [BaseViews].[dbo].[bv_UFDIRPeopleActive] FOR XML RAW('Person'), ROOT('People'))
	*/
    
    
    -- Insert Results into table
        INSERT  INTO @Addresses
                (
                 UFID
                )
                SELECT
                    UFIDs.Person.value('@UFID[1]', 'varchar(8)') AS UFID
                FROM
                    @UFIDs.nodes('//People/Person') AS UFIDs (Person)    
        
    /*
    2 = UF Business Mailing, 
    4 = Local Home Mailing, 
    5 = Local Home Physical, 
    6 = Permanent Home Mailing, 
    22 = Parents Mailing, 
    23 = Emergency Contact, 
    94 = Housing
	*/
	
    ----------------------------------------------------------------
	
	
        IF @AddressType = 'L'   -- Local Addresses
            BEGIN
    	        -- Preferred address is Type 4 = Local Home Mailing
                UPDATE
                    @Addresses
                SET 
                    [Address] = [BaseViews].[dbo].[fn_ProperCaseAddress]([addr].[Address])
                   ,[City] = [BaseViews].[dbo].[fn_ProperCase]([addr].[City])
                   ,[State] = [addr].[State]
                   ,[Zip] = [addr].[Zip]
                FROM
                    @Addresses AS [r]
                LEFT JOIN [BaseViews].[dbo].[bv_UFDirPeopleAddress] AS [addr]
                    ON [r].[UFID] = [addr].[UFID]
                WHERE
                    [addr].[Type] = 4
                
                
                -- Add type 5 addresses if address still blank
                -- Type 5 = Local home physical
                UPDATE
                    @Addresses
                SET 
                    [Address] = [BaseViews].[dbo].[fn_ProperCaseAddress]([addr].[Address])
                   ,[City] = [BaseViews].[dbo].[fn_ProperCase]([addr].[City])
                   ,[State] = [addr].[State]
                   ,[Zip] = [addr].[Zip]
                FROM
                    @Addresses AS [r]
                LEFT JOIN [BaseViews].[dbo].[bv_UFDirPeopleAddress] AS [addr]
                    ON [r].[UFID] = [addr].[UFID]
                WHERE
                    [addr].[Type] = 5
                    AND COALESCE([r].[Address], '') = ''
        
        
                -- Add type 2 addresses if address still blank
                -- Type 2 = UF Business Mailing
                UPDATE
                    @Addresses
                SET 
                    [Address] = [BaseViews].[dbo].[fn_ProperCaseAddress]([addr].[Address])
                   ,[City] = [BaseViews].[dbo].[fn_ProperCase]([addr].[City])
                   ,[State] = [addr].[State]
                   ,[Zip] = [addr].[Zip]
                FROM
                    @Addresses AS [r]
                LEFT JOIN [BaseViews].[dbo].[bv_UFDirPeopleAddress] AS [addr]
                    ON [r].[UFID] = [addr].[UFID]
                WHERE
                    [addr].[Type] = 2
                    AND COALESCE([r].[Address], '') = ''
        
  
                -- Add type 94 addresses  if address still blank
                -- Type 94 = UF Housing
                UPDATE
                    @Addresses
                SET 
                    [Address] = [BaseViews].[dbo].[fn_ProperCaseAddress]([addr].[Address])
                   ,[City] = [BaseViews].[dbo].[fn_ProperCase]([addr].[City])
                   ,[State] = [addr].[State]
                   ,[Zip] = [addr].[Zip]
                FROM
                    @Addresses AS [r]
                LEFT JOIN [BaseViews].[dbo].[bv_UFDirPeopleAddress] AS [addr]
                    ON [r].[UFID] = [addr].[UFID]
                WHERE
                    [addr].[Type] = 94
                    AND COALESCE([r].[Address], '') = ''
                    
                -- Add type 6 addresses if address still blank
                -- Type 6 = Permanent Home Mailing
                UPDATE
                    @Addresses
                SET 
                    [Address] = [BaseViews].[dbo].[fn_ProperCaseAddress]([addr].[Address])
                   ,[City] = [BaseViews].[dbo].[fn_ProperCase]([addr].[City])
                   ,[State] = [addr].[State]
                   ,[Zip] = [addr].[Zip]
                FROM
                    @Addresses AS [r]
                LEFT JOIN [BaseViews].[dbo].[bv_UFDirPeopleAddress] AS [addr]
                    ON [r].[UFID] = [addr].[UFID]
                WHERE
                    [addr].[Type] = 6
                    AND COALESCE([r].[Address], '') = ''
                    
                -- Add type 23 addresses if address still blank
                -- Type 23 = Emergency Contact
                UPDATE
                    @Addresses
                SET 
                    [Address] = [BaseViews].[dbo].[fn_ProperCaseAddress]([addr].[Address])
                   ,[City] = [BaseViews].[dbo].[fn_ProperCase]([addr].[City])
                   ,[State] = [addr].[State]
                   ,[Zip] = [addr].[Zip]
                FROM
                    @Addresses AS [r]
                LEFT JOIN [BaseViews].[dbo].[bv_UFDirPeopleAddress] AS [addr]
                    ON [r].[UFID] = [addr].[UFID]
                WHERE
                    [addr].[Type] = 23
                    AND COALESCE([r].[Address], '') = ''
            
            END -- End Local Addresses
            
        ELSE IF @AddressType = 'P'   -- Permanent Addresses
            BEGIN
                
                -- Preferred address is Type 6 = Permanent Home Mailing
                UPDATE
                    @Addresses
                SET 
                    [Address] = [BaseViews].[dbo].[fn_ProperCaseAddress]([addr].[Address])
                   ,[City] = [BaseViews].[dbo].[fn_ProperCase]([addr].[City])
                   ,[State] = [addr].[State]
                   ,[Zip] = [addr].[Zip]
                FROM
                    @Addresses AS [r]
                LEFT JOIN [BaseViews].[dbo].[bv_UFDirPeopleAddress]
                    AS [addr]
                    ON [r].[UFID] = [addr].[UFID]
                WHERE
                    [addr].[Type] = 6
    
    
                -- Add type 22 addresses if address still blank
                -- Type 22 = Parents
                UPDATE
                    @Addresses
                SET 
                    [Address] = [BaseViews].[dbo].[fn_ProperCaseAddress]([addr].[Address])
                   ,[City] = [BaseViews].[dbo].[fn_ProperCase]([addr].[City])
                   ,[State] = [addr].[State]
                   ,[Zip] = [addr].[Zip]
                FROM
                    @Addresses AS [r]
                LEFT JOIN [BaseViews].[dbo].[bv_UFDirPeopleAddress]
                    AS [addr]
                    ON [r].[UFID] = [addr].[UFID]
                WHERE
                    [addr].[Type] = 22
                    AND COALESCE([r].[Address], '') = ''
    
    
                -- Add type 2 addresses if address still blank
                -- Type 2 = UF Business
                UPDATE
                    @Addresses
                SET 
                    [Address] = [BaseViews].[dbo].[fn_ProperCaseAddress]([addr].[Address])
                   ,[City] = [BaseViews].[dbo].[fn_ProperCase]([addr].[City])
                   ,[State] = [addr].[State]
                   ,[Zip] = [addr].[Zip]
                FROM
                    @Addresses AS [r]
                LEFT JOIN [BaseViews].[dbo].[bv_UFDirPeopleAddress]
                    AS [addr]
                    ON [r].[UFID] = [addr].[UFID]
                WHERE
                    [addr].[Type] = 2
                    AND COALESCE([r].[Address], '') = ''
                
                -- Add type 23 addresses if address still blank
                -- Type 23 = Emergency Contact
                UPDATE
                    @Addresses
                SET 
                    [Address] = [BaseViews].[dbo].[fn_ProperCaseAddress]([addr].[Address])
                   ,[City] = [BaseViews].[dbo].[fn_ProperCase]([addr].[City])
                   ,[State] = [addr].[State]
                   ,[Zip] = [addr].[Zip]
                FROM
                    @Addresses AS [r]
                LEFT JOIN [BaseViews].[dbo].[bv_UFDirPeopleAddress] AS [addr]
                    ON [r].[UFID] = [addr].[UFID]
                WHERE
                    [addr].[Type] = 23
                    AND COALESCE([r].[Address], '') = ''
            END
       
        RETURN
    END
