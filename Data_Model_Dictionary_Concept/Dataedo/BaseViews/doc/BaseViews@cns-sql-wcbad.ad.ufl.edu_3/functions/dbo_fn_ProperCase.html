CREATE FUNCTION [dbo].[fn_ProperCase] 
(
    @Text AS VARCHAR(8000)
)

/**************************************************************************************************

  Name:		        fn_ProperCase

  Description:	    Convert a text string to proper case.
  
  Input:            @Text:  The string to convert
  
  Usage:            SELECT BaseViews.dbo.fn_ProperCase('some text string')
  
                    Example: SELECT BaseViews.dbo.fn_ProperCase('some text string')
                    Returns: 'Some Text String'

  Created:	        07/09/2009 by John Holmes

  Modified:         10/06/2014 by John Holmes
                    Add generational suffix corrections
  	
**************************************************************************************************/

RETURNS VARCHAR(8000)

AS 

BEGIN
    DECLARE @Reset BIT ;
    DECLARE @Ret VARCHAR(8000) ;
    DECLARE @i INT ;
    DECLARE @c CHAR(1) ;

    SELECT
        @Reset = 1,
        @i = 1,
        @Ret = '' ;
   
    WHILE (@i <= LEN(@Text))
        SELECT
            @c = SUBSTRING(@Text, @i, 1),
            @Ret = @Ret + CASE WHEN @Reset = 1 THEN UPPER(@c)
                               ELSE LOWER(@c)
                          END,
            @Reset = CASE WHEN @c LIKE '[a-zA-Z]' THEN 0
                          ELSE 1
                     END,
            @i = @i + 1

    -- Correct generational suffixes
    SET @Ret = REPLACE(@Ret, ' Ii', ' II') ;
    SET @Ret = REPLACE(@Ret, ' Iii', ' III') ;
    SET @Ret = REPLACE(@Ret, ' Iv', ' IV') ;
    SET @Ret = REPLACE(@Ret, ' Vi', ' VI') ;
    SET @Ret = REPLACE(@Ret, ' Vii', ' VII') ;

    RETURN @Ret ;
END
