RETURN
	WITH  
		 [L0] AS (SELECT 1 AS C UNION ALL SELECT 1)			--			  2 rows
		,[L1] AS (SELECT 1 AS C FROM [L0] AS A, [L0] AS B)	--			  4 rows

		,[tally] AS (SELECT ROW_NUMBER() OVER(ORDER BY (SELECT NULL)) AS [n] FROM [L5])
	SELECT TOP(@n) [n] FROM [tally]
