CREATE FUNCTION [dbo].[fn_ConvertTerm] 
(
    @termIN AS VARCHAR(20),
    @type AS INT
)

/**************************************************************************************************

  Name:		        fn_ConvertTerm

  Description:	    BaseViews - Convert 3, 4, 5 or 6 char Term string into another format.
  
  Input:            @termIN:    Character string representing the term. 2 < LEN < 7
                    @type:      Integer representing output format (num char) [ 3 | 4 | 5 | 6 ]
  
  Usage:            BaseViews.dbo.fn_ConvertTerm('Term_To_Convert', Type)
  
                    Example: SELECT BaseViews.dbo.fn_ConvertTerm('200608', 3)
                    Returns: '068'

  Created:	        09/25/2009 by John Holmes

  Modified:
  	
**************************************************************************************************/

RETURNS VARCHAR(6)

AS

BEGIN
    
	-- Declare and initialize the return and working variables
	DECLARE @termOUT AS VARCHAR(6)
	DECLARE @term6 AS VARCHAR(6)
	SET @termOUT = ''
	SET @term6 = ''
	
	
	-- Clean up the input
	SET @termIN = BaseViews.dbo.fn_LRTrim(@termIN)
	
	
	-- Error Checking: input length
    IF (LEN(@termIN) < 3 OR LEN(@termIN) > 6)
    BEGIN
        IF(LEN(LTRIM(RTRIM(@termIN))) = 0)
        BEGIN
            SET @termOUT = ''
        END
        ELSE SET @termOUT = 'INPERR'
    END
    
    
    -- Error Checking: type specification
    ELSE IF (@type <3 OR @type > 6)
    BEGIN
        SET @termOUT = 'TYPERR'
    END
    
    
    -- If no input errors, perform conversion
    ELSE
    BEGIN
        -- STEP 1: Convert to 6-char format
        SET @term6 = 
	        CASE
	            -- 3 char input
	            WHEN LEN(@termIN) = 3 
	            THEN (CASE WHEN LEFT(@termIN,2) > 40 THEN '19' ELSE '20' END) + LEFT(@termIN,2) + '0' + RIGHT(@termIN, 1)
	            
	            -- 4 char input
	            WHEN LEN(@termIN) = 4 
	            THEN (CASE WHEN LEFT(@termIN,2) > 40 THEN '19' ELSE '20' END) + LEFT(@termIN,2) + RIGHT(@termIN, 2)
	            
	            -- 5 char input
	            WHEN LEN(@termIN) = 5 
	            THEN LEFT(@termIN,4) + '0' + RIGHT(@termIN, 1)
	            
	            -- 6 char input
	            ELSE @termIN
            END
            
        -- STEP 2: Convert to requested format
        SET @termOUT = 
            CASE
                -- 3 char output
                WHEN @type = 3
                THEN RIGHT(LEFT(@term6, 4), 2) + RIGHT(@term6, 1)
                
                -- 4 char output
                WHEN @type = 4
                THEN RIGHT(LEFT(@term6, 4), 2) + RIGHT(@term6, 2)
                
                -- 5 char output
                WHEN @type = 5
                THEN LEFT(@term6, 4) + RIGHT(@term6, 1)
                
                -- 6 char output
                ELSE @term6 
            END
	END
	
	
	-- Return the result
	RETURN @termOUT 
	
END
