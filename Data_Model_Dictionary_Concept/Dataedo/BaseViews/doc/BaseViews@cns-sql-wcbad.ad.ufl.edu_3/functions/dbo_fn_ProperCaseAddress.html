CREATE FUNCTION [dbo].[fn_ProperCaseAddress] (
    @input AS VARCHAR(255)
)

RETURNS VARCHAR(255)

AS 

/**************************************************************************************************

  Name:		        fn_ProperCaseAddress

  Description:	    Proper case an address string with correct quandrant (NW, SE, etc.) and street
                    numbering (1st, 207th, etc.) casing.
  
  Input:            @input:  an address string
  
  Usage:            BaseViews.dbo.fn_ProperCaseAddress(@input)
  
                    Example: SELECT BaseViews.dbo.fn_ProperCaseAddress('1000 SW 8TH AVE')
                    Returns: '1000 SW 8th Ave'

  Created:	        10/22/2009 by John Holmes

  Modified:         07/15/2011 by John Holmes
                    Add 'Po' to 'PO' conversion.
  	
**************************************************************************************************/

BEGIN

    -- Declare the return variable
    DECLARE @out AS VARCHAR(255)

    -- Cleanse the input
    SET @out = [BaseViews].[dbo].[fn_LRTrim](@input)

    -- Apply proper casing
    SET @out = [BaseViews].[dbo].[fn_ProperCase](@out)
    
    -- Replace Po with PO
    SET @out = REPLACE(@out, 'Po ', 'PO ')

    -- Replace Nw, Ne, Sw, and Se with upper case if surrounded by spaces
    SET @out = REPLACE(@out, ' Nw ', ' NW ')
    SET @out = REPLACE(@out, ' Sw ', ' SW ')
    SET @out = REPLACE(@out, ' Ne ', ' NE ')
    SET @out = REPLACE(@out, ' Se ', ' SE ')
    SET @out = REPLACE(@out, ' Po ', ' PO ')
    SET @out = REPLACE(@out, ' Rr ', ' RR ')


    -- Replace Nw, Ne, Sw, and Se with upper case if at the end of address
    SET @out = REPLACE(@out, ' Nw', ' NW')
    SET @out = REPLACE(@out, ' Sw', ' SW')
    SET @out = REPLACE(@out, ' Ne', ' NE')
    SET @out = REPLACE(@out, ' Se', ' SE')

    -- Replace St, Nd, Rd and Th with lower case
    SET @out = REPLACE(@out, '1St ', '1st ')
    SET @out = REPLACE(@out, '2Nd ', '2nd ')
    SET @out = REPLACE(@out, '3Rd ', '3rd ')
    SET @out = REPLACE(@out, '4Th ', '4th ')
    SET @out = REPLACE(@out, '5Th ', '5th ')
    SET @out = REPLACE(@out, '6Th ', '6th ')
    SET @out = REPLACE(@out, '7Th ', '7th ')
    SET @out = REPLACE(@out, '8Th ', '8th ')
    SET @out = REPLACE(@out, '9Th ', '9th ')
    SET @out = REPLACE(@out, '0Th ', '0th ')

    -- Return the result
    RETURN @out

END
