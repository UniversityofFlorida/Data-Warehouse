CREATE FUNCTION [dbo].[fn_GetTermDate]
(
    @term varchar(2),
    @datetype varchar(20)
)

/**************************************************************************************************

  Name:		        fn_GetTermDate

  Description:	    Returns the specified date for the (P)revious, (C)urrent, (N)ext or 
                    (NN)extNext term.
                    
  Input:            @term:      Character representing requested term [P | C | N | NN ]
                    @datetype:  Character representing requested date type:
                                'TermStart' | 'TermEnd' | 'ClassStart' | 'ClassEnd' | 'DropAddEnd'
  
  Output:           datetime2
                    Input error conditions return NULL
  
  Usage:            BaseViews.dbo.fn_GetTermDate(@term,@datetype)
                    Example: 
                        SELECT BaseViews.dbo.fn_GetTermDate('C', 'DropAddEnd')
                        returns start date of current term

  Created:	        04/07/2016 by John Holmes

  Modified:			
					
  	
**************************************************************************************************/

RETURNS [datetime2](0)

AS

BEGIN
    
	DECLARE @out [datetime2](0) ;
	
	SET @term = [BaseViews].[dbo].[fn_LRTrim](@term) ;
	SET @datetype = [BaseViews].[dbo].[fn_LRTrim](@datetype) ;

	SET @out = 
		CASE @datetype
			WHEN 'TermStart' THEN (SELECT [termStart] FROM [BaseViews].[dbo].[dv_termsModules] WHERE [status] = @term)
			WHEN 'TermEnd' THEN (SELECT [termEnd] FROM [BaseViews].[dbo].[dv_termsModules] WHERE [status] = @term)
			WHEN 'ClassStart' THEN (SELECT [classesStart] FROM [BaseViews].[dbo].[dv_termsModules] WHERE [status] = @term)
			WHEN 'ClassEnd' THEN (SELECT [classesEnd] FROM [BaseViews].[dbo].[dv_termsModules] WHERE [status] = @term)
			WHEN 'DropAddEnd' THEN (SELECT [dropAddEnd] FROM [BaseViews].[dbo].[dv_termsModules] WHERE [status] = @term)
			ELSE NULL
		END ;
        
    RETURN @out ;
	
END ;
