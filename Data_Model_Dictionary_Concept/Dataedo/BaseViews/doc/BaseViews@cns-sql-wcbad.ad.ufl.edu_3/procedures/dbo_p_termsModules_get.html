CREATE PROCEDURE [dbo].[p_termsModules_get] 

	@type INTEGER = NULL

AS 

/**************************************************************************************************

  Name:		    p_termsModules_get

  Description:  Returns term and module date data for Previous, Current, Next and NextNext terms.
  
				PARAM NAME      DATATYPE        REQUIRED    NOTES
  Input:		@type			integer			No			Type code
															NULL (default)	- All data
															1				- Grad only
															2				- Undergrad only
                                  
  Output:       A record set
  
  Usage:        EXEC [BaseViews].[dbo].[p_termsModules_get] @type = 2
  
  Created:	    07/19/2016 by John Holmes

  Modified:		02/16/2017 by John Holmes
				Add @type input parameter

				05/22/2017 by John Holmes
				Add gsMidpointOfTerm
  	
**************************************************************************************************/

BEGIN

    SET NOCOUNT ON ;

	IF @type = 1	-- Grad only
		SELECT
			 [tm].[term6]
			,[tm].[baseTerm6]
			,[tm].[termName]
			,[tm].[moduleType]
			,[s].[status]
			,[tm].[priorityRegistrationStart]
			,[tm].[regularRegistrationStart]
			,[tm].[termStart]
			,[tm].[termEnd]
			,[tm].[classesStart]
			,[tm].[classesEnd]
			,[tm].[dropAddStart]
			,[tm].[dropAddEnd]
			,[tm].[degreeApplicationDeadline]
			,[tm].[feesDue]
			,[tm].[dropDeadline]
			,[tm].[breakStart]
			,[tm].[breakEnd]
			,[tm].[finalsStart]
			,[tm].[finalsEnd]
			,tm.gsMidpointOfTerm
		FROM
			[dbo].[dv_termsModules] AS [tm]
		JOIN 
			(SELECT [baseTerm6], [status] FROM [dbo].[dv_termsModules] WHERE [status] IS NOT NULL) AS [s]
			ON [s].[baseTerm6] = [tm].[baseTerm6]
		WHERE 
			[tm].[baseTerm6] IN (
				SELECT [term6] FROM [dbo].[dv_termsModules] WHERE [status] IS NOT NULL
			) 
			AND tm.moduleTypeID IN (2,4);

	ELSE IF @type = 2	-- Undergrad only
		SELECT
			 [tm].[term6]
			,[tm].[baseTerm6]
			,[tm].[termName]
			,[tm].[moduleType]
			,[s].[status]
			,[tm].[advRegistrationStart]
			,[tm].[termStart]
			,[tm].[termEnd]
			,[tm].[classesStart]
			,[tm].[classesEnd]
			,[tm].[dropAddStart]
			,[tm].[dropAddEnd]
			,[tm].[degreeApplicationDeadline]
			,[tm].[feesDue]
			,[tm].[dropDeadline]
			,[tm].[breakStart]
			,[tm].[breakEnd]
			,[tm].[finalsStart]
			,[tm].[finalsEnd]
			,tm.gsMidpointOfTerm
		FROM
			[dbo].[dv_termsModules] AS [tm]
		JOIN 
			(SELECT [baseTerm6], [status] FROM [dbo].[dv_termsModules] WHERE [status] IS NOT NULL) AS [s]
			ON [s].[baseTerm6] = [tm].[baseTerm6]
		WHERE 
			[tm].[baseTerm6] IN (
				SELECT [term6] FROM [dbo].[dv_termsModules] WHERE [status] IS NOT NULL
			)
			AND tm.moduleTypeID != 2 ;

	ELSE	-- All data
		SELECT
			 [tm].[term6]
			,[tm].[baseTerm6]
			,[tm].[termName]
			,[tm].[moduleType]
			,[s].[status]
			,[tm].[advRegistrationStart]
			,[tm].[priorityRegistrationStart]
			,[tm].[regularRegistrationStart]
			,[tm].[termStart]
			,[tm].[termEnd]
			,[tm].[classesStart]
			,[tm].[classesEnd]
			,[tm].[dropAddStart]
			,[tm].[dropAddEnd]
			,[tm].[degreeApplicationDeadline]
			,[tm].[feesDue]
			,[tm].[dropDeadline]
			,[tm].[breakStart]
			,[tm].[breakEnd]
			,[tm].[finalsStart]
			,[tm].[finalsEnd]
			,tm.gsMidpointOfTerm
		FROM
			[dbo].[dv_termsModules] AS [tm]
		JOIN 
			(SELECT [baseTerm6], [status] FROM [dbo].[dv_termsModules] WHERE [status] IS NOT NULL) AS [s]
			ON [s].[baseTerm6] = [tm].[baseTerm6]
		WHERE 
			[tm].[baseTerm6] IN (
				SELECT [term6] FROM [dbo].[dv_termsModules] WHERE [status] IS NOT NULL
			) ;

END
