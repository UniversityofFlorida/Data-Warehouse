CREATE PROCEDURE [dbo].[p_snoopy_studentsRegForGradCourses_get] 
    
    @college varchar(2) = NULL

AS 

/**************************************************************************************************

  Name:			p_snoopy_studentsRegForGradCourses_get
				
  Description:	GLIDs of students registered for a graduate level course in the current term
				
				PARAM NAME		DATATYPE    REQUIRED    NOTES
  Input:		@college		VARCHAR(2)	No			Restrict output to students registered
														for graduate level courses in a specified college
														NULL (default) = Restricts to Business (BA, AC) courses.
				
  Usage:		EXEC [BaseViews].[dbo].[p_snoopy_studentsRegForGradCourses_get]
				EXEC [BaseViews].[dbo].[p_snoopy_studentsRegForGradCourses_get] 'BA'
				
  Created:		01/18/2017 by John Holmes
				
  Modified:		01/05/2018 by John Holmes
				Use bv_UFDirGLIDActive
				
**************************************************************************************************/

BEGIN

	-- For testing
	--DECLARE @college VARCHAR(2) = 'BA' ;
	
	SET NOCOUNT ON ;

	IF @college IS NULL
		SELECT
			g.GLID
		FROM
			BaseViews.dbo.bv_UFSections AS s
		INNER JOIN 
			BaseViews.dbo.bv_StudentRegistration AS r
			ON s.Section = r.Section
			AND s.Term6 = r.Term6
		LEFT JOIN 
			BaseViews.dbo.bv_UFDirGLIDActive AS g
			ON r.UFID = g.UFID
		WHERE
			s.Term6 = BaseViews.dbo.fn_GetTerm('C',6)
			AND s.course LIKE '___[56789]%'
			AND s.College IN ('BA','AC')
		GROUP BY
			g.GLID
	ELSE 
		SELECT
			g.GLID
		FROM
			BaseViews.dbo.bv_UFSections AS s
		INNER JOIN 
			BaseViews.dbo.bv_StudentRegistration AS r
			ON s.Section = r.Section
			AND s.Term6 = r.Term6
		LEFT JOIN 
			BaseViews.dbo.bv_UFDirGLIDActive AS g
			ON r.UFID = g.UFID
		WHERE
			s.Term6 = BaseViews.dbo.fn_GetTerm('C',6)
			AND s.course LIKE '___[56789]%'
			AND s.College = @college
		GROUP BY
			g.GLID
	;

END ;
