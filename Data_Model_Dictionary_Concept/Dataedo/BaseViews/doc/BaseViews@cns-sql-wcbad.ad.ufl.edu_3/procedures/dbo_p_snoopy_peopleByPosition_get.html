CREATE PROCEDURE [dbo].[p_snoopy_peopleByPosition_get] 

     @primaryOuID			INTEGER	= NULL 
	,@positionTypeID		INTEGER	= NULL
	,@positionCategoryID	INTEGER	= NULL
	
AS 

/**************************************************************************************************

  Name:			p_snoopy_peopleByPosition_get

  Description:	Return list of active WCB employee GLIDs who hold a position of the specified type 
				and/or category and/or have a primary affilation with the specifed OU
  
				PARAM NAME			DATATYPE    REQUIRED    NOTES
  Input:		@primaryOuID		integer		No			Primary affiliation OUID
															Null (default) = No OU restriction
															To view OUs:
															SELECT * FROM BaseViews.dbo.pv_wdir_OUs
				
				@positionTypeID		integer		No			Position typeID
															Null (default) = No Position Type restriction
															To view Position Types:
															SELECT * FROM warringtonDirectory.dbo.dv_positionTypes
				
				@positionCategoryID	integer		No			Position categoryID
															Null (default) = No Position Category restriction
															To view Position Categories:
															SELECT * FROM warringtonDirectory.dbo.dv_positionCategories

  Output:		A record set
				
  Usage:		Return TEAMS Exempt Staff whose primary affilition is ITSP :
					EXEC dbo.p_snoopy_peopleByPosition_get
						 @primaryOuID			= 3
						,@positionTypeID		= 2
						,@positionCategoryID	= 6
					;
				
  Created:		01/19/2017 by John Holmes
				
  Modified:		
  	
**************************************************************************************************/

BEGIN
	
	-- For Testing
	-- DECLARE @primaryOuID INTEGER = 3, @positionTypeID INTEGER = 2, @positionCategoryID INTEGER = 6 ;
	
	SET NOCOUNT ON ;
	
	SELECT 
		p.GLID AS GatorLinkID
	FROM
		dbo.pv_wdir_people_active AS p
	JOIN
		dbo.pv_wdir_positions_active AS pos
		ON pos.UFID = p.UFID
	WHERE 
		p.primaryAffiliationOUID = COALESCE(@primaryOuID, p.primaryAffiliationOUID)
		AND pos.positionTypeID = COALESCE(@positionTypeID, pos.positionTypeID)
		AND pos.positionCategoryID = COALESCE(@positionCategoryID, pos.positionCategoryID)
	;
    
END ;
