CREATE PROCEDURE [dbo].[p_sectionsWithInstructors] 

	@term6 VARCHAR(6) = NULL

AS

/**************************************************************************************************

  Name:				p_sectionsWithInstructors

  Description:		List of courses being taught for specified term, organized by department and
					including instructor information.
  
					PARAM NAME      DATATYPE        REQUIRED    NOTES
  Input:			@term6			VARCHAR(8)		No			6 char (YYYYMM) term code
																NULL (default) - Current term
  
  Usage:			EXEC [dbo].[p_sectionsWithInstructors] 
						@term6 = '201601'
					;
  
  Created:			10/20/2016 by John Holmes

  Modified:         

**************************************************************************************************/

BEGIN
	
    SET NOCOUNT ON;
	
	-- For testing
	--DECLARE @term6 VARCHAR(6) = NULL ;
    
	DECLARE @nextTerm6 VARCHAR(6) = COALESCE(@term6, (SELECT BaseViews.dbo.fn_GetTerm('C', 6))) ;
	
    SELECT
		 dept.OUShortName AS Department
		,sec.Term6
		,sec.Course
		,sec.CourseTitle
		,sec.Section
		,sec.ClassBeginDate
		,sec.ClassEndDate
		,mod.termName AS Module
		,sec.Cancelled
		,COALESCE(inst.InstructorName, 'STAFF') AS Instructor
		,spec.Description AS SpecialProgram
		,types.ProgramTypeDesc
    FROM
        BaseViews.dbo.bv_UFSections AS sec
    LEFT JOIN 
		BaseViews.dbo.dv_OrganizationalUnits AS dept
        ON sec.DepartmentCode = dept.DeptCode
    LEFT JOIN 
		BaseViews.dbo.bv_UFInstructorSection AS inst
        ON inst.Term6 = sec.Term6
		AND inst.Section = sec.Section
    LEFT JOIN 
		BaseViews.dbo.dv_SpecialPrograms AS spec
        ON sec.SpecialProgram = spec.SpecialProgCode
        AND(
            sec.Term6 >= spec.StartTerm6
            OR COALESCE(spec.StartTerm6, '') = ''
		)
        AND(
            sec.Term6 <= spec.EndTerm6
            OR COALESCE(spec.EndTerm6, '') = ''
		)
    LEFT JOIN 
		BaseViews.dbo.dv_Programtypes AS types
        ON spec.ProgramTypeID = types.ProgramTypeID
    LEFT JOIN 
		BaseViews.dbo.dv_termsModules AS mod
        ON mod.baseTerm6 = sec.Term6
        AND sec.ClassBeginDate >= mod.classesStart
        AND sec.ClassEndDate <= mod.classesEnd
        AND (
            (
                mod.moduleType = 'Grad Module'
                AND sec.Course LIKE '___[56789]%'
            )
            OR (
                mod.moduleType = 'UGrad Module'
                AND sec.Course LIKE '___[1234]%'
			)
            OR (mod.moduleType = 'UF Module')
		)
    WHERE
        sec.College IN ('BA', 'AC')
        AND sec.Term6 = @nextTerm6
	ORDER BY
		 Department
		,Course
		,Section
	;

END ;
