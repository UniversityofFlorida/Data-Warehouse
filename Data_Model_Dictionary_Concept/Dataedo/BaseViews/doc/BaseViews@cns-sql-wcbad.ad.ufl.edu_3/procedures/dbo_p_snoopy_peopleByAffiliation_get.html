CREATE PROCEDURE [dbo].[p_snoopy_peopleByAffiliation_get] 

     @affiliationTypeID	INTEGER	= NULL 
	,@affiliatedOuID	INTEGER	= NULL
	
AS 

/**************************************************************************************************

  Name:			p_snoopy_peopleByAffiliation_get

  Description:	Return list of active WCB employee GLIDs who have the specified affiliation type
				with the specified OU
  
				PARAM NAME			DATATYPE    REQUIRED    NOTES
  Input:		@affiliationTypeID	integer		No			Affiliation typeID
															To view affilation types:
															SELECT * FROM warringtonDirectory.dbo.dv_types WHERE categoryID = -7
				
				@affiliatedOuID		integer		No			Affiliated ouID
															To view OUs:
															SELECT * FROM BaseViews.dbo.pv_wdir_OUs

  Output:		A record set
				
  Usage:		Return active active people with a Directory affiliation with ITSP :
					EXEC [dbo].[p_snoopy_peopleByAffiliation_get] 
						 @affiliationTypeID	= -7001
						,@affiliatedOuID		= 3
					;
				
  Created:		01/18/2017 by John Holmes
				
  Modified:		06/15/2017 by John Holmes
				Make all parameters optional
				Correct parameter name spelling
				Do not return affiliations with Default OU
  	
**************************************************************************************************/

BEGIN
	
	-- For Testing
	-- DECLARE @affiliationTypeID INTEGER = -7002, @affiliatedOuID INTEGER = 3 ;
	
	SET NOCOUNT ON ;
	
	SELECT
		p.GLID AS GatorLinkID
	FROM
		dbo.pv_wdir_affiliations_active AS a
	JOIN 
		dbo.pv_wdir_people_active AS p
		ON p.UFID = a.UFID
	WHERE 
		a.affiliatedOuID = COALESCE(@affiliatedOuID, a.affiliatedOuID)
		AND a.typeID = COALESCE(@affiliationTypeID, a.typeID)
		AND a.affiliatedOuID != 0
	GROUP BY
		p.GLID
	;
    
END ;
