CREATE PROCEDURE [dbo].[pp_DirCompatibility_BuildingResidentsXML_get] 

	@Bldg [varchar](5) = NULL
	
AS 

/**************************************************************************************************

  Name:		        pp_DirCompatibility_BuildingResidentsXML_get

  Description:		Legacy directory compatibility procedure. Replaces WCB_Directory.dbo.p_GetBuildingResidentsXML
					Return contact info for residents of a specified building in XML format.

  Note:				This is a functional replacement for the legacy directory procedure above.
					It is used for BACKWARD COMPATIBILITY ONLY so that old 'n busted ASP.net web pages
					don't know the directory has been upgraded.
  
					PARAM NAME		DATATYPE    REQUIRED    NOTES
  Input:			@Bldg			varchar(5)	No          Building Code
															NULL (default) = No Building restriction
  
  Output:			A record set
  
  Usage:            EXEC [BaseViews].[dbo].[pp_DirCompatibility_BuildingResidentsXML_get] 'BRY'
  
  Created:	        04/26/2016 by John Holmes

  Modified:			01/17/2017 by John Holmes
					Honor publicDisplay and ufProtect flags
					Return only people with at least one Directory affiliation

					01/31/2017 by John Holmes
					Include area code in cphone

					02/14/2017 by John Holmes
					Return Warrington email if it exists, UF Email otherwise
  	
**************************************************************************************************/

BEGIN
	
	SET NOCOUNT ON ;

	-- For Testing
	--DECLARE @Bldg [varchar](5) = 'BRY' ;

	-- Translate building code into buildingID
	DECLARE @buildingID AS [integer] = (SELECT [buildingID] FROM [warringtonDirectory].[dbo].[dv_buildings] WHERE [code] = @Bldg) ;

	WITH 
		[ufids](
			 [UFID]
			 )
			AS (
				SELECT
					 pl.[UFID]
				FROM
					-- Active people with an office in @buildingID
					[warringtonDirectory].[dbo].[fn_people_list_get](1, NULL, NULL, NULL, NULL, NULL, @buildingID, NULL, NULL) AS pl
				JOIN (
					SELECT 
						UFID
					FROM 
						warringtonDirectory.dbo.v_affiliations
					WHERE 
						typeID = -7001		-- Directory listing affilation
					GROUP BY 
						UFID
				) AS aff
				ON aff.UFID = pl.UFID
			)
		,[people](
			 [UFID]
			,[ufBusinessName]
			,[ufDisplayName]
			,[ufWorkingTitle]
			,[salutation]
			,publicDisplay
			,ufProtect
			 )
			AS (
				SELECT
					 [UFID]
					,[ufBusinessName]
					,[ufDisplayName]
					,[ufWorkingTitle]
					,[salutation]
					,publicDisplay
					,ufProtect
				FROM
					[warringtonDirectory].[dbo].[v_people]
				WHERE 
					[active] = 1
			)
		,warrEmail(
			 UFID
			,EMAIL
			 )
			AS (
				SELECT
					 UFID
					,email
				FROM
					warringtonDirectory.dbo.v_email
				WHERE 
					typeID = -2001	-- Warrington Email
			)
		,ufEmail(
			 UFID
			,EMAIL
			 )
			AS (
				SELECT
					 UFID
					,email
				FROM
					warringtonDirectory.dbo.v_email
				WHERE 
					typeID = 1	-- UF Business Email
			)
		,[position](
			 [UFID]
			,[positionName]
			,[positionTypeID]
			,[positionTypeCode]
			)
			AS (
				SELECT
					 [UFID]
					,[positionName]
					,[positionTypeID]
					,[positionTypeCode]
				FROM
					[warringtonDirectory].[dbo].[v_positionAssignments]
			)
		,[office](
			 [UFID]
			,[room]
			,[buildingCode]
			,[buildingName]
			)
			AS (
				SELECT
					 [UFID]
					,[room]
					,[buildingCode]
					,[buildingName]
				FROM
					[warringtonDirectory].[dbo].[v_locations]
				WHERE 
					[typeID] = -5000
			)
		,[telephone](
			 [UFID]
			,[typeID]
			,[areaCode]
			,[phone]
			,[ext]
			)
			AS (
				SELECT
					 [UFID]
					,[typeID]
					,[areaCode]
					,[phone]
					,[ext]
				FROM
					[warringtonDirectory].[dbo].[v_telephone]
				WHERE 
					[typeID] = 10
			)
		,[primaryAff](
			 [UFID]
			,[ouID]
			,[ouName]
			,[ouCode]
			,[url]
			)
			AS (
				SELECT
					 [p].[UFID]
					,[o].[ouID]
					,[o].[name]
					,[o].[code]
					,[u].[url]
				FROM
					[warringtonDirectory].[dbo].[v_people] AS [p]
				JOIN
					[warringtonDirectory].[dbo].[dv_organizationalUnits] AS [o]
					ON [p].[primaryAffiliationOUID] = [o].[ouID]
				LEFT JOIN 
					[warringtonDirectory].[dbo].[v_urls] AS [u]
					ON [u].[ouID] = [o].[ouID]
			)
		,[urls](
			 [UFID]
			,[url]
			)
			AS (
				SELECT
					 [UFID]
					,[url]
				FROM
					[warringtonDirectory].[dbo].[v_urls]
				WHERE 
					[urlTypeID] = 2
			)
	SELECT 
		 COALESCE([people].[salutation], '') AS [Prefix]
		,[BaseViews].[dbo].[fn_GetUFBusinessNamePart]([people].[ufBusinessName], 'F') AS [FirstName]
		,LEFT([BaseViews].[dbo].[fn_GetUFBusinessNamePart]([people].[ufBusinessName], 'M'),1) AS [MiddleInitial]
		,[BaseViews].[dbo].[fn_GetUFBusinessNamePart]([people].[ufBusinessName], 'L') AS [LastName]
		,[BaseViews].[dbo].[fn_GetUFBusinessNamePart]([people].[ufBusinessName], 'S') AS [Suffix]
		,[BaseViews].[dbo].[fn_ProperCase](
			CASE
				WHEN [BaseViews].[dbo].[fn_GetUFBusinessNamePart]([people].[ufDisplayName],'F') != [BaseViews].[dbo].[fn_GetUFBusinessNamePart]([people].[ufBusinessName],'F')
					THEN [BaseViews].[dbo].[fn_GetUFBusinessNamePart]([people].[ufDisplayName],'F')
				ELSE ''
			END 
		 ) AS [Nickname]
        ,CASE 
            WHEN LEN(COALESCE([people].[ufWorkingTitle],'')) > 0 THEN [people].[ufWorkingTitle]
            ELSE [position].[positionName]
         END AS [Title]
		,[office].[room] AS [Room]
		,COALESCE(telephone.areaCode + '.', '') + COALESCE(LEFT([telephone].[phone],3) + '.' + RIGHT([telephone].[phone],4),'') AS [Phone]
		,COALESCE([urls].[url],'') AS [PhotoURL]
		,CASE 
			WHEN warrEmail.EMAIL IS NULL THEN ufEmail.EMAIL
			ELSE warrEmail.EMAIL
		 END AS [Email]
	FROM 
		[ufids]
	JOIN 
		[people]
		ON [ufids].[UFID] = [people].[UFID]
	LEFT JOIN  
		warrEmail
		ON [ufids].[UFID] = warrEmail.UFID
	LEFT JOIN  
		ufEmail
		ON [ufids].[UFID] = ufEmail.UFID
	LEFT JOIN 
		[position]
		ON [ufids].[UFID] = [position].[UFID]
	LEFT JOIN 
		[primaryAff]
		ON [ufids].[UFID] = [primaryAff].[UFID]
	LEFT JOIN 
		[office]
		ON [ufids].[UFID] = [office].[UFID]
	LEFT JOIN 
		[telephone]
		ON [ufids].[UFID] = [telephone].[UFID]
	LEFT JOIN 
		[urls]
		ON [ufids].[UFID] = [urls].[UFID]
	WHERE 
		[primaryAff].[ouID] > 0
		AND publicDisplay = 1
		AND ufProtect = 0
	ORDER BY
		 [LastName]
		,[FirstName]
		,[MiddleInitial]
		,[Suffix]
	FOR   
		XML PATH('Person')
			,ELEMENTS
	;

END ;
