CREATE PROCEDURE [dbo].[p_GetEmployeeList] 

     @Dept varchar(6) = NULL 
	,@Type varchar(6) = NULL 
	,@Active varchar(1) = NULL 
	
AS 

/**************************************************************************************************

  Name:		        p_GetEmployeeList

  Description:	    Return list of employee gatorlink usernames with non-NULL email addresses
  
					PARAM NAME		DATATYPE    REQUIRED    NOTES
  Input:			@Dept			varchar(6)	No			Department Code
															NULL (default) = No restriction

					@Type			varchar(6)	No			Employee Type [ FAC | STAFF | OPS ]
															NULL (default) = No restriction

					@Activew		varchar(1)	No			Active Flag
															NULL (default) = No restriction
															'Y'			   = Active only
															'N'			   = Inactive only  
  
  Output:			A record set
  
  Usage:			Return active Dean staff:
						EXEC [dbo].[p_GetEmployeeList] 
							 @Dept		= 'DEAN'
							,@Type		= 'STAFF'
							,@Active	= 'Y'
						;
  
  Created:	        04/28/2016 by John Holmes

  Modified:
  	
**************************************************************************************************/

BEGIN
	
	-- For Testing
	-- DECLARE @Dept VARCHAR(6) = 'DEAN', @Type VARCHAR(6) = 'STAFF', @Active VARCHAR(1) = 'Y' ;
	
	SET NOCOUNT ON ;
	
	-- FAC
	IF @Type = 'FAC'
		SELECT
			 [GatorLinkID]
			,[Email]
		FROM
			[BaseViews].[dbo].[pv_WcbaPeople]
		WHERE
			[Department] = @Dept
			AND [UFPositionType] IN ('FAC')
			--AND [UFPositionCategory] != 'OPS'
			AND [ActiveFlag] = COALESCE(@Active, [ActiveFlag])
			AND [GatorLinkID] IS NOT NULL
			AND COALESCE([Email], '') != ''
		;

	-- STAFF
	IF @Type = 'STAFF'
		SELECT
			 [GatorLinkID]
			,[Email]
		FROM
			[BaseViews].[dbo].[pv_WcbaPeople]
		WHERE
			[Department] = @Dept
			AND [UFPositionType] = 'STAFF'
			AND [UFPositionCategory] != 'OPS'
			AND [ActiveFlag] = COALESCE(@Active, [ActiveFlag])
			AND [GatorLinkID] IS NOT NULL
			AND COALESCE([Email], '') != ''
		;

	-- OPS
	IF @Type = 'OPS'
		SELECT
			 [GatorLinkID]
			,[Email]
		FROM
			[BaseViews].[dbo].[pv_WcbaPeople]
		WHERE
			[Department] = @Dept
			AND [UFPositionType] != 'FAC'
			AND [UFPositionCategory] = 'OPS'
			AND [ActiveFlag] = COALESCE(@Active, [ActiveFlag])
			AND [GatorLinkID] IS NOT NULL
			AND COALESCE([Email], '') != ''
		;
    
END ;
