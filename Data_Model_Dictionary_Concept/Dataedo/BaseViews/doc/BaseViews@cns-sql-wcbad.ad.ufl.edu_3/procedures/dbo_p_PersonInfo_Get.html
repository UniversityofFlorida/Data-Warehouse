CREATE PROCEDURE [dbo].[p_PersonInfo_Get]

    @gl [varchar](70) = NULL
   ,@ufid [varchar](8) = NULL
   ,@name [varchar](44) = NULL

AS 

/**************************************************************************************************

  Name:		        p_PersonInfo_Get

  Description:	    Retrieves user information. 
                    If parameters are supplied, returns date from UF Directory.
                    If no parameters supplied, returns all active people in WCBA Directory.
  
  Input:            @gl     varchar(70)     OPTIONAL. GatorLink ID. e.g. "alberta".
                    
					@ufid   varchar(8)      OPTIONAL. UF Identifier. e.g. "02062630".

					@name   varchar(44)     OPTIONAL. Name fragment. e.g. "gator".

  Output:           On Success: A table of UFID, UFBusinessName and GatorlinkID.
                    On Failure: System error message.
  
  Notes:            Input parameter priority is GatorLink, UFID, Name.
                    I.E., If @gl is specified, @ufid and @name are ignored.
                    If @ufid is specified, @name is ignored
  
  Usage:            EXEC [dbo],[p_PersonInfo_Get] @gl, @ufid, @name
                    Examples:
                        EXEC [dbo].[p_PersonInfo_Get] @gl = 'alberta'
					    EXEC [dbo].[p_PersonInfo_Get] @ufid = '02062630'
					    EXEC [dbo].[p_PersonInfo_Get] @name = 'gator'
  
  Created:	        01/08/2015 by John Holmes

  Modified:         
  	
**************************************************************************************************/

BEGIN
	

	SET NOCOUNT ON;
	
    IF @gl IS NOT NULL 
        BEGIN
            SELECT
                 [UFID]
                ,[UFBusinessName]
                ,[GatorLinkID]
				,[Email]
            FROM
                [BaseViews].[dbo].[bv_UFDirPeople]
            WHERE
                [GatorLinkID] = @gl ;
        END ;
    ELSE IF @ufid IS NOT NULL 
        BEGIN
            SELECT
                 @ufid AS [UFID]
                ,[UFBusinessName]
                ,[GatorLinkID]
				,[Email]
            FROM
                [BaseViews].[dbo].[bv_UFDirPeople]
            WHERE
                [UFID] = @ufid ;
        END ;
    ELSE IF @name IS NOT NULL 
        BEGIN
            SELECT
                 [UFID]
                ,[UFBusinessName]
                ,[GatorLinkID]
				,[Email]
            FROM
                [BaseViews].[dbo].[bv_UFDirPeople]
            WHERE
                [UFBusinessName] LIKE ('%' + @name + '%') ;
        END ;
    ELSE 
        BEGIN
            SELECT
                 WCBA.UFID
                ,[People].[UFBusinessName]
                ,[People].[GatorLinkID]
				,[People].[Email]
            FROM
                [BaseViews].[dbo].[pv_WcbaPeople] AS [WCBA]
            LEFT JOIN 
                [BaseViews].[dbo].[bv_UFDirPeopleActive] AS [People]
                ON [People].[UFID] = [WCBA].[UFID]
            WHERE
                [WCBA].[ActiveFlag] = 'Y'
				AND [People].[UFBusinessName] IS NOT NULL ;
        END ;
END ;
