CREATE PROCEDURE [dbo].[p_snoopy_nonWCBInstructors_get]

AS 

/**************************************************************************************************

  Name:		        p_snoopy_nonWCBInstructors_get

  Description:	    GLIDs of instructors teaching WCB courses who are not, or never were, WCB employees
  
  Usage:            EXEC [dbo].[p_snoopy_nonWCBInstructors_get]
  
  Created:	        01/18/2017 by John Holmes

  Modified:			
  	
**************************************************************************************************/

BEGIN

	SET NOCOUNT ON ;

	DECLARE @term6 VARCHAR(6) = (SELECT Baseviews.dbo.fn_GetTerm('C', 6)) ;

	SELECT
		p.GatorLinkID
	FROM
		BaseViews.dbo.bv_UFSections AS s
	INNER JOIN
		BaseViews.dbo.bv_UFInstructorSection AS i
		ON s.Term6		= i.Term6
		AND	s.Section	= i.Section
	LEFT JOIN
		BaseViews.dbo.pv_wdir_people AS wcbp
		ON wcbp.UFID = i.InstructorUFID
	LEFT JOIN
		BaseViews.dbo.bv_UFDirPeople AS p
		ON i.InstructorUFID = p.UFID
	WHERE
		s.Term6 = @term6
		AND (
			wcbp.UFID IS NULL
			OR	wcbp.active = 0
		)
	GROUP BY
		p.GatorLinkID
	;

END ;
