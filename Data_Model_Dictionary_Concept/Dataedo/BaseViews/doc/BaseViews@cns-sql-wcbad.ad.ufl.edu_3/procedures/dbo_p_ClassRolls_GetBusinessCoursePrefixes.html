CREATE PROCEDURE [dbo].[p_ClassRolls_GetBusinessCoursePrefixes] 

    @Term6 VARCHAR(6)

AS 

/**************************************************************************************************

  Name:		        p_ClassRolls_GetBusinessCoursePrefixes

  Description:	    Creates a listing of all Business course prefixes for the specified term.
  
  Input:            @Term6 VARCHAR(6)   - A term with which to filter the output
  
  Usage:            EXEC [BaseViews].[dbo].[p_ClassRolls_GetBusinessCoursePrefixes] @Term6
                    Example: 
                        EXEC [BaseViews].[dbo].[p_ClassRolls_GetBusinessCoursePrefixes]
                            '201001' ;
  
  Orig Created:	    07/16/2010 by John Holmes in ClassRolls DB

  Modified:         03/15/2012 by John Holmes
                    Migrated to BaseViews DB
  	
**************************************************************************************************/

BEGIN

    SET NOCOUNT ON ;
    
    -- For testing
    --DECLARE @Term6 VARCHAR(6) = '201001' ;
    
    SELECT 
        DISTINCT LEFT([sec].[Course],3) AS [Prefix]
    FROM 
        [BaseViews].[dbo].[bv_UFSections] AS [sec]
    WHERE 
        [sec].[Term6] = @Term6
        AND (
            [sec].[DepartmentCode] LIKE '0117%'
            OR LEFT([sec].[Course],3) = 'SLS'
        ) ;

END ;
