CREATE PROCEDURE [dbo].[p_FERPA_FullAuditReport] 
	
AS 

/**************************************************************************************************

  Name:		        p_FERPA_FullAuditReport

  Description:	    Create a FERPA Certification report for all Active users in the WCBA Directory.
  
  Usage:            EXEC [BaseViews].[dbo].[p_FERPA_FullAuditReport]
  
  Created:	        10/12/2012 by John Holmes

  Modified:			04/09/2013 by John Holmes
                    Modified join clause order.
					
					01/04/2017 by John Holmes
					Reference bv_FERPACertifications_simple
  	
**************************************************************************************************/

BEGIN

    SET NOCOUNT ON ;
    
    SELECT 
         [p].[Department]
        ,[p].[UFID]
        ,[BaseViews].[dbo].[fn_GetUFBusinessNamePart]([People].[UFBusinessName], 'L') AS [LastName]
        ,[BaseViews].[dbo].[fn_GetUFBusinessNamePart]([People].[UFBusinessName], 'F') AS [FirstName]
        ,COALESCE([People].[Email],'') AS [Email]
        ,COALESCE(CAST([cert].[ExpirationDate] AS VARCHAR(10)),'') AS [ExpirationDate]
        ,COALESCE([cert].[DaysUntilExpiration],'') AS [DaysRemaining]
    FROM 
        [BaseViews].[dbo].[pv_WcbaPeople] AS [p]
    LEFT JOIN 
        [BaseViews].[dbo].[bv_FERPACertifications_simple] AS [cert]
        ON [cert].[UFID] = [p].[UFID]
    LEFT JOIN 
        [BaseViews].[dbo].[bv_UFDirPeople] AS [People] 
        ON [p].[UFID] = [People].[UFID]
    WHERE 
        [p].[ActiveFlag] = 'Y'
    ORDER BY 
         [p].[Department]
        ,[p].[LastName] ;
    
END ;
