CREATE PROCEDURE [dbo].[p_ClassRolls_GetClassRollListing] 

    @Prefix VARCHAR(3),
    @OrderBy VARCHAR(10)

AS 

/**************************************************************************************************

  Name:		        p_ClassRolls_GetClassRollListing

  Description:	    Creates an Class Roll listing (for the UI to display).
  
  Input:            @Prefix VARCHAR(3)  - A course prefix with which to filter the output
                    @OrderBy VARCHAR(12 - An ordering parameter [Course | Section | Instructor]
                                          
                    Note: The Class Rolls Application layer guarantees that @Prefix is either a 
                    non-empty string or NULL.  This procedures logic is predicated on that guarantee.
  
  Usage:            EXEC [BaseViews].[dbo].[p_ClassRolls_GetClassRollListing] @Prefix, @OrderBy
                    Example:
                        EXEC [BaseViews].[dbo].[p_ClassRolls_GetClassRollListing]
                             NULL
                            ,'Instructor' ;
  
  Orig Created:	    07/16/2010 by John Holmes in ClassRolls DB

  Modified:         03/15/2012 by John Holmes
                    Migrated to BaseViews DB
  	
**************************************************************************************************/

BEGIN

    SET NOCOUNT ON ;
    
    -------------------------------------------------
    --  Initialize working variables
    -------------------------------------------------
    
    -- For testing
    --DECLARE @Prefix VARCHAR(3) = NULL, @OrderBy VARCHAR(10) = 'Instructor' ;
    
    -- Current term
    DECLARE @currentTerm6 VARCHAR(6) = [BaseViews].[dbo].[fn_GetTerm]('C',6) ;
    
    -- Query parts
    DECLARE @SQL VARCHAR(MAX) = '' ;
    DECLARE @baseQuery VARCHAR(MAX) = '' ;
    DECLARE @whereClause VARCHAR(MAX) = '' ;
    DECLARE @orderClause VARCHAR(MAX) = '' ;
    
    SET @baseQuery = '
        SELECT
            [sec].[Course]
            ,[sec].[Section]
            ,COALESCE([is].[InstructorName],'''') as [InstructorName]
            ,[sec].[CourseTitle]
            ,[sec].[SpecialProgram]
        FROM
            [BaseViews].[dbo].[bv_UFSections] AS [sec]
        LEFT JOIN 
            [BaseViews].[dbo].[bv_UFInstructorSection] AS [is]
            ON [sec].[Section] = [is].[Section]
            AND [sec].[Term6] = [is].[Term6]
        WHERE
            [sec].[Cancelled] != ''Y''
            AND (
                [sec].[DepartmentCode] LIKE ''0117%''
                OR [sec].[Course] = ''SLS1102''
            )
            AND [sec].[Term6] = ''' + @currentTerm6 + '''' ;
    
    
    -------------------------------------------------
    --  Construct the query
    -------------------------------------------------
    
    -- Set WHERE clause
    IF (@Prefix IS NOT NULL)
        SET @whereClause = '
            AND LEFT([sec].[Course],3) = ''' + @Prefix + '''' ;
    ELSE
        SET @whereClause = '' ;
    
    -- Set ORDER BY clause
    SET @orderClause = 
    CASE 
        WHEN @OrderBy = 'Course' THEN '
        ORDER BY
            [sec].[Course]'
        WHEN @OrderBy = 'Section' THEN '
        ORDER BY
            [sec].[Section]'
        WHEN @OrderBy = 'Instructor' THEN '
        ORDER BY
            [is].[InstructorName]'
        ELSE ''                    
    END ;
    
    -- Construct the final query
    SET @SQL = @baseQuery + @whereClause + @orderClause ;
    
    
    -------------------------------------------------
    --  Return results
    -------------------------------------------------
    
    -- For testing
    --PRINT @SQL ;
    
    EXEC (@SQL) ;

    -------------------------------------------------
