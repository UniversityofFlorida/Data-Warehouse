CREATE PROCEDURE dbo.p_FERPA_Details_Get 

	 @ufid varchar(8) = NULL
	,@glid varchar(70) = NULL
	
AS 

/**************************************************************************************************

  Name:			p_FERPA_Details_Get

  Description:	Return FERPA Certification details
  
                PARAM NAME      DATATYPE        REQUIRED    NOTES
  Input:        @ufid			varchar(8)		No			UF identifier (UFID)
															NULL (default) - no restriction

				@glid			varchar(70)		No			GatorLink ID
															NULL (default) - no restriction
  
  Output:		A record set
  
  Usage:		EXEC [BaseViews].[dbo].[p_FERPA_Details_Get] @ufid, @glid ;
				Examples:
					
					-- Get info for Albert via ufid
					EXEC [BaseViews].[dbo].[p_FERPA_Details_Get] 
						@ufid = '00000542' ;
						
					-- Get info for Albert via glid
					EXEC [BaseViews].[dbo].[p_FERPA_Details_Get] 
						@glid = 'albert' ;

					-- Get info for all WCBA people
					EXEC [BaseViews].[dbo].[p_FERPA_Details_Get] ;

  
  Created:		04/24/2015 by John Holmes

  Modified:     01/04/2017 by John Holmes
				Reference bv_FERPACertifications_simple
					
				01/17/2017 by John Holmes
				Reference v_wdir_people
  					
**************************************************************************************************/

BEGIN
	
	SET NOCOUNT ON ;
	
    SELECT
		 f.UFID
		,p.GatorLinkID
		,p.UFBusinessName
		,p.Email
		,wp.active AS IsActiveWcbaDir
		,f.CertificationDate
		,f.ExpirationDate
		,f.DaysUntilExpiration
		,f.IsValid
		,f.Course
		,f.CourseTitle
    FROM
        dbo.bv_FERPACertifications_simple AS f
	JOIN
		dbo.bv_UFDirPeopleActive AS p
		ON f.UFID = p.UFID
	LEFT JOIN 
		dbo.pv_wdir_people AS wp
		ON wp.UFID = f.UFID
	WHERE 
		f.UFID = COALESCE(@ufid, f.UFID)
		AND p.GatorLinkID = COALESCE(@glid, p.GatorLinkID)
	;
    
END ;
