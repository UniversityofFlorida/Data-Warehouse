CREATE PROCEDURE [dbo].[p_FERPA_EventLogReport] 

     @testEmail VARCHAR(254) = NULL
	
AS 

/**************************************************************************************************

  Name:		        p_FERPA_EventLogReport

  Description:	    Email a report of todays expiration email recipients
                    
					PARAM NAME      DATATYPE		REQUIRED    NOTES
  Input:			@testEmail      varchar(254)	No			Email address for test mode
																NULL (default) = System "Reports" address
                    
  
  Output:           One or more email messages are sent.  
  Usage:            EXEC [BaseViews].[dbo].[p_FERPA_EventLogReport] @testEmail
                    Example:
                        Enable Test mode:
                        EXEC [BaseViews].[dbo].[p_FERPA_EventLogReport] 
                             @testEmail = 'jcholmes@ufl.edu' ;
  
  Created:	        05/20/2015 by John Holmes

  Modified:         02/13/2017 by John Holmes
					Add DMBTC as recipient

					09/26/2017 by John Holmes
					Handle null expriation date condition
  	
**************************************************************************************************/

BEGIN

    SET NOCOUNT ON ;
        
	-- For Testing
	--DECLARE @testEmail VARCHAR(254) = 'jcholmes@ufl.edu' ;
            
	--------------------- Declare working vars ---------------------
            
	DECLARE @date NVARCHAR(20) = DATENAME(MM, GETDATE()) + RIGHT(CONVERT(VARCHAR(12), GETDATE(), 107), 9) ;  -- Today's data formatted as Month DD, YYYY
	DECLARE @body NVARCHAR(MAX) = '' ;          -- Final body of email
	DECLARE @body1 NVARCHAR(MAX) = '' ;         -- Pt 1 of email body
	DECLARE @body2 NVARCHAR(MAX) = '' ;         -- Pt 2 of email body
	DECLARE @subjectText VARCHAR(MAX) = '' ;    -- Email subject line
	DECLARE @xml NVARCHAR(MAX) = '' ;           -- Report data
	DECLARE @reportRecip VARCHAR(256) = '' ;    -- Report recipient
            
            
	--------------------- Send report ---------------------
            
	-- Get report data 
	SET @xml = 
		CAST(
			(SELECT
				[Event] AS 'td'
				,''     -- Do not remove this line
			   ,[UFID] AS 'td'
				,''     -- Do not remove this line
			   ,[Name] AS 'td'
				,''     -- Do not remove this line
			   ,[Email] AS 'td'
				,''     -- Do not remove this line
			   ,[Course] + ' - ' + [CourseTitle] AS 'td'
				,''     -- Do not remove this line
			   ,CASE [Event]
					WHEN 'Updated' THEN CAST([CertDate] AS [varchar])
					ELSE '-------------'
				END AS 'td'
				,''     -- Do not remove this line
			   ,CASE [Event]
					WHEN 'Updated' THEN '-------------'
					ELSE COALESCE(CAST([ExpDate] AS [varchar]), 'no data')
				END AS 'td'
			FROM
				[BaseData].[bd].[t_FERPAEventLog]
			WHERE 
				[Datestamp] = CAST(GETDATE() AS [date])
			ORDER BY
				[Event]
			FOR XML 
				PATH('tr')
				,ELEMENTS 
			) AS NVARCHAR(MAX)
		) ;
                
	--Check for results
	IF LEFT(@xml, 3) = '<tr'  -- If was data on which to report
		BEGIN 
			-- Set up the HTML document
			SET @body1 = 
				'<html>
				<head>
					<style>
						body {
							margin-left:20px;
							font:10px Verdana, Geneva, Arial, Helvetica, sans-serif;
						}
						h1 {
							font:12px Verdana, Geneva, Arial, Helvetica, sans-serif;
							font-weight:bold;
							margin-bottom:5px;
						}
						h2 {
							font:10px Verdana, Geneva, Arial, Helvetica, sans-serif;
							font-weight:bold;
							margin-top:0px;
							padding-bottom:10px;
						}
						table {
							border:1px solid #AAAAAA;
						}
						th {
							padding:2px;
							text-align: left;
							font:10px Verdana, Geneva, Arial, Helvetica, sans-serif;
							font-weight:bold;
							border-bottom:1px solid #AAAAAA;
							border-right:1px solid #AAAAAA;
						}
						td {
    						vertical-align:top;
							text-align:left;
							border-bottom:1px solid #AAAAAA;
							border-right:1px solid #AAAAAA;
							padding:2px 2px 2px 2px;
							font:10px Verdana, Geneva, Arial, Helvetica, sans-serif;
						}
					</style>
				</head>
				<body>
					<h1>FERPA Event Log for ' + @date + '</h1>
					<table>
						<tr>
							<th>Event</th>
							<th>UFID</th>
							<th>Name</th>
							<th>Email</th>
							<th>Course</th>
							<th>Cert Date</th>
							<th>Exp Date</th>
						</tr>'  ;
                                
			SET @body2 = 
				'   </table>
				</body>
				</html>' ;
                        
			-- Construct the message body
			SET @body = @body1 + @xml + @body2 ;

			-- Construct the message subject and recipient
			SET @subjectText = 'FERPA Report - Event Log' ;
                    
			-- Determine report recipient
			--SET @reportRecip = COALESCE(@testEmail, [AdminTools].[dbo].[fn_GetSystemEmailAddress]('Reports')) ;
			SET @reportRecip =	CASE 
									WHEN @testEmail IS NULL 
										THEN [AdminTools].[dbo].[fn_GetSystemEmailAddress]('Reports') + ';' + [AdminTools].[dbo].[fn_GetSystemEmailAddress]('DMBTC')
									ELSE @testEmail
								END ;
                    
			-- Send the email
			EXEC [BaseViews].[dbo].[p_SendEmail]
				 @recipList = @reportRecip
				,@subjectText = @subjectText
				,@bodyText = @body
				,@bodyFormatType = 'HTML'
				,@profileName = 'FERPANotifications' ;
                        
		END ; -- End data to report
        
END ;   -- End p_FERPA_EventLogReport
