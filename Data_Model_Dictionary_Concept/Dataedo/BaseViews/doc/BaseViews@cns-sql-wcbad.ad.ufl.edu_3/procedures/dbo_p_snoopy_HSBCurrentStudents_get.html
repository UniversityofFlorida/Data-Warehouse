CREATE PROCEDURE [dbo].[p_snoopy_HSBCurrentStudents_get]

AS 

/**************************************************************************************************

  Name:		        p_snoopy_HSBCurrentStudents_get

  Description:	    GLIDs of undergraduate BA students registered for the current term
  
  Usage:            EXEC BaseViews.dbo.p_snoopy_HSBCurrentStudents_get
  
  Created:	        01/18/2017 by John Holmes

  Modified:			
  	
**************************************************************************************************/

BEGIN
	
	SET NOCOUNT ON ;

    SELECT
        p.GatorLinkID
    FROM
        BaseViews.dbo.bv_StudentCurrentRecord AS csr 
    LEFT JOIN 
        BaseViews.dbo.bv_UFDirPeopleActive AS p 
        ON csr.UFID = p.UFID
    WHERE
        csr.RegistrationCode = 'R'
        AND (
			csr.CurrentCollege = 'BA'
			OR csr.CurrentCollege2 = 'BA'
			OR csr.CurrentCollege3 = 'BA'
			OR csr.CurrentCollege4 = 'BA'
			OR csr.CurrentCollege5 = 'BA'
		)
        AND csr.CurrentClass BETWEEN '0' AND '4'
    GROUP BY 
		p.GatorLinkID
    ;

END ;
