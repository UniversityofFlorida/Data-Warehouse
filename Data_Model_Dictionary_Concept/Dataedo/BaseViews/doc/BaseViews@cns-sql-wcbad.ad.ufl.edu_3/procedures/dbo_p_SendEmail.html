CREATE PROCEDURE [dbo].[p_SendEmail] 
     
	 @recipList [varchar](MAX) = '' 
	,@subjectText [nvarchar](255) = ''
	,@bodyText [varchar](MAX) = ''
	,@bodyFormatType [varchar](20) = NULL
	,@copyRecipList [varchar](MAX) = ''
	,@profileName [sysname] = NULL
    
AS 

/**************************************************************************************************

  Name:		        p_SendEmail

  Description:	    Send an email.
                    See BOL for sp_send_dbmail for full parameter list
  
  Input:            @recipList      VARCHAR(MAX)    List of recipient email addresses (required)
                    @subjectText    VARCHAR(255)    Email subject line (required)
                    @bodyText       VARCHAR(MAX)    Email body text (required)
                    @bodyFormatType VARCHAR(20)     ['TEXT' (default) | 'HTML'] (optional)
                    @copyRecipList  VARCHAR(MAX)    List of cc recipient email addresses (optional)
                    @profileName    sysname         DBMail profile to use, if other than default (optional)
  
  Usage:            EXEC [BaseViews].[dbo].[p_SendEmail] 
                        'jcholmes@ufl.edu',

  Modified:         11/21/2012 by John Holmes
                    Modified error msg.
  	
**************************************************************************************************/

SET NOCOUNT ON ;

----------------------------------------------------------------

BEGIN

    IF (
        LEN(@recipList) > 0 AND 
        LEN(@subjectText) > 0 AND 
        LEN(@bodyText) > 0
    )
    BEGIN	
        EXEC msdb.dbo.sp_send_dbmail 
            @recipients = @recipList,
            @subject = @subjectText,
            @body = @bodyText, 
            @body_format = @bodyFormatType,
            @copy_recipients = @copyRecipList, 
            @profile_name = @profileName ;
    END ;
    ELSE 
        SELECT 'p_SendEmail: Required input parameters are missing.' ;
END ;
