CREATE PROCEDURE [dbo].[p_ClassRolls_GetClassRollData] 

	@Course VARCHAR(8) = NULL
	,@Section VARCHAR(4) = NULL
	,@AdditionalInfo BIT = 0
	,@Term6 VARCHAR(6) = NULL
AS 

/**************************************************************************************************

  Name:		        p_ClassRolls_GetClassRollData

  Description:	    Creates an Class Roll report.
  
  Input:            @Course VARCHAR(8)  - Default: NULL.    The course upon which to report.
                    @Section VARCHAR(4) - Default: NULL.    The section upon which to report
                    @AdditionalInfo BIT - Default: 0.       The report type
                                          0 = Base report: Term, UFID, Name, Section, Grade, PrivacyFlag
                                          1 = Base report + Current College and Major, Email, Gatorlink ID
                    @Term6 VARCHAR96)   - Default: NULL.    The term upon which to report
                                          NULL = current term
                                          
                    Note: The calling application is responsible for guaranteeing that one, and ONLY 
                    one of the Course and Section input parameters will be NULL.  This procedure's 
                    logic is predicated on this guarantee.
  
  Usage:            EXEC [BaseViews].[dbo].[p_ClassRolls_GetClassRollData] @Course, @Section, @AdditionalInfo, @Term6
                    Example:
                        EXEC [BaseViews].[dbo].[p_ClassRolls_GetClassRollData]
                             'BUL4310'
                            ,'8143'
                            ,1
                            ,NULL ;
  
  Orig Created:	    07/14/2010 by John Holmes in ClassRolls DB

  Modified:         09/08/2010 by John Holmes
                    Add Term6 input parameter
                    
                    09/09/2010 by John Holmes
                    Add Term6 to output
  	
  					10/05/2010 by William McCombie
  					Adjusted course search as course suffix (e.g. ACG4352C) was getting truncated
  					
  					03/15/2012 by John Holmes
                    Migrated to BaseViews DB
  					
**************************************************************************************************/

BEGIN

    SET NOCOUNT ON 
    
    /************************************************
      Initialize working variables
    ************************************************/
    
    -- For testing
    --DECLARE @Course VARCHAR(7) = 'BUL4310', @AdditionalInfo BIT = 1, @Section VARCHAR(4) = '8143', @Term6 VARCHAR(6) = NULL ;
    
    
    -- Term validation and calculation
    SET @Term6 = 
        CASE 
            WHEN LEN(@Term6) != 6 THEN [BaseViews].[dbo].[fn_ConvertTerm](@Term6,6)
            WHEN @Term6 IS NULL THEN [BaseViews].[dbo].[fn_GetTerm]('C',6) 
            ELSE @Term6
        END ;
        
    -- For testing
    --SELECT @Term6 ;
    
    
    -- Calculate isCurrentTerm flag
    DECLARE @isCurrentTerm BIT =
        CASE 
            WHEN @Term6 = [BaseViews].[dbo].[fn_GetTerm]('C',6) THEN 1
            ELSE 0
        END ;
        
    -- For testing
    --SELECT @isCurrentTerm ;
    
    
    -- Table of sections upon which to report.  Restrict to:
    --  1. Business sections or SLS1102
    --  2. Non-cancelled sections
    DECLARE @sectionList TABLE(
        [Term6] VARCHAR(6)
        ,[Section] VARCHAR(4)
    ) ;
    IF (@Course IS NOT NULL)
        BEGIN
            INSERT INTO @sectionList(
                [Term6]
                ,[Section]
            )
            SELECT  -- All (Business) sections of a course
                [sec].[Term6]
                ,[sec].[Section]
            FROM 
                [BaseViews].[dbo].[bv_UFSections] AS [sec]
            WHERE 
                [sec].[Course] = @Course
                AND (
                    [sec].[DepartmentCode] LIKE '0117%' -- Business sections
                    OR [sec].[Course] = 'SLS1102'
                )
                AND [sec].[Cancelled] != 'Y'
                AND [sec].[Term6] = @Term6 ;
        END ;
    ELSE 
        BEGIN
            INSERT INTO @sectionList(
                [Term6]
                ,[Section]
            )
            VALUES(
                @Term6
                ,@Section
            ) ;
        END ;
        
    -- For testing
    --SELECT * FROM @sectionList
    
    
    /************************************************
      Get data
    ************************************************/
    
    -- In-memory table to hold results
    
    DECLARE @results TABLE(
        [Section] VARCHAR(4)
        ,[UFID] VARCHAR(8)
        ,[Name] VARCHAR(44)
        ,[Grade] VARCHAR(1)
        ,[S/U Option] VARCHAR(1)
        ,[Major] VARCHAR(3)
        ,[College] VARCHAR(2)
        ,[GatorLink] VARCHAR(70)
        ,[Email] VARCHAR(254)
        ,[Privacy Flag] VARCHAR(14)
    ) ;
    
    IF @isCurrentTerm = 1
        BEGIN 
            INSERT INTO @results(
                [Section]
                ,[UFID]
                ,[Name]
                ,[Grade]
                ,[S/U Option]
                ,[Major]
                ,[College]
                ,[GatorLink]
                ,[Email]
                ,[Privacy Flag]
            )
            SELECT
                [s].[Section] AS [Section]
                ,[sr].[UFID] AS [UFID]
                ,[p].[UFBusinessName] AS [Name]
                ,'' AS [Grade]
                ,'' AS [S/U OPTION]
                ,[scr].[CurrentMajor] AS [Major]
                ,[scr].[CurrentCollege] AS [College]
                ,[p].[GatorLinkID] AS [GatorLink]
                ,[p].[Email] AS [Email]
                ,CASE [p].[PublishFlag] 
                    WHEN 'Y' THEN ''
                    ELSE 'DO NOT PUBLISH'
                 END AS [Privacy Flag]
            FROM 
                @sectionList AS [s]
            LEFT JOIN
                [BaseViews].[dbo].[bv_StudentRegistration] AS [sr]
                ON [s].[Section] = [sr].[Section]
                AND [s].[Term6] = [sr].[Term6]
            LEFT JOIN
                [BaseViews].[dbo].[bv_UFDirPeople] AS [p]
                ON [sr].[UFID] = [p].[UFID]
            LEFT JOIN
                [BaseViews].[dbo].[bv_StudentCurrentRecord] AS [scr]
                ON [sr].[UFID] = [scr].[UFID]
            WHERE
                -- Remove students who have withdrawn
                [sr].[Grade] NOT IN ('W','ZW') ;
        END ;
    ELSE
        BEGIN 
            INSERT INTO @results(
                [Section]
                ,[UFID]
                ,[Name]
                ,[Grade]
                ,[S/U Option]
                ,[Major]
                ,[College]
                ,[GatorLink]
                ,[Email]
                ,[Privacy Flag]
            )
            SELECT
                [s].[Section] AS [Section]
                ,[sr].[UFID] AS [UFID]
                ,[p].[UFBusinessName] AS [Name]
                ,'' AS [Grade]
                ,'' AS [S/U OPTION]
                ,[scr].[CurrentMajor] AS [Major]
                ,[scr].[CurrentCollege] AS [College]
                ,[p].[GatorLinkID] AS [GatorLink]
                ,[p].[Email] AS [Email]
                ,CASE [p].[PublishFlag] 
                    WHEN 'Y' THEN ''
                    ELSE 'DO NOT PUBLISH'
                 END AS [Privacy Flag]
            FROM 
                @sectionList AS [s]
            LEFT JOIN
                [BaseViews].[dbo].[hv_StudentRegistration] AS [sr]  -- History view
                ON [s].[Section] = [sr].[Section]
                AND [s].[Term6] = [sr].[Term6]
            LEFT JOIN
                [BaseViews].[dbo].[bv_UFDirPeople] AS [p]
                ON [sr].[UFID] = [p].[UFID]
            LEFT JOIN
                [BaseViews].[dbo].[hv_StudentCurrentRecord] AS [scr]    -- History view
                ON [sr].[UFID] = [scr].[UFID]
                AND [s].[Term6] = [scr].[Term6]
            WHERE
                -- Remove students who have withdrawn
                [sr].[Grade] NOT IN ('W','ZW') ;
        END ;
        
    -- For testing
    --SELECT * FROM @results ORDER BY [Name] ;
    
    
    /************************************************
      Return report
    ************************************************/
    
    IF (@AdditionalInfo = 0)    -- Return basic report
        BEGIN
            SELECT
                @Term6 AS [Term]
                ,[Section]
                ,[UFID]
                ,[Name]
                ,[Grade]
                ,[S/U Option]
            FROM 
                @results
            ORDER BY 
                [Name] ;
        END ;
    ELSE    -- Return additional info
        BEGIN 
            SELECT
                @Term6 AS [Term]
                ,[Section]
                ,[UFID]
                ,[Name]
                ,[Grade]
                ,[S/U Option]
                ,[Major]
                ,[College]
                ,[GatorLink]
                ,[Email]
                ,[Privacy Flag]
            FROM 
                @results
            ORDER BY 
                [Name] ;
        END  ;

END ;
