CREATE PROCEDURE [dbo].[pp_WcbaStaffEmailList] 

	@FacStaffFlag int = NULL
	
AS 

/**************************************************************************************************

  Name:		        pp_WcbaStaffEmailList

  Description:	    Return the email addresses for a specified group of employees
					(Stub for warringtonDirectory.dbo.fn_listServ_members_get)

  Input:		    @FacStaffFlag - Determines group of employees as follows
					Input	Description				listServName		listServID
			        1		All faculty and staff	wcbafacstaff-L		8
			        2		Staff					wcbastaff-L			6
			        3		Faculty					wcbafaculty-L		7
			        4		Graduate Faculty		wcbagradfac-L		9
			        5		Other Faculty			wcbaotherfac-L		10
			        6		CBAInfo					cbaAInfo-L			11
  
  Usage:            EXEC [BaseViews].[dbo].[pp_WcbaStaffEmailList] 2
  
  Created:	        08/04/2015 by John Holmes

  Modified:
  	
**************************************************************************************************/

BEGIN
	
	SET NOCOUNT ON ;

	-- For testing
	--DECLARE @FacStaffFlag int = 2 ;

	-- Working variables
	DECLARE @listServID [integer] = NULL ;
	DECLARE @ufids AS TABLE(
		[ufid] [varchar](8)
	) ;

	-- Convert input
	SET @listServID = 
		CASE @FacStaffFlag
			WHEN 1 THEN 8
			WHEN 2 THEN 6
			WHEN 3 THEN 7
			WHEN 4 THEN 9
			WHEN 5 THEN 10
			WHEN 6 THEN 11
		END ;
	
	-- Get ufid list
    INSERT INTO @ufids([ufid])
    SELECT [UFID] FROM  [warringtonDirectory].[dbo].[fn_listServ_members_get](@listServID) ;

	-- Get emails
	SELECT 
		[e].[email]
	FROM 
		@ufids AS [u]
	JOIN 
		warringtonDirectory.[dbo].v_email AS [e]
		ON [u].[ufid] = [e].[UFID]
	WHERE 
		[e].[typeID] = 1
	;
    
END
