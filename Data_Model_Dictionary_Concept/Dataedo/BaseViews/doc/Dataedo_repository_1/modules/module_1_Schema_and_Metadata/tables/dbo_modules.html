create TRIGGER [trg_modules_Modify]
   ON [modules]
   AFTER INSERT,UPDATE
AS 
BEGIN
	UPDATE [modules]
	SET  last_modification_date = GETDATE(),
		 modified_by = suser_sname()
	WHERE module_id IN (SELECT DISTINCT module_id FROM Inserted)
END
