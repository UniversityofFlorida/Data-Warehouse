CREATE TRIGGER [trg_procedures_Modify]
   ON  [procedures] 
   AFTER INSERT, UPDATE
AS 
BEGIN
	UPDATE [procedures]
	SET last_modification_date = GETDATE(),
		modified_by = suser_sname()
	WHERE procedure_id IN (SELECT DISTINCT procedure_id FROM Inserted)
END
