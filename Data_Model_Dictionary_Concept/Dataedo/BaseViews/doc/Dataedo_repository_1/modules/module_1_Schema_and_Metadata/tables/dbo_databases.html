CREATE TRIGGER [trg_databases_Modify]
   ON  [databases] 
   AFTER INSERT,UPDATE
AS 
BEGIN
	UPDATE [databases]
	SET  last_modification_date = GETDATE(),
		 modified_by = suser_sname()
	WHERE database_id IN (SELECT DISTINCT database_id FROM Inserted)
END
